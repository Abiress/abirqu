import math
import numpy as np
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Patent analysis
# ============================================

print("Quantum Intellectual Property:")
print("=" * 50)

def quantum_patent_analysis(claims, citations):
    """Quantum patent analysis."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QPatent")
    
    # Encode claims
    circuit.ry(0, claims[0] * math.pi)
    circuit.ry(1, claims[1] * math.pi)
    
    # Encode citations
    circuit.ry(2, citations[0] * math.pi)
    circuit.ry(3, citations[1] * math.pi)
    
    # Analysis
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

print("Quantum Intellectual Property:")
print("-" * 50)

patent_types = [
    ("Utility", [0.8, 0.2], [0.7, 0.3]),
    ("Design", [0.5, 0.5], [0.4, 0.6]),
    ("Provisional", [0.6, 0.4], [0.5, 0.5]),
    ("Continuation", [0.4, 0.6], [0.6, 0.4]),
]

for name, claims, citations in patent_types:
    circuit = quantum_patent_analysis(claims, citations)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 2: IP portfolio management
# ============================================

print("\n\nIP Portfolio Management:")
print("-" * 50)

    circuit.cnot(1, 3)

    circuit.measure_all()
    return circuit

portfolio_scenarios = [
    ("Defensive", [0.3, 0.7], [0.4, 0.6]),
    ("Offensive", [0.7, 0.3], [0.8, 0.2]),
    ("Licensing", [0.5, 0.5], [0.6, 0.4]),
    ("Cross-Licensing", [0.6, 0.4], [0.5, 0.5]),

for name, assets, strategy in portfolio_scenarios:
    circuit = quantum_ip_portfolio(assets, strategy)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 3: Infringement detection
# ============================================

print("\n\nInfringement Detection:")
print("-" * 50)

def quantum_infringement(patent, product):
    """Quantum infringement detection."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QInfringement")
    
    # Encode patent
    circuit.ry(0, patent[0] * math.pi)

infringement_scenarios = [
