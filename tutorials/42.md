import math
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Magic state concept
# ============================================

print("Magic State Distillation:")
print("=" * 50)

# T gate magic state: |T⟩ = (|0⟩ + e^{iπ/4}|1⟩)/√2
print("Magic State |T⟩:")
print("  |T⟩ = (|0⟩ + e^{iπ/4}|1⟩)/√2")
print("  Enables fault-tolerant T gate")
print("  Essential for universal quantum computing")

# ============================================
# Part 2: 15-to-1 distillation
# ============================================

print("\n\n15-to-1 Distillation Protocol:")
print("-" * 50)

def magic_state_distillation_circuit():
    """Build 15-to-1 magic state distillation circuit."""
    # 15 input noisy magic states → 1 clean magic state
    circuit = Circuit(16, name="MagicDistill")
    
    # Prepare noisy magic states (simplified)
    for i in range(15):
        circuit.h(i)
        circuit.s(i)
        circuit.t(i)
    
    # Distillation via error detection
    # (Simplified - real protocol is complex)
    for i in range(15):
        circuit.cnot(i, 15)
    
    # Measure syndrome
    circuit.measure_all()
    
    return circuit

circuit = magic_state_distillation_circuit()
result = QuantumRun(circuit, shots=100)

print(f"Distillation result: {result.counts}")
print("Successful distillation: syndrome = 0")

# ============================================
# Part 3: Resource estimation
# ============================================

print("\n\nResource Estimation:")
print("-" * 50)

def estimate_distillation_overhead(target_error, initial_error):
    """Estimate resources for magic state distillation."""
    # Each round reduces error by factor ~10
    rounds = 0
    error = initial_error
    
    while error > target_error:
        error *= 0.1  # 10x improvement per round
        rounds += 1
    
    # Resources grow exponentially with rounds
    overhead = 15 ** rounds
    
    return rounds, overhead

target = 1e-12
initial = 0.01

rounds, overhead = estimate_distillation_overhead(target, initial)
print(f"Target error rate: {target:.0e}")
print(f"Initial error rate: {initial}")
print(f"Distillation rounds: {rounds}")
print(f"Resource overhead: {overhead}x")

print("\nMagic state distillation is resource-intensive but essential!")

## Key Concepts

### Magic States
- States that enable non-Clifford gates
- T gate requires magic state for fault tolerance
- Universal quantum computing needs magic states

### Distillation Protocol
1. Prepare many noisy magic states
2. Perform error detection
3. Keep states that pass syndrome check
4. Repeat until desired fidelity achieved

### Applications
- **Fault-tolerant T gate**: Universal computation
- **Quantum algorithms**: Shor, Grover, etc.
- **Quantum simulation**: Chemical and physical systems
