## Step-by-Step Code

```python
import math
from abirqu import Circuit, H, X, Y, Z
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: 3-qubit bit-flip code
# ============================================

print("Quantum Error Correction Codes:")
print("=" * 50)

def bit_flip_encode(circuit, data_qubit=0):
    """Encode using 3-qubit bit-flip code."""
    circuit.cnot(data_qubit, 1)
    circuit.cnot(data_qubit, 2)
    return circuit

def bit_flip_decode(circuit):
    """Decode and correct bit-flip errors."""
    # Syndrome measurement
    circuit.cnot(0, 1)
    circuit.cnot(0, 2)
    return circuit

print("3-Qubit Bit-Flip Code:")
print("-" * 50)

# Encode
circuit = Circuit(3, name="BitFlip")
circuit.h(0)  # Initial state
bit_flip_encode(circuit)

# Introduce error
circuit.x(1)  # Bit-flip on qubit 1

# Decode
bit_flip_decode(circuit)
circuit.measure_all()

result = QuantumRun(circuit, shots=100)
print(f"After error and correction: {result.counts}")
print("Error corrected: qubit 1 was flipped, syndrome detected!")

# ============================================
# Part 2: 3-qubit phase-flip code
# ============================================

print("\n\n3-Qubit Phase-Flip Code:")
print("-" * 50)

def phase_flip_encode(circuit, data_qubit=0):
    """Encode using 3-qubit phase-flip code."""
    circuit.h(data_qubit)
    circuit.cnot(data_qubit, 1)
    circuit.cnot(data_qubit, 2)
    circuit.h(data_qubit)
    circuit.h(1)
    circuit.h(2)
    return circuit

circuit = Circuit(3, name="PhaseFlip")
circuit.h(0)  # Initial state
phase_flip_encode(circuit)

# Introduce phase error
circuit.z(1)  # Phase-flip on qubit 1

# Syndrome measurement (in X basis)
circuit.h(0)
circuit.h(1)
circuit.h(2)
circuit.cnot(0, 1)
circuit.cnot(0, 2)
circuit.h(0)
circuit.h(1)
circuit.h(2)

circuit.measure_all()

result = QuantumRun(circuit, shots=100)
print(f"After phase error: {result.counts}")

# ============================================
# Part 3: Shor's 9-qubit code
# ============================================

print("\n\nShor's 9-Qubit Code:")
print("-" * 50)

def shor_encode(circuit, data_qubit=0):
    """Encode using Shor's 9-qubit code."""
    # Phase-flip encoding
    circuit.cnot(data_qubit, 3)
    circuit.cnot(data_qubit, 6)
    
    # Bit-flip encoding for each block
    for i in [0, 3, 6]:
        circuit.cnot(i, i + 1)
        circuit.cnot(i, i + 2)
    
    return circuit

circuit = Circuit(9, name="Shor")
circuit.h(0)  # Initial state
shor_encode(circuit)

# Introduce error on qubit 4
circuit.x(4)

print("Shor's 9-qubit code can correct any single-qubit error!")
print(f"Circuit depth: {circuit.depth()}")
print(f"Number of qubits: {9}")
```

## Key Concepts

### Error Correction Principle
1. **Encode** logical qubit in multiple physical qubits
2. **Syndrome measurement** detects errors without destroying information
3. **Correct** errors based on syndrome
4. **Decode** to recover original state

### Code Properties
- **Distance**: Number of errors correctable
- **Rate**: Ratio of logical to physical qubits
- **Threshold**: Maximum physical error rate for fault tolerance

### Applications
- **Fault-tolerant quantum computing**: Reliable computation
- **Quantum memories**: Long-term storage
- **Quantum communication**: Reliable transmission
- **Quantum networks**: Distributed quantum computing
