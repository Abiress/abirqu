# Part 1: Toric code

print("Advanced Quantum Error Correction:")

def toric_code_circuit(L=3):
    """Build toric code on L×L lattice."""
    # Number of qubits: 2L² (edges)
    num_qubits = 2 * L * L
    circuit = Circuit(num_qubits, name="ToricCode")
    
    # Initialize in superposition
    for i in range(num_qubits):
        circuit.h(i)
    
    # X stabilizers (vertices)
    for i in range(L):
        for j in range(L):
            v = i * L + j
            circuit.cnot(v, (v + 1) % num_qubits)
            circuit.cnot(v, (v + L) % num_qubits)
    
    # Z stabilizers (faces)
    for i in range(L):
        for j in range(L):
            f = L * L + i * L + j
            circuit.cnot(f, (f + 1) % num_qubits)
            circuit.cnot(f, (f + L) % num_qubits)
    
    return circuit

print("Toric Code:")
print("-" * 50)

circuit = toric_code_circuit(L=3)
print(f"  Lattice size: 3×3")
print(f"  Number of qubits: {circuit.num_qubits}")
print(f"  Distance: 3 (can correct 1 error)")

def color_code_circuit():
    """Build 4.8.8 color code."""

    circuit = Circuit(7, name="ColorCode")
    
    # Data qubits
    for i in range(7):
        circuit.h(i)
    
    # Stabilizers (simplified)
    circuit.cnot(0, 1)
    circuit.cnot(1, 2)
    circuit.cnot(2, 0)
    
    circuit.cnot(3, 4)
    circuit.cnot(4, 5)
    circuit.cnot(5, 3)
    
    return circuit

## Key Concepts

### Topological Codes
- **Toric code**: On torus, topologically protected
- **Color code**: On trivalent lattice, transversal gates
- **Surface code**: Planar version of toric code

### LDPC Codes
- **Low-density parity-check**: Sparse parity matrix
- **Constant rate**: k/n constant as n grows
- **Efficient decoding**: Belief propagation

### Decoding
- **MWPM**: Optimal but slow
- **Union-Find**: Fast, near-optimal
- **Neural**: Fast inference after training

### Applications
- **Fault-tolerant computing**: Reliable quantum computers
- **Quantum memories**: Long-term storage
- **Quantum communication**: Error-resistant transmission
