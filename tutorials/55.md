```python
import math
import random
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Quantum policy gradient
# ============================================

print("Advanced Quantum RL:")
print("=" * 50)

def quantum_policy_gradient(state, params):
    """Quantum policy using parameterized circuit."""
    circuit = Circuit(3, name="PolicyGrad")
    
    # Encode state
    for i, s in enumerate(state[:3]):
        circuit.ry(i, s * math.pi)
    
    # Policy network
    circuit.ry(0, params[0])
    circuit.ry(1, params[1])
    circuit.ry(2, params[2])
    
    # Entanglement
    circuit.cnot(0, 1)
    circuit.cnot(1, 2)
    
    circuit.measure_all()
    return circuit

def get_action(state, params):
    """Get action from quantum policy."""
    circuit = quantum_policy_gradient(state, params)
    result = QuantumRun(circuit, shots=100)
    
    # Convert to action probabilities
    probs = {}
    for bitstring, count in result.counts.items():
        action = int(bitstring, 2) % 4
        probs[action] = probs.get(action, 0) + count / 100
    
    # Sample action
    actions = list(probs.keys())
    weights = list(probs.values())
    return random.choices(actions, weights=weights)[0]

# ============================================
# Part 2: Quantum value function
# ============================================

print("\nQuantum Value Function:")
print("-" * 50)

def quantum_value_function(state, params):
    """Estimate state value using quantum circuit."""
    circuit = Circuit(2, name="ValueFunc")
    
    # Encode state
    circuit.ry(0, state[0] * math.pi)
    circuit.ry(1, state[1] * math.pi)
    
    # Value estimation
    circuit.ry(0, params[0])
    circuit.cnot(0, 1)
    circuit.ry(1, params[1])
    
    circuit.measure_all()
    return circuit

# Train value function
params_v = [0.0, 0.0]
for _ in range(20):
    state = [random.random(), random.random()]
    target = random.random()  # Simplified
    
    circuit = quantum_value_function(state, params_v)
    result = QuantumRun(circuit, shots=100)
    
    value = result.probabilities.get('1', 0.5)
    error = value - target
    
    params_v[0] -= 0.1 * error
    params_v[1] -= 0.1 * error

print(f"Trained value function parameters: {params_v}")

# ============================================
# Part 3: Multi-agent quantum RL
# ============================================

print("\n\nMulti-Agent Quantum RL:")
print("-" * 50)

class QuantumAgent:
    """Quantum RL agent."""
    
    def __init__(self, agent_id):
        self.id = agent_id
        self.params = [0.0, 0.0, 0.0]
    
    def get_action(self, state):
        circuit = Circuit(2, name=f"Agent{self.id}")
        circuit.ry(0, state[0] * math.pi)
        circuit.ry(1, self.params[0])
        circuit.cnot(0, 1)
        circuit.measure_all()
        
        result = QuantumRun(circuit, shots=100)
        return int(max(result.counts, key=result.counts.get)[0])

# Create agents
agents = [QuantumAgent(i) for i in range(3)]

print("Multi-agent setup:")
for agent in agents:
    state = [random.random(), random.random()]
    action = agent.get_action(state)
    print(f"  Agent {agent.id}: State={state[:2]}, Action={action}")

print("\nApplications:")
print("  - Game playing")
print("  - Robotics coordination")
print("  - Traffic optimization")
print("  - Resource allocation")
```

- Scalable to many agents
