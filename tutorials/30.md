## Step-by-Step Code

```python
import math
import numpy as np
from abirqu import Circuit, H, X, CNOT, Ry, Rz, Measure
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Molecular Hamiltonian
# ============================================

print("Advanced VQE for Chemistry:")
print("=" * 50)

# H2O molecule (simplified 2-qubit model)
# In reality, needs 14+ qubits for minimal basis
hamiltonian = {
    'II': -0.81261,
    'IZ': 0.17120,
    'ZI': 0.17120,
    'ZZ': -0.22279,
    'XX': 0.04532,
    'YY': 0.04532,
}

print("H2O Hamiltonian (simplified):")
for term, coeff in hamiltonian.items():
    print(f"  {term}: {coeff:.5f}")

# ============================================
# Part 2: UCCSD Ansatz
# ============================================

print("\n\nUCCSD Ansatz:")
print("-" * 50)

def build_uccsd_ansatz(num_qubits, params):
    """Build UCCSD (Unitary Coupled Cluster Singles and Doubles) ansatz."""
    circuit = Circuit(num_qubits, name="UCCSD")
    
    # Hartree-Fock reference state |01⟩ (one electron in orbital 1)
    circuit.x(0)
    
    # Singles excitations
    circuit.ry(0, params[0])
    circuit.cnot(0, 1)
    
    # Doubles excitations
    circuit.ry(0, params[1])
    circuit.cnot(0, 1)
    circuit.cnot(1, 0)
    
    # Additional correlation
    circuit.ry(0, params[2])
    circuit.ry(1, params[3])
    
    return circuit

# ============================================
# Part 3: VQE optimization
# ============================================

print("\nVQE Optimization:")
print("-" * 50)

def compute_hamiltonian_expectation(params, hamiltonian, shots=1000):
    """Compute expectation value of Hamiltonian."""
    circuit = build_uccsd_ansatz(2, params)
    
    energy = 0
    
    # Measure each Pauli term
    for term, coeff in hamiltonian.items():
        measure_circuit = circuit.copy()
        
        # Apply basis rotations for measurement
        if term[0] == 'X':
            measure_circuit.h(0)
        elif term[0] == 'Y':
            measure_circuit.s(0)
            measure_circuit.h(0)
        
        if term[1] == 'X':
            measure_circuit.h(1)
        elif term[1] == 'Y':
            measure_circuit.s(1)
            measure_circuit.h(1)
        
        measure_circuit.measure([0, 1])
        
        result = QuantumRun(measure_circuit, shots=shots)
        
        # Compute expectation
        expectation = 0
        for bitstring, count in result.counts.items():
            # Map bits to ±1
            val = 1
            if bitstring[0] == '1':
                val *= -1
            if bitstring[1] == '1':
                val *= -1
            expectation += val * count / shots
        
        energy += coeff * expectation
    
    return energy

# Optimize
print("Optimizing VQE:")
best_energy = float('inf')
best_params = None

for _ in range(100):
    params = np.random.uniform(-0.5, 0.5, 4)
    energy = compute_hamiltonian_expectation(params, hamiltonian)
    
    if energy < best_energy:
        best_energy = energy
        best_params = params.copy()
        if _ % 20 == 0:
            print(f"  Iteration {_}: Energy = {energy:.6f} Ha")

print(f"\nFinal energy: {best_energy:.6f} Ha")
print(f"Exact H2O energy (2-qubit): ≈ -0.81261 Ha")
```

## Key Concepts

### UCCSD Ansatz
- **Singles**: Single electron excitations
- **Doubles**: Double electron excitations
- Physically motivated, accurate for many systems

### Molecular Simulation Pipeline
1. **Encode** molecular orbitals as qubits
2. **Prepare** trial wavefunction (UCCSD)
3. **Measure** Hamiltonian expectation
4. **Optimize** parameters to minimize energy

### Applications
- **Drug discovery**: Molecular properties
- **Materials science**: Catalyst design
- **Battery research**: Electrolyte optimization
- **Climate science**: Carbon capture molecules
