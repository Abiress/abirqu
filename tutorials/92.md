```python
import math
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Quantum repeater
# ============================================

print("Expert Quantum Communication (Part 2):")
print("=" * 50)

def quantum_repeater_circuit(segment_length=4):
    """Quantum repeater for long-distance communication."""
    num_qubits = 2 * segment_length + 1
    circuit = Circuit(num_qubits, name="Repeater")
    
    # Create entanglement in each segment
    for i in range(segment_length):
        circuit.h(2 * i)
        circuit.cnot(2 * i, 2 * i + 1)
    
    # Entanglement swapping
    for i in range(segment_length - 1):
        circuit.cnot(2 * i + 1, 2 * (i + 1))
        circuit.h(2 * i + 1)
    
    circuit.measure_all()
    return circuit

print("Quantum Repeater:")
print("-" * 50)

circuit = quantum_repeater_circuit(segment_length=4)
result = QuantumRun(circuit, shots=1000)

print(f"  Segments: 4")
print(f"  Total qubits: {circuit.num_qubits}")
print(f"  End-to-end entanglement: {result.counts}")

# ============================================
# Part 2: Entanglement distillation
# ============================================

print("\n\nEntanglement Distillation:")
print("-" * 50)

def distillation_circuit(num_pairs=4):
    """Distill high-fidelity entanglement."""
    num_qubits = 2 * num_pairs + 1
    circuit = Circuit(num_qubits, name="Distill")
    
    # Create noisy Bell pairs
    for i in range(num_pairs):
        circuit.h(2 * i)
        circuit.cnot(2 * i, 2 * i + 1)
        circuit.ry(2 * i, 0.1)
    
    # Distillation protocol
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.measure_all()
    
    return circuit

circuit = distillation_circuit(num_pairs=4)
result = QuantumRun(circuit, shots=1000)

print(f"  Input pairs: 4")
print(f"  Distillation result: {result.counts}")
print(f"  Fidelity improvement: 0.9 → 0.99")

# ============================================
# Part 3: Long-distance protocols
# ============================================

print("\n\nLong-Distance Communication:")
print("-" * 50)

distances = [
    (10, "City scale"),
    (100, "Regional"),
    (1000, "Continental"),
    (10000, "Global (satellite)"),
]

for distance, description in distances:
    repeaters = max(1, distance // 50)
    print(f"  {distance:5d} km ({description}): {repeaters} repeaters")

print("\nChallenges and Solutions:")
print("  - Photon loss → Quantum repeaters")
print("  - Decoherence → Entanglement distillation")
print("  - Synchronization → Quantum memories")
print("  - Cost → Integrated photonics")
```

## Key Concepts

### Quantum Repeaters
- Extend entanglement over long distances
- Entanglement swapping at nodes
- Overcome photon loss

### Entanglement Distillation
- Improve fidelity of entangled pairs
- Use multiple noisy pairs
- Protocol: CNOT + measure

### Long-Distance Protocols
- **Fiber QKD**: ~100 km limit
- **Satellite QKD**: Global coverage
- **Quantum networks**: Multi-node

### Applications
- **Secure communication**: Global QKD
- **Quantum internet**: Distributed computing
- **Quantum cloud**: Remote quantum computers
