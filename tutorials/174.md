```python
import math
import numpy as np
from abirqu import Circuit, H, X, CNOT, Ry, Rz, Measure
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Market prediction
# ============================================

print("Quantum Real Estate Analytics:")
print("=" * 50)

def quantum_market_prediction(trends, indicators):
    """Quantum market prediction."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QMarket")
    
    # Encode trends
    circuit.ry(0, trends[0] * math.pi)
    circuit.ry(1, trends[1] * math.pi)
    
    # Encode indicators
    circuit.ry(2, indicators[0] * math.pi)
    circuit.ry(3, indicators[1] * math.pi)
    
    # Prediction
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure([2])
    return circuit

print("Quantum Real Estate Analytics:")
print("-" * 50)

market_scenarios = [
    ("Bull", [0.8, 0.2], [0.7, 0.3]),
    ("Bear", [0.3, 0.7], [0.2, 0.8]),
    ("Stable", [0.5, 0.5], [0.5, 0.5]),
    ("Volatile", [0.6, 0.4], [0.4, 0.6]),
]

for name, trends, indicators in market_scenarios:
    circuit = quantum_market_prediction(trends, indicators)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 2: Property valuation
# ============================================

print("\n\nProperty Valuation:")
print("-" * 50)

def quantum_property_valuation(property, comparable):
    """Quantum property valuation."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QValuation")
    
    # Encode property
    circuit.ry(0, property[0] * math.pi)
    circuit.ry(1, property[1] * math.pi)
    
    # Encode comparable
    circuit.ry(2, comparable[0] * math.pi)
    circuit.ry(3, comparable[1] * math.pi)
    
    # Valuation
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure([2, 3])
    return circuit

property_types = [
    ("Residential", [0.5, 0.5], [0.5, 0.5]),
    ("Commercial", [0.7, 0.3], [0.6, 0.4]),
    ("Industrial", [0.4, 0.6], [0.4, 0.6]),
    ("Mixed-Use", [0.6, 0.4], [0.7, 0.3]),
]

for name, property, comparable in property_types:
    circuit = quantum_property_valuation(property, comparable)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 3: Investment analysis
# ============================================

print("\n\nInvestment Analysis:")
print("-" * 50)

def quantum_investment_analysis(property, market):
    """Quantum investment analysis."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QInvestment")
    
    # Encode property
    circuit.ry(0, property[0] * math.pi)
    circuit.ry(1, property[1] * math.pi)
    
    # Encode market
    circuit.ry(2, market[0] * math.pi)
    circuit.ry(3, market[1] * math.pi)
    
    # Analysis
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure([2])
    return circuit

investment_scenarios = [
    ("Buy", [0.7, 0.3], [0.8, 0.2]),
    ("Hold", [0.5, 0.5], [0.5, 0.5]),
    ("Sell", [0.3, 0.7], [0.2, 0.8]),
    ("Refinance", [0.6, 0.4], [0.6, 0.4]),
]

for name, property, market in investment_scenarios:
    circuit = quantum_investment_analysis(property, market)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

print("\nApplications:")
print("  - Market analysis")
print("  - Property appraisal")
print("  - Investment decisions")
print("  - Portfolio management")
```
