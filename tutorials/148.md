```python
import math
import numpy as np
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Player performance
# ============================================

print("Quantum Sports Analytics:")
print("=" * 50)

def quantum_player_performance(stats, conditions):
    """Quantum player performance analysis."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QPlayer")
    
    # Encode stats
    circuit.ry(0, stats[0] * math.pi)
    circuit.ry(1, stats[1] * math.pi)
    
    # Encode conditions
    circuit.ry(2, conditions[0] * math.pi)
    circuit.ry(3, conditions[1] * math.pi)
    
    # Performance analysis
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

print("Quantum Sports Analytics:")
print("-" * 50)

players = [
    ("Star", [0.9, 0.1], [0.8, 0.2]),
    ("Starter", [0.7, 0.3], [0.6, 0.4]),
    ("Role Player", [0.5, 0.5], [0.5, 0.5]),
    ("Bench", [0.3, 0.7], [0.4, 0.6]),
]

for name, stats, conditions in players:
    circuit = quantum_player_performance(stats, conditions)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 2: Game strategy
# ============================================

print("\n\nGame Strategy:")
print("-" * 50)

def quantum_game_strategy(opponent, tactics):
    """Quantum game strategy."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QStrategy")
    
    # Encode opponent
    circuit.ry(0, opponent[0] * math.pi)
    circuit.ry(1, opponent[1] * math.pi)
    
    # Encode tactics
    circuit.ry(2, tactics[0] * math.pi)
    circuit.ry(3, tactics[1] * math.pi)
    
    # Strategy optimization
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

strategies = [
    ("Offensive", [0.3, 0.7], [0.8, 0.2]),
    ("Defensive", [0.7, 0.3], [0.2, 0.8]),
    ("Balanced", [0.5, 0.5], [0.5, 0.5]),
    ("Counter", [0.6, 0.4], [0.4, 0.6]),
]

for name, opponent, tactics in strategies:
    circuit = quantum_game_strategy(opponent, tactics)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 3: Injury prediction
# ============================================

print("\n\nInjury Prediction:")
print("-" * 50)

def quantum_injury_prediction(load, history):
    """Quantum injury prediction."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QInjury")
    
    # Encode workload
    circuit.ry(0, load[0] * math.pi)
    circuit.ry(1, load[1] * math.pi)
    
    # Encode injury history
    circuit.ry(2, history[0] * math.pi)
    circuit.ry(3, history[1] * math.pi)
    
    # Risk assessment
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

athletes = [
    ("Sprint", [0.8, 0.2], [0.3, 0.7]),
    ("Endurance", [0.6, 0.4], [0.5, 0.5]),
    ("Team Sport", [0.7, 0.3], [0.4, 0.6]),
    ("Combat", [0.9, 0.1], [0.6, 0.4]),
]

for name, load, history in athletes:
    circuit = quantum_injury_prediction(load, history)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

print("\nApplications:")
print("  - Performance optimization")
print("  - Game planning")
print("  - Injury prevention")
print("  - Talent scouting")
```
