```python
import math
import numpy as np
from abirqu import Circuit, H, X, CNOT, Ry, Rz, Measure
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Decentralized applications
# ============================================

print("Quantum Web3:")
print("=" * 50)

def quantum_dapp(frontend, backend):
    """Quantum decentralized application."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QDapp")
    
    # Encode frontend
    circuit.ry(0, frontend[0] * math.pi)
    circuit.ry(1, frontend[1] * math.pi)
    
    # Encode backend
    circuit.ry(2, backend[0] * math.pi)
    circuit.ry(3, backend[1] * math.pi)
    
    # Dapp logic
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure([2, 3])
    return circuit

print("Quantum Web3:")
print("-" * 50)

dapp_types = [
    ("DeFi", [0.7, 0.3], [0.8, 0.2]),
    ("NFT", [0.6, 0.4], [0.5, 0.5]),
    ("DAO", [0.5, 0.5], [0.6, 0.4]),
    ("Social", [0.4, 0.6], [0.4, 0.6]),
]

for name, frontend, backend in dapp_types:
    circuit = quantum_dapp(frontend, backend)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 2: Smart contracts
# ============================================

print("\n\nSmart Contracts:")
print("-" * 50)

def quantum_smart_contract(input_data, logic):
    """Quantum smart contract."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QSmartContract")
    
    # Encode input
    circuit.ry(0, input_data[0] * math.pi)
    circuit.ry(1, input_data[1] * math.pi)
    
    # Encode logic
    circuit.ry(2, logic[0] * math.pi)
    circuit.ry(3, logic[1] * math.pi)
    
    # Execution
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure([2])
    return circuit

contract_types = [
    ("Token", [0.6, 0.4], [0.7, 0.3]),
    ("Exchange", [0.7, 0.3], [0.6, 0.4]),
    ("Lending", [0.5, 0.5], [0.5, 0.5]),
    ("Insurance", [0.4, 0.6], [0.4, 0.6]),
]

for name, input_data, logic in contract_types:
    circuit = quantum_smart_contract(input_data, logic)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 3: DAO governance
# ============================================

print("\n\nDAO Governance:")
print("-" * 50)

def quantum_dao(proposals, voting):
    """Quantum DAO governance."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QDAO")
    
    # Encode proposals
    circuit.ry(0, proposals[0] * math.pi)
    circuit.ry(1, proposals[1] * math.pi)
    
    # Encode voting
    circuit.ry(2, voting[0] * math.pi)
    circuit.ry(3, voting[1] * math.pi)
    
    # Governance
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure([2, 3])
    return circuit

dao_types = [
    ("Protocol", [0.8, 0.2], [0.7, 0.3]),
    ("Investment", [0.6, 0.4], [0.5, 0.5]),
    ("Social", [0.5, 0.5], [0.4, 0.6]),
    ("Collector", [0.4, 0.6], [0.6, 0.4]),
]

for name, proposals, voting in dao_types:
    circuit = quantum_dao(proposals, voting)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

print("\nApplications:")
print("  - Decentralized apps")
print("  - Smart contracts")
print("  - DAO organizations")
print("  - Web3 infrastructure")
```
