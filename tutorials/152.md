```python
import math
import numpy as np
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Aircraft design
# ============================================

print("Quantum Aerospace:")
print("=" * 50)

def quantum_aircraft_design(aerodynamics, materials):
    """Quantum aircraft design."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QAircraft")
    
    # Encode aerodynamics
    circuit.ry(0, aerodynamics[0] * math.pi)
    circuit.ry(1, aerodynamics[1] * math.pi)
    
    # Encode materials
    circuit.ry(2, materials[0] * math.pi)
    circuit.ry(3, materials[1] * math.pi)
    
    # Design optimization
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

print("Quantum Aerospace:")
print("-" * 50)

aircraft_types = [
    ("Commercial", [0.7, 0.3], [0.6, 0.4]),
    ("Military", [0.8, 0.2], [0.7, 0.3]),
    ("Cargo", [0.5, 0.5], [0.5, 0.5]),
    ("Drone", [0.6, 0.4], [0.4, 0.6]),
]

for name, aerodynamics, materials in aircraft_types:
    circuit = quantum_aircraft_design(aerodynamics, materials)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 2: Flight optimization
# ============================================

print("\n\nFlight Optimization:")
print("-" * 50)

def quantum_flight_optimization(route, weather):
    """Quantum flight optimization."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QFlight")
    
    # Encode route
    circuit.ry(0, route[0] * math.pi)
    circuit.ry(1, route[1] * math.pi)
    
    # Encode weather
    circuit.ry(2, weather[0] * math.pi)
    circuit.ry(3, weather[1] * math.pi)
    
    # Optimization
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

flights = [
    ("Transatlantic", [0.6, 0.4], [0.3, 0.7]),
    ("Domestic", [0.4, 0.6], [0.5, 0.5]),
    ("Cargo", [0.5, 0.5], [0.4, 0.6]),
    ("Charter", [0.3, 0.7], [0.6, 0.4]),
]

for name, route, weather in flights:
    circuit = quantum_flight_optimization(route, weather)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 3: Predictive maintenance
# ============================================

print("\n\nPredictive Maintenance:")
print("-" * 50)

def quantum_aero_maintenance(sensors, history):
    """Quantum aerospace predictive maintenance."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QAeroMaint")
    
    # Encode sensors
    circuit.ry(0, sensors[0] * math.pi)
    circuit.ry(1, sensors[1] * math.pi)
    
    # Encode history
    circuit.ry(2, history[0] * math.pi)
    circuit.ry(3, history[1] * math.pi)
    
    # Prediction
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

components = [
    ("Engine", [0.8, 0.2], [0.3, 0.7]),
    ("Avionics", [0.5, 0.5], [0.5, 0.5]),
    ("Hydraulics", [0.6, 0.4], [0.4, 0.6]),
    ("Landing Gear", [0.4, 0.6], [0.6, 0.4]),
]

for name, sensors, history in components:
    circuit = quantum_aero_maintenance(sensors, history)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

print("\nApplications:")
print("  - Aircraft design")
print("  - Route planning")
print("  - Fuel optimization")
print("  - Safety systems")
```
