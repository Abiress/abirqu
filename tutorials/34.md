## Step-by-Step Code

```python
import math
import numpy as np
from abirqu import Circuit, H, X, CNOT, Ry, Rz, Measure
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Molecular Hamiltonian (LiH simplified)
# ============================================

print("Quantum Molecular Simulation:")
print("=" * 50)

# LiH molecule (simplified 4-qubit model)
# Real LiH needs 12+ qubits for minimal basis
hamiltonian_terms = [
    (-0.8105, 'IIII'),
    (0.1712, 'IIZI'),
    (0.1712, 'IZII'),
    (-0.2227, 'IZIZ'),
    (0.0453, 'IXIX'),
    (0.0453, 'IYIY'),
    (0.1204, 'ZIII'),
    (0.1204, 'IIIZ'),
    (-0.0631, 'ZIZI'),
    (0.0453, 'XXII'),
    (0.0453, 'YYII'),
    (0.0453, 'IIXX'),
    (0.0453, 'IIYY'),
]

print("LiH Hamiltonian (simplified 4-qubit):")
print(f"Number of terms: {len(hamiltonian_terms)}")
for coeff, term in hamiltonian_terms[:5]:
    print(f"  {coeff:+.4f} × {term}")
print("  ...")

# ============================================
# Part 2: UCCSD Ansatz for LiH
# ============================================

print("\n\nUCCSD Ansatz:")
print("-" * 50)

def build_uccsd_4q(params):
    """Build UCCSD ansatz for 4-qubit LiH."""
    circuit = Circuit(4, name="UCCSD_LiH")
    
    # Hartree-Fock reference |1100⟩ (2 electrons in first orbitals)
    circuit.x(0)
    circuit.x(1)
    
    # Singles excitations
    circuit.ry(0, params[0])
    circuit.cnot(0, 2)
    
    circuit.ry(1, params[1])
    circuit.cnot(1, 3)
    
    # Doubles excitations
    circuit.ry(0, params[2])
    circuit.cnot(0, 1)
    circuit.cnot(1, 2)
    circuit.cnot(2, 3)
    
    # Additional correlation
    circuit.ry(0, params[3])
    circuit.ry(1, params[4])
    circuit.ry(2, params[5])
    circuit.ry(3, params[6])
    
    return circuit

# ============================================
# Part 3: Energy optimization
# ============================================

print("\nVQE Optimization:")
print("-" * 50)

def compute_energy(params):
    """Compute molecular energy."""
    circuit = build_uccsd_4q(params)
    
    energy = 0
    for coeff, term in hamiltonian_terms:
        # Measure Pauli term
        measure_circuit = circuit.copy()
        
        # Basis rotations
        for i, pauli in enumerate(term):
            if pauli == 'X':
                measure_circuit.h(i)
            elif pauli == 'Y':
                measure_circuit.s(i)
                measure_circuit.h(i)
        
        measure_circuit.measure([0, 1, 2, 3])
        
        result = QuantumRun(measure_circuit, shots=200)
        
        # Compute expectation
        expectation = 0
        for bitstring, count in result.counts.items():
            val = 1
            for j, bit in enumerate(bitstring):
                if bit == '1':
                    val *= -1 if term[j] in ['Z', 'I'] else 1
            expectation += val * count / 200
        
        energy += coeff * expectation
    
    return energy

# Optimize
print("Optimizing:")
best_energy = float('inf')

for _ in range(50):
    params = np.random.uniform(-0.3, 0.3, 7)
    energy = compute_energy(params)
    
    if energy < best_energy:
        best_energy = energy
        if _ % 10 == 0:
            print(f"  Iteration {_}: Energy = {energy:.6f} Ha")

print(f"\nFinal energy: {best_energy:.6f} Ha")
print("Exact LiH energy (4-qubit): ≈ -7.863 Ha (simplified)")
```

## Key Concepts

### Molecular Simulation Pipeline
1. **Encode** molecular orbitals as qubits (Jordan-Wigner)
2. **Prepare** trial wavefunction (UCCSD)
3. **Measure** Hamiltonian expectation
4. **Optimize** to find ground state energy

### Jordan-Wigner Encoding
- Maps fermionic operators to qubit operators
- Preserves antisymmetry of fermions
- Creates non-local Pauli strings
