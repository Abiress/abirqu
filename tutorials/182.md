import math
import numpy as np
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: NFT creation
# ============================================

print("Quantum NFTs & Digital Art:")
print("=" * 50)

def quantum_nft_creation(artwork, metadata):
    """Quantum NFT creation."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QNFT")
    
    # Encode artwork
    circuit.ry(0, artwork[0] * math.pi)
    circuit.ry(1, artwork[1] * math.pi)
    
    # Encode metadata
    circuit.ry(2, metadata[0] * math.pi)
    circuit.ry(3, metadata[1] * math.pi)
    
    # Creation
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

print("Quantum NFTs & Digital Art:")
print("-" * 50)

nft_types = [
    ("Art", [0.8, 0.2], [0.7, 0.3]),
    ("Music", [0.6, 0.4], [0.5, 0.5]),
    ("Video", [0.5, 0.5], [0.6, 0.4]),
    ("Collectible", [0.4, 0.6], [0.4, 0.6]),
]

for name, artwork, metadata in nft_types:
    circuit = quantum_nft_creation(artwork, metadata)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 2: Digital ownership
# ============================================

print("\n\nDigital Ownership:")

def quantum_digital_ownership(owner, asset):
    """Quantum digital ownership."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QOwnership")
    
    # Encode owner
    circuit.ry(0, owner[0] * math.pi)
    circuit.ry(1, owner[1] * math.pi)
    
    # Encode asset
    circuit.ry(2, asset[0] * math.pi)
    circuit.ry(3, asset[1] * math.pi)
    
    # Ownership verification
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    
    circuit.measure_all()
    return circuit

ownership_scenarios = [
    ("Individual", [0.7, 0.3], [0.6, 0.4]),
    ("Shared", [0.5, 0.5], [0.5, 0.5]),
    ("DAO", [0.6, 0.4], [0.4, 0.6]),
    (" fractional", [0.4, 0.6], [0.3, 0.7]),
]

for name, owner, asset in ownership_scenarios:
    circuit = quantum_digital_ownership(owner, asset)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 3: Art authentication
# ============================================

print("\n\nArt Authentication:")
print("-" * 50)

def quantum_art_authentication(artist, provenance):
    """Quantum art authentication."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QAuth")
    
    # Encode artist
    circuit.ry(0, artist[0] * math.pi)
    circuit.ry(1, artist[1] * math.pi)
    
    # Encode provenance
    circuit.ry(2, provenance[0] * math.pi)
    circuit.ry(3, provenance[1] * math.pi)
    
    # Authentication
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

art_scenarios = [
    ("Original", [0.9, 0.1], [0.8, 0.2]),
    ("Print", [0.5, 0.5], [0.3, 0.7]),
    ("Remix", [0.6, 0.4], [0.5, 0.5]),

for name, artist, provenance in art_scenarios:
    circuit = quantum_art_authentication(artist, provenance)
    result = QuantumRun(circuit, shots=100)

print("\nApplications:")

print("  - Music licensing")
