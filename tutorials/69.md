import math
import numpy as np
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Quantum annealing
# ============================================

print("Expert Quantum Optimization:")
print("=" * 50)

def quantum_annealing_circuit(num_qubits=4, schedule_steps=10):
    """Simulate quantum annealing."""
    circuit = Circuit(num_qubits, name="QAnnealing")
    
    # Initial Hamiltonian: transverse field
    for i in range(num_qubits):
        circuit.h(i)
    
    # Annealing schedule
    for step in range(schedule_steps):
        s = step / (schedule_steps - 1)
        
        # Transverse field decreases
        for i in range(num_qubits):
            circuit.rx(i, (1 - s) * 0.1)
        
        # Problem Hamiltonian increases
        for i in range(num_qubits - 1):
            circuit.cnot(i, i + 1)
            circuit.rz(i + 1, s * 0.1)
            circuit.cnot(i, i + 1)
    
    circuit.measure_all()
    return circuit

print("Quantum Annealing:")
print("-" * 50)

circuit = quantum_annealing_circuit()
result = QuantumRun(circuit, shots=1000)
print(f"Annealing result: {result.counts}")

# ============================================
# Part 2: Adiabatic quantum computation
# ============================================

print("\n\nAdiabatic Quantum Computation:")

def adiabatic_circuit(problem_hamiltonian, time=1.0, steps=20):
    """Adiabatic quantum computation."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="Adiabatic")
    
    # Initial state: ground state of H_initial
    for i in range(num_qubits):
        circuit.h(i)
    
    # Adiabatic evolution
    dt = time / steps
    for step in range(steps):
        s = step / steps
        
        # H(s) = (1-s)H_initial + sH_problem
        for i in range(num_qubits):
            circuit.rx(i, (1 - s) * dt)
        
        for i in range(num_qubits - 1):
            circuit.cnot(i, i + 1)
            circuit.rz(i + 1, s * dt)
            circuit.cnot(i, i + 1)
    
    circuit.measure_all()
    return circuit

circuit = adiabatic_circuit(None)
print(f"Adiabatic circuit depth: {circuit.depth()}")

## Key Concepts

### Quantum Annealing
- Start in easy ground state
- Slowly change to problem Hamiltonian
- Stay in ground state (adiabatic theorem)

### Adiabatic QC
- Equivalent to gate model
- Polynomial overhead
- Potential quantum advantage

### Optimization Guarantees
- **QAOA**: Heuristic, no worst-case guarantee
- **Annealing**: Adiabatic theorem guarantees
- **VQE**: Variational principle

### Applications
- **Combinatorial optimization**: MaxCut, TSP
- **Machine learning**: Training neural networks
- **Finance**: Portfolio optimization
- **Logistics**: Route planning
