```python
import math
import numpy as np
from abirqu import Circuit, H, X, CNOT, Ry, Rz, Measure
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Drug discovery
# ============================================

print("Advanced Quantum Chemistry:")
print("=" * 50)

def drug_interaction_circuit(molecule_a, molecule_b, params):
    """Simulate drug-target interaction."""
    circuit = Circuit(4, name="DrugInteraction")
    
    # Encode molecule A
    circuit.ry(0, molecule_a[0] * math.pi)
    circuit.rz(0, molecule_a[1] * math.pi)
    
    # Encode molecule B
    circuit.ry(1, molecule_b[0] * math.pi)
    circuit.rz(1, molecule_b[1] * math.pi)
    
    # Interaction terms
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, params[0])
    circuit.rz(3, params[1])
    
    # Energy calculation
    circuit.cnot(2, 3)
    
    circuit.measure([2, 3])
    return circuit

drugs = [
    ("Aspirin", [0.6, 0.3]),
    ("Ibuprofen", [0.7, 0.4]),
    ("Paracetamol", [0.5, 0.2]),
]

targets = [
    ("COX-2", [0.8, 0.5]),
    ("ACE2", [0.3, 0.7]),
]

params = [0.5, 0.5]

print("Drug Discovery:")
print("-" * 50)

for drug_name, drug_props in drugs:
    for target_name, target_props in targets:
        circuit = drug_interaction_circuit(drug_props, target_props, params)
        result = QuantumRun(circuit, shots=100)
        print(f"  {drug_name} + {target_name}: {result.counts}")

# ============================================
# Part 2: Materials science
# ============================================

print("\n\nMaterials Science:")
print("-" * 50)

def material_property_circuit(bond_lengths, angles, params):
    """Calculate material properties."""
    circuit = Circuit(4, name="MaterialProp")
    
    # Encode bond lengths
    circuit.ry(0, bond_lengths[0])
    circuit.ry(1, bond_lengths[1])
    
    # Encode angles
    circuit.ry(2, angles[0])
    circuit.ry(3, angles[1])
    
    # Property calculation
    circuit.cnot(0, 1)
    circuit.cnot(2, 3)
    circuit.cnot(1, 3)
    
    # Optimization
    for i in range(4):
        circuit.ry(i, params[i])
    
    circuit.measure([0, 1])
    return circuit

materials = [
    ("Graphene", [1.42, 1.42], [120, 120]),
    ("Diamond", [1.54, 1.54], [109, 109]),
    ("Silicon", [2.35, 2.35], [110, 110]),
]

params = [0.5, 0.5, 0.5, 0.5]

for name, bonds, angles in materials:
    circuit = material_property_circuit(bonds, angles, params)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 3: Protein folding
# ============================================

print("\n\nProtein Folding:")
print("-" * 50)

def protein_folding_circuit(amino_acids, backbone_angles, params):
    """Simulate protein folding."""
    circuit = Circuit(4, name="ProteinFold")
    
    # Encode amino acids
    for i, aa in enumerate(amino_acids[:4]):
        circuit.ry(i, aa * math.pi)
    
    # Backbone interactions
    circuit.cnot(0, 1)
    circuit.cnot(1, 2)
    circuit.cnot(2, 3)
    
    # Folding energy
    for i in range(4):
        circuit.rz(i, backbone_angles[i] * 0.1)
    
    # Optimization
    for i in range(4):
        circuit.ry(i, params[i])
    
    circuit.measure([0, 1])
    return circuit

proteins = [
    ("Alanine", [0.2, 0.3, 0.4, 0.5], [60, 60, 60, 60]),
    ("Glycine", [0.1, 0.1, 0.1, 0.1], [180, 180, 180, 180]),
]

params = [0.5, 0.5, 0.5, 0.5]

for name, aas, angles in proteins:
    circuit = protein_folding_circuit(aas, angles, params)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

print("\nApplications:")
print("  - Drug design")
print("  - Material discovery")
print("  - Protein engineering")
print("  - Catalysis optimization")
```
