import math

from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Zero-noise extrapolation
# ============================================

print("Expert Error Mitigation:")
print("=" * 50)

def noisy_circuit(noise_level=0.1):
    """Circuit with adjustable noise."""
    circuit = Circuit(2, name="Noisy")
    circuit.h(0)
    circuit.cnot(0, 1)
    
    if noise_level > 0:
        circuit.rz(0, noise_level)
        circuit.rz(1, noise_level)
    
    circuit.measure_all()
    return circuit

print("Zero-Noise Extrapolation:")
print("-" * 50)

noise_levels = [0.0, 0.05, 0.1, 0.15, 0.2]
results = {}

for noise in noise_levels:
    circuit = noisy_circuit(noise)
    result = QuantumRun(circuit, shots=1000)
    fidelity = result.probabilities.get('00', 0) + result.probabilities.get('11', 0)
    results[noise] = fidelity
    print(f"  Noise={noise:.2f}: Fidelity={fidelity:.4f}")

# Linear extrapolation
x = np.array(noise_levels)
y = np.array([results[n] for n in noise_levels])
coeffs = np.polyfit(x, y, 1)
zero_noise_fidelity = coeffs[1]

print(f"\n  Extrapolated zero-noise fidelity: {zero_noise_fidelity:.4f}")

def build_calibration_matrix(num_qubits=2):
    """Build calibration matrix."""
    calibration = {}
    for state in ['00', '01', '10', '11']:
        circuit = Circuit(num_qubits, name=f"Cal_{state}")
        if state[0] == '1':
            circuit.x(0)
        if state[1] == '1':
            circuit.x(1)
        circuit.measure_all()
        
        result = QuantumRun(circuit, shots=1000)
        calibration[state] = result.counts
    return calibration

## Key Concepts

### ZNE
- Run at multiple noise levels
- Extrapolate to zero noise
- No extra qubits needed

### PEC
- Represent noise as quasi-probability
- Sample from distribution
- Combine with negative weights

### Measurement Mitigation
- Calibrate measurement errors
- Invert calibration matrix
- Fast and effective
