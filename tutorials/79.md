```python
import math
from abirqu import Circuit, H, X, CNOT, Ry, Rz, Measure
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Complexity classes
# ============================================

print("Quantum Complexity Theory:")
print("=" * 50)

classes = {
    'P': {
        'definition': 'Problems solvable in polynomial time',
        'quantum': 'Contained in BQP',
        'examples': ['Sorting', 'Shortest path'],
    },
    'NP': {
        'definition': 'Problems with polynomial-time verification',
        'quantum': 'Unknown relation to BQP',
        'examples': ['SAT', 'Graph coloring'],
    },
    'BQP': {
        'definition': 'Problems solvable by quantum computers in polynomial time',
        'classical': 'Contains P, contained in PSPACE',
        'examples': ['Factoring', 'Simulation'],
    },
    'QMA': {
        'definition': 'Quantum analog of NP',
        'classical': 'Contains NP, contained in PP',
        'examples': ['Local Hamiltonian'],
    },
}

for name, properties in classes.items():
    print(f"\n{name}:")
    for prop, value in properties.items():
        print(f"  {prop}: {value}")

# ============================================
# Part 2: Lower bounds
# ============================================

print("\n\nQuantum Lower Bounds:")
print("-" * 50)

lower_bounds = [
    ("Search", "Ω(√N)", "Grover's algorithm is optimal"),
    ("Element Distinctness", "Ω(N^{2/3})", "Ambainis' algorithm"),
    ("Parity", "Ω(N)", "No quantum speedup"),
    ("OR", "Ω(√N)", "Grover's algorithm"),
]

for problem, bound, note in lower_bounds:
    print(f"  {problem:25s}: {bound:10s} ({note})")

# ============================================
# Part 3: Quantum advantage
# ============================================

print("\n\nQuantum Advantage:")
print("-" * 50)

advantages = [
    ("Factoring", "Exponential", "Shor's algorithm"),
    ("Simulation", "Exponential", "Hamiltonian simulation"),
    ("Search", "Quadratic", "Grover's algorithm"),
    ("Optimization", "Unknown", "QAOA, annealing"),
    ("Machine Learning", "Polynomial-Exponential", "QSVM, QPCA"),
]

for problem, speedup, algorithm in advantages:
    print(f"  {problem:20s}: {speedup:15s} ({algorithm})")

print("\nOpen Questions:")
print("  - Is BQP = BPP? (Probably not)")
print("  - Is factoring in BPP? (Probably not)")
print("  - What is the power of quantum proofs?")
```

## Key Concepts

### Complexity Classes
- **P**: Efficiently solvable classically
- **NP**: Efficiently verifiable classically
- **BQP**: Efficiently solvable quantumly
- **QMA**: Quantum proofs help

### Lower Bounds
- Prove quantum algorithms are optimal
- No faster algorithm exists
- Fundamental limits

### Quantum Advantage
- Exponential: Shor, simulation
- Quadratic: Grover, search
- Unknown: Many problems

### Open Problems
- **BQP vs BPP**: Do quantum computers solve more?
- **Factoring**: Is it in BPP?
- **QMA vs NP**: Power of quantum proofs
