```python
import math
import numpy as np
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Exploration optimization
# ============================================

print("Quantum Mining:")
print("=" * 50)

def quantum_mining_exploration(geology, seismic):
    """Quantum mining exploration."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QMining")
    
    # Encode geology
    circuit.ry(0, geology[0] * math.pi)
    circuit.ry(1, geology[1] * math.pi)
    
    # Encode seismic
    circuit.ry(2, seismic[0] * math.pi)
    circuit.ry(3, seismic[1] * math.pi)
    
    # Exploration analysis
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

print("Quantum Mining:")
print("-" * 50)

mining_sites = [
    ("Gold", [0.7, 0.3], [0.6, 0.4]),
    ("Copper", [0.5, 0.5], [0.5, 0.5]),
    ("Iron", [0.6, 0.4], [0.4, 0.6]),
    ("Lithium", [0.4, 0.6], [0.7, 0.3]),
]

for name, geology, seismic in mining_sites:
    circuit = quantum_mining_exploration(geology, seismic)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 2: Resource extraction
# ============================================

print("\n\nResource Extraction:")
print("-" * 50)

def quantum_resource_extraction(depth, grade):
    """Quantum resource extraction."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QExtract")
    
    # Encode depth
    circuit.ry(0, depth[0] * math.pi)
    circuit.ry(1, depth[1] * math.pi)
    
    # Encode grade
    circuit.ry(2, grade[0] * math.pi)
    circuit.ry(3, grade[1] * math.pi)
    
    # Extraction optimization
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

extraction_methods = [
    ("Open Pit", [0.3, 0.7], [0.5, 0.5]),
    ("Underground", [0.8, 0.2], [0.7, 0.3]),
    ("Placer", [0.2, 0.8], [0.4, 0.6]),
    ("In-Situ", [0.5, 0.5], [0.6, 0.4]),
]

for name, depth, grade in extraction_methods:
    circuit = quantum_resource_extraction(depth, grade)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 3: Safety monitoring
# ============================================

print("\n\nSafety Monitoring:")
print("-" * 50)

def quantum_safety_monitoring(conditions, equipment):
    """Quantum mining safety monitoring."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QSafety")
    
    # Encode conditions
    circuit.ry(0, conditions[0] * math.pi)
    circuit.ry(1, conditions[1] * math.pi)
    
    # Encode equipment
    circuit.ry(2, equipment[0] * math.pi)
    circuit.ry(3, equipment[1] * math.pi)
    
    # Safety analysis
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

safety_areas = [
    ("Ventilation", [0.6, 0.4], [0.5, 0.5]),
    ("Ground Stability", [0.7, 0.3], [0.6, 0.4]),
    ("Gas Detection", [0.5, 0.5], [0.4, 0.6]),
    ("Equipment", [0.4, 0.6], [0.7, 0.3]),
]

for name, conditions, equipment in safety_areas:
    circuit = quantum_safety_monitoring(conditions, equipment)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

print("\nApplications:")
print("  - Site exploration")
print("  - Resource estimation")
print("  - Extraction planning")
print("  - Safety management")
```
