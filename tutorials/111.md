```python
import math
import numpy as np
from abirqu import Circuit, H, X, CNOT, Ry, Rz, Measure
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Quantum NLP
# ============================================

print("Advanced Quantum ML Applications:")
print("=" * 50)

def quantum_text_classifier(text_embedding, params):
    """Quantum text classifier."""
    circuit = Circuit(4, name="QTextClassifier")
    
    # Encode text embedding
    for i in range(4):
        circuit.ry(i, text_embedding[i] * math.pi)
    
    # Classification layers
    circuit.cnot(0, 1)
    circuit.cnot(2, 3)
    circuit.cnot(1, 3)
    
    # Parameterized layer
    for i in range(4):
        circuit.ry(i, params[i])
    
    circuit.measure([0])
    return circuit

print("Quantum NLP:")
print("-" * 50)

# Simple text embeddings
texts = [
    ("I love quantum computing", [0.8, 0.2, 0.6, 0.4]),
    ("Quantum is amazing", [0.9, 0.3, 0.7, 0.5]),
    ("I hate quantum", [0.1, 0.8, 0.2, 0.3]),
]

params = [0.5, 0.5, 0.5, 0.5]

for text, embedding in texts:
    circuit = quantum_text_classifier(embedding, params)
    result = QuantumRun(circuit, shots=100)
    pred = 1 if result.probabilities.get('1', 0.5) > 0.5 else 0
    print(f"  '{text}' → {'Positive' if pred else 'Negative'}")

# ============================================
# Part 2: Quantum computer vision
# ============================================

print("\n\nQuantum Computer Vision:")
print("-" * 50)

def quantum_image_classifier(image_features, params):
    """Quantum image classifier."""
    circuit = Circuit(4, name="QImgClassifier")
    
    # Encode image features
    for i in range(4):
        circuit.ry(i, image_features[i] * math.pi)
    
    # Convolution layer
    circuit.cnot(0, 1)
    circuit.cnot(2, 3)
    
    # Pooling layer
    circuit.ry(0, params[0])
    circuit.ry(2, params[1])
    
    circuit.measure([0, 2])
    return circuit

# Sample images
images = [
    ("Cat", [0.8, 0.2, 0.6, 0.4]),
    ("Dog", [0.7, 0.3, 0.5, 0.5]),
    ("Car", [0.2, 0.8, 0.3, 0.7]),
]

params = [0.5, 0.5]

for name, features in images:
    circuit = quantum_image_classifier(features, params)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 3: Quantum time series
# ============================================

print("\n\nQuantum Time Series:")
print("-" * 50)

def quantum_ts_forecaster(series_window, params):
    """Quantum time series forecaster."""
    circuit = Circuit(4, name="QTSForecaster")
    
    # Encode time window
    for i in range(4):
        circuit.ry(i, series_window[i] * math.pi)
    
    # Recurrent layer
    circuit.cnot(0, 1)
    circuit.cnot(1, 2)
    circuit.cnot(2, 3)
    
    # Forecast layer
    for i in range(4):
        circuit.ry(i, params[i])
    
    circuit.measure([0])
    return circuit

# Sample series
series = [0.1, 0.3, 0.5, 0.7]
params = [0.5, 0.5, 0.5, 0.5]

circuit = quantum_ts_forecaster(series, params)
result = QuantumRun(circuit, shots=100)
print(f"  Series: {series}")
print(f"  Forecast: {result.counts}")

print("\nApplications:")
print("  - Sentiment analysis")
print("  - Image classification")
print("  - Stock prediction")
print("  - Anomaly detection")
```
