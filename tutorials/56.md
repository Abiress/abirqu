import math
import numpy as np
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Active space selection
# ============================================

print("Advanced Quantum Chemistry:")
print("=" * 50)

def active_space_selection(molecule='H2O', basis='sto-3g'):
    """Select active space for molecular simulation."""
    # Active space: most important orbitals
    active_spaces = {
        'H2O': {'electrons': 4, 'orbitals': 4},
        'LiH': {'electrons': 2, 'orbitals': 4},
        'BeH2': {'electrons': 4, 'orbitals': 6},
    }
    return active_spaces.get(molecule, {'electrons': 2, 'orbitals': 2})

print("Active Space Selection:")
print("-" * 50)

for mol in ['H2O', 'LiH', 'BeH2']:
    as_ = active_space_selection(mol)
    print(f"  {mol}: {as_['electrons']} electrons, {as_['orbitals']} orbitals")

# ============================================
# Part 2: UCCSD with active space
# ============================================

print("\n\nUCCSD with Active Space:")
print("-" * 50)

def active_space_uccsd(num_active_electrons, num_active_orbitals, params):
    """Build UCCSD circuit for active space."""
    num_qubits = num_active_orbitals
    circuit = Circuit(num_qubits, name="ActiveUCCSD")
    
    # Fill active orbitals
    for i in range(num_active_electrons // 2):
        circuit.x(2 * i)
        circuit.x(2 * i + 1)
    
    # Singles
    for i in range(num_active_electrons // 2):
        for j in range(num_active_orbitals // 2):
            if i != j:
                circuit.ry(2 * i, params[0])
                circuit.cnot(2 * i, 2 * j)
    
    # Doubles
    for i in range(num_active_electrons // 2):
        circuit.ry(2 * i, params[1])
        circuit.cnot(2 * i, (2 * i + 2) % num_qubits)
    
    return circuit

# ============================================
# Part 3: Energy decomposition
# ============================================

print("\nEnergy Decomposition:")
print("-" * 50)

def energy_decomposition(molecule='H2O'):
    """Decompose molecular energy."""
    components = {
        'H2O': {
            'nuclear_repulsion': 9.164,
            'electronic': -85.0,
            'correlation': -0.726,
        },
        'LiH': {
            'nuclear_repulsion': 1.0,
            'electronic': -8.0,
            'correlation': -0.863,
        },
    }
    return components.get(molecule, {'nuclear_repulsion': 0, 'electronic': 0, 'correlation': 0})

print("Energy Components (H2O):")
decomp = energy_decomposition('H2O')
total = sum(decomp.values())
print(f"  Nuclear repulsion: {decomp['nuclear_repulsion']:.3f} Ha")
print(f"  Electronic: {decomp['electronic']:.3f} Ha")
print(f"  Correlation: {decomp['correlation']:.3f} Ha")
print(f"  Total: {total:.3f} Ha")

## Key Concepts

### Active Space
- Select most important orbitals
- Reduces computational cost
- Focuses on chemically relevant orbitals

### Energy Decomposition
- **Nuclear repulsion**: Classical Coulomb
- **Electronic**: Hartree-Fock energy
- **Correlation**: Electron correlation effects

### Excited States
- **VQD**: Sequential state finding
- **QPE**: Direct phase estimation
- **SSVQE**: Simultaneous multi-state

### Applications
- **Drug design**: Binding affinities
- **Materials**: Band gaps, conductivity
- **Catalysis**: Reaction mechanisms
- **Spectroscopy**: Excitation energies
