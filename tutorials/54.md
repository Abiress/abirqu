import math
import numpy as np
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Quantum kernel methods
# ============================================

print("Advanced Quantum ML:")
print("=" * 50)

def quantum_kernel_matrix(X1, X2, num_qubits=2):
    """Compute quantum kernel matrix."""
    n1, n2 = len(X1), len(X2)
    K = np.zeros((n1, n2))
    
    for i in range(n1):
        for j in range(n2):
            # Build kernel circuit
            circuit = Circuit(2 * num_qubits, name="Kernel")
            
            # Encode x1
            for k in range(num_qubits):
                circuit.ry(k, X1[i][k] * math.pi)
            
            # Encode x2
            for k in range(num_qubits):
                circuit.ry(num_qubits + k, X2[j][k] * math.pi)
            
            # Compute overlap
            circuit.h(0)
            circuit.cnot(0, num_qubits)
            circuit.h(0)
            
            circuit.measure_all()
            result = QuantumRun(circuit, shots=100)
            
            K[i, j] = result.probabilities.get('0', 0.5)
    
    return K

print("Quantum Kernel Methods:")
print("-" * 50)

# Sample data
X_train = np.array([[0.1, 0.2], [0.3, 0.4], [0.7, 0.8], [0.9, 0.8]])
y_train = np.array([0, 0, 1, 1])

K = quantum_kernel_matrix(X_train, X_train)
print(f"Kernel matrix shape: {K.shape}")
print(f"Kernel diagonal: {np.diag(K).round(3)}")

# ============================================
# Part 2: Quantum generative models
# ============================================

print("\n\nQuantum Generative Models:")
print("-" * 50)

def quantum_generative_circuit(params, num_qubits=2):
    """Build quantum generative circuit."""
    circuit = Circuit(num_qubits, name="QGen")
    
    # Generate from parameters
    for i in range(num_qubits):
        circuit.ry(i, params[i])
    
    # Entanglement
    for i in range(num_qubits - 1):
        circuit.cnot(i, i + 1)
        circuit.ry(i + 1, params[num_qubits + i])
    
    circuit.measure_all()
    return circuit

# Train generator
params = np.random.uniform(0, math.pi, 4)
print("Training quantum generator:")

for epoch in range(10):
    circuit = quantum_generative_circuit(params)
    result = QuantumRun(circuit, shots=100)
    
    # Simple update
    params += 0.1 * np.random.randn(4)
    
    if (epoch + 1) % 5 == 0:
        print(f"  Epoch {epoch+1}: Generated distribution shape = {len(result.counts)}")

# ============================================
# Part 3: Quantum transfer learning
# ============================================

print("\n\nQuantum Transfer Learning:")
print("-" * 50)

def quantum_feature_extractor(circuit, params):
    """Extract features using quantum circuit."""
    for i, param in enumerate(params[:4]):
        circuit.ry(i, param)
    
    # Entangling layers
    circuit.cnot(0, 1)
    circuit.cnot(2, 3)
    
    return circuit

print("Transfer Learning Pipeline:")
print("  1. Pre-train quantum feature extractor on source domain")
print("  2. Freeze feature extractor")
print("  3. Train classifier on target domain")
print("  4. Fine-tune end-to-end")

print("\nAdvantages:")
print("  - Less labeled data needed")
print("  - Faster convergence")
print("  - Better generalization")

## Key Concepts

### Quantum Kernel Methods
- Map data to quantum feature space
- Compute kernel via quantum overlap
- Can capture complex patterns

### Quantum Generative Models
- **QGAN**: Adversarial training
- **QVAE**: Variational autoencoder
- **Born machine**: Learn probability distributions

### Transfer Learning
- Leverage pre-trained quantum features
- Adapt to new tasks with less data
- Combine classical and quantum components
