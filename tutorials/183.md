```python
import math
import numpy as np
from abirqu import Circuit, H, X, CNOT, Ry, Rz, Measure
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Virtual worlds
# ============================================

print("Quantum Metaverse:")
print("=" * 50)

def quantum_virtual_world(environment, users):
    """Quantum virtual world simulation."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QWorld")
    
    # Encode environment
    circuit.ry(0, environment[0] * math.pi)
    circuit.ry(1, environment[1] * math.pi)
    
    # Encode users
    circuit.ry(2, users[0] * math.pi)
    circuit.ry(3, users[1] * math.pi)
    
    # Simulation
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure([2, 3])
    return circuit

print("Quantum Metaverse:")
print("-" * 50)

world_scenarios = [
    ("Gaming", [0.7, 0.3], [0.8, 0.2]),
    ("Social", [0.5, 0.5], [0.6, 0.4]),
    ("Commerce", [0.6, 0.4], [0.5, 0.5]),
    ("Education", [0.4, 0.6], [0.4, 0.6]),
]

for name, environment, users in world_scenarios:
    circuit = quantum_virtual_world(environment, users)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 2: Avatar systems
# ============================================

print("\n\nAvatar Systems:")
print("-" * 50)

def quantum_avatar(appearance, identity):
    """Quantum avatar system."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QAvatar")
    
    # Encode appearance
    circuit.ry(0, appearance[0] * math.pi)
    circuit.ry(1, appearance[1] * math.pi)
    
    # Encode identity
    circuit.ry(2, identity[0] * math.pi)
    circuit.ry(3, identity[1] * math.pi)
    
    # System
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure([2, 3])
    return circuit

avatar_types = [
    ("Realistic", [0.8, 0.2], [0.7, 0.3]),
    ("Stylized", [0.6, 0.4], [0.5, 0.5]),
    ("Fantasy", [0.4, 0.6], [0.3, 0.7]),
    ("Anonymous", [0.2, 0.8], [0.2, 0.8]),
]

for name, appearance, identity in avatar_types:
    circuit = quantum_avatar(appearance, identity)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 3: Digital economies
# ============================================

print("\n\nDigital Economies:")
print("-" * 50)

def quantum_digital_economy(marketplace, currencies):
    """Quantum digital economy."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QEconomy")
    
    # Encode marketplace
    circuit.ry(0, marketplace[0] * math.pi)
    circuit.ry(1, marketplace[1] * math.pi)
    
    # Encode currencies
    circuit.ry(2, currencies[0] * math.pi)
    circuit.ry(3, currencies[1] * math.pi)
    
    # Economy
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure([2, 3])
    return circuit

economy_types = [
    ("Play-to-Earn", [0.7, 0.3], [0.6, 0.4]),
    ("Free-to-Play", [0.5, 0.5], [0.5, 0.5]),
    ("Subscription", [0.4, 0.6], [0.4, 0.6]),
    ("Marketplace", [0.6, 0.4], [0.7, 0.3]),
]

for name, marketplace, currencies in economy_types:
    circuit = quantum_digital_economy(marketplace, currencies)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

print("\nApplications:")
print("  - Virtual worlds")
print("  - Digital avatars")
print("  - Virtual economies")
print("  - Social platforms")
```
