import math
import numpy as np
from abirqu import Circuit, H, X, CNOT, Ry, Rz, Measure
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Quantum MRI enhancement
# ============================================

print("Quantum Medical Imaging:")
print("=" * 50)

def quantum_mri_reconstruction(k_space_data, params):
    """Quantum MRI reconstruction."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QMRI")
    
    # Encode k-space data
    for i in range(num_qubits):
        circuit.ry(i, k_space_data[i] * math.pi)
    
    # Quantum Fourier Transform
    for i in range(num_qubits):
        circuit.h(i)
        for j in range(i + 1, num_qubits):
            circuit.rz(j, math.pi / (2 ** (j - i)))
            circuit.cnot(i, j)
    
    # Reconstruction parameters
    for i in range(num_qubits):
        circuit.ry(i, params[i])
    
    circuit.measure(list(range(num_qubits)))
    return circuit

print("Quantum MRI Enhancement:")
print("-" * 50)

k_space = [0.5, 0.3, 0.7, 0.2]
params = [0.5, 0.5, 0.5, 0.5]

circuit = quantum_mri_reconstruction(k_space, params)
result = QuantumRun(circuit, shots=1000)
print(f"  K-space data: {k_space}")
print(f"  Reconstructed: {result.counts}")

# ============================================
# Part 2: Quantum CT reconstruction
# ============================================

print("\n\nQuantum CT Reconstruction:")
print("-" * 50)

def quantum_ct_reconstruction(projections, angles):
    """Quantum CT reconstruction."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QCT")
    
    # Encode projections
    for i in range(num_qubits):
        circuit.ry(i, projections[i] * math.pi)
    
    # Back-projection
    for angle in angles[:4]:
        for i in range(num_qubits):
            circuit.rz(i, angle * math.pi / 180)
    
    # Filtering
    circuit.cnot(0, 1)
    circuit.cnot(2, 3)
    circuit.cnot(1, 3)
    
    circuit.measure(list(range(num_qubits)))
    return circuit

projections = [0.8, 0.6, 0.4, 0.2]
angles = [0, 45, 90, 135, 180]

circuit = quantum_ct_reconstruction(projections, angles)
result = QuantumRun(circuit, shots=1000)
print(f"  Projections: {projections}")
print(f"  Angles: {angles}")
print(f"  Reconstructed: {result.counts}")

# ============================================
# Part 3: Quantum PET imaging
# ============================================

print("\n\nQuantum PET Imaging:")
print("-" * 50)

def quantum_pet_reconstruction(detections, params):
    """Quantum PET reconstruction."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QPET")
    
    # Encode coincidence detections
    for i in range(num_qubits):
        circuit.ry(i, detections[i] * math.pi)
    
    # Image reconstruction
    circuit.cnot(0, 1)
    circuit.cnot(2, 3)
    circuit.rz(1, params[0])
    circuit.rz(3, params[1])
    
    circuit.measure([0, 1])
    return circuit

detections = [0.9, 0.7, 0.5, 0.3]
params = [0.5, 0.5]

circuit = quantum_pet_reconstruction(detections, params)
result = QuantumRun(circuit, shots=1000)
print(f"  Detections: {detections}")
print(f"  Reconstructed: {result.counts}")

print("\nApplications:")
print("  - Brain imaging")
print("  - Cancer detection")
print("  - Cardiac imaging")
print("  - Bone scanning")
