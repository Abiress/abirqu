```python
import math
from abirqu import Circuit, H, X, Z, CNOT, Rx, Ry, Rz, Measure
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Quantum Logistic Map
# ============================================

print("Quantum Simulation of Chaotic Systems:")
print("=" * 50)

def quantum_logistic_map_circuit(num_qubits=4, iterations=3):
    """Build quantum circuit for logistic map simulation.
    
    x_{n+1} = r * x_n * (1 - x_n)
    Chaotic for r > 3.57
    """
    circuit = Circuit(num_qubits, name="LogisticMap")
    
    # Initialize in superposition
    for i in range(num_qubits):
        circuit.h(i)
    
    # Apply chaotic evolution
    for _ in range(iterations):
        # Nonlinear transformation (simplified)
        for i in range(num_qubits - 1):
            circuit.cnot(i, i + 1)
            circuit.rz(i + 1, math.pi / 4)
        
        # Mixing
        circuit.h(0)
    
    return circuit

# Run and analyze
circuit = quantum_logistic_map_circuit(num_qubits=4, iterations=3)
result = QuantumRun(circuit, shots=1000)

print("Logistic Map Quantum Simulation:")
print(f"Measurement distribution: {result.counts}")

# ============================================
# Part 2: Quantum Kicked Rotator
# ============================================

print("\n\nQuantum Kicked Rotator:")
print("=" * 50)

def quantum_kicked_rotator(num_qubits=4, kicks=5):
    """Build quantum kicked rotator circuit.
    
    Classic example of quantum chaos.
    """
    circuit = Circuit(num_qubits, name="KickedRotator")
    
    # Initialize
    for i in range(num_qubits):
        circuit.h(i)
    
    # Apply kicks (periodic driving)
    for kick in range(kicks):
        # Rotation (free evolution)
        for i in range(num_qubits):
            circuit.rz(i, kick * math.pi / 7)
        
        # Kicking (nonlinear)
        for i in range(num_qubits - 1):
            circuit.cnot(i, i + 1)
            circuit.rz(i + 1, math.pi / 3)
            circuit.cnot(i, i + 1)
    
    return circuit

circuit = quantum_kicked_rotator(num_qubits=4, kicks=5)
result = QuantumRun(circuit, shots=1000)

print("Kicked Rotator Results:")
print(f"States sampled: {len(result.counts)} unique states")
print(f"Most probable: {max(result.counts, key=result.counts.get)}")

# ============================================
# Part 3: Out-of-Time-Order Correlator (OTOC)
# ============================================

print("\n\nOut-of-Time-Order Correlator (OTOC):")
print("=" * 50)

def otoc_circuit(num_qubits=3, time_steps=3):
    """Build circuit to compute OTOC.
    
    OTOC measures quantum chaos and information scrambling.
    """
    circuit = Circuit(num_qubits + 1, name="OTOC")
    
    # Initialize in Bell state
    circuit.h(0)
    circuit.cnot(0, 1)
    
    # Apply time evolution
    for t in range(time_steps):
        # Forward evolution
        for i in range(num_qubits):
            circuit.rz(i, t * math.pi / 5)
            circuit.rx(i, math.pi / 4)
        
        # Backward evolution (for OTOC)
        if t < time_steps - 1:
            for i in range(num_qubits):
                circuit.rx(i, -math.pi / 4)
                circuit.rz(i, -t * math.pi / 5)
    
    # Measure
    circuit.measure(list(range(num_qubits + 1)))
    
    return circuit

circuit = otoc_circuit(num_qubits=3, time_steps=3)
result = QuantumRun(circuit, shots=1000)

print("OTOC Measurement:")
print(f"Counts: {result.counts}")
print("OTOC → 0 indicates quantum chaos (information scrambling)")
```

## Key Concepts

### Quantum Chaos Signatures
1. **Level statistics**: Energy levels follow Wigner-Dyson distribution
2. **OTOC decay**: Information scrambling is fast
3. **Entanglement growth**: Linear growth in time

### OTOC Definition
- C(t) = ⟨W(t)†V†W(t)V⟩
- W, V are local operators
- Decay of C(t) indicates chaos

### Applications
- **Black holes**: Information scrambling
- **Many-body physics**: Thermalization
- **Quantum computing**: Noise modeling
- **Cryptography**: Quantum-resistant protocols
