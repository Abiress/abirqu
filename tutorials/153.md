```python
import math
import numpy as np
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Route optimization
# ============================================

print("Quantum Maritime:")
print("=" * 50)

def quantum_maritime_route(distance, weather, cargo):
    """Quantum maritime route optimization."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QMaritime")
    
    # Encode distance
    circuit.ry(0, distance * math.pi)
    
    # Encode weather
    circuit.ry(1, weather[0] * math.pi)
    circuit.ry(2, weather[1] * math.pi)
    
    # Encode cargo
    circuit.ry(3, cargo * math.pi)
    
    # Route optimization
    circuit.cnot(0, 1)
    circuit.cnot(1, 2)
    circuit.cnot(2, 3)
    
    circuit.measure_all()
    return circuit

print("Quantum Maritime:")
print("-" * 50)

routes = [
    ("Trans-Pacific", 0.9, [0.3, 0.7], 0.8),
    ("Trans-Atlantic", 0.7, [0.5, 0.5], 0.6),
    ("Asia-Europe", 0.8, [0.4, 0.6], 0.7),
    ("Coastal", 0.2, [0.6, 0.4], 0.3),
]

for name, distance, weather, cargo in routes:
    circuit = quantum_maritime_route(distance, weather, cargo)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 2: Port operations
# ============================================

print("\n\nPort Operations:")
print("-" * 50)

def quantum_port_operations(berths, cranes):
    """Quantum port operations optimization."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QPort")
    
    # Encode berths
    circuit.ry(0, berths[0] * math.pi)
    circuit.ry(1, berths[1] * math.pi)
    
    # Encode cranes
    circuit.ry(2, cranes[0] * math.pi)
    circuit.ry(3, cranes[1] * math.pi)
    
    # Operations optimization
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

ports = [
    ("Singapore", [0.8, 0.2], [0.7, 0.3]),
    ("Shanghai", [0.9, 0.1], [0.8, 0.2]),
    ("Rotterdam", [0.6, 0.4], [0.5, 0.5]),
    ("Dubai", [0.5, 0.5], [0.6, 0.4]),
]

for name, berths, cranes in ports:
    circuit = quantum_port_operations(berths, cranes)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 3: Cargo management
# ============================================

print("\n\nCargo Management:")
print("-" * 50)

def quantum_cargo_management(volume, weight):
    """Quantum cargo management."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QCargo")
    
    # Encode volume
    circuit.ry(0, volume[0] * math.pi)
    circuit.ry(1, volume[1] * math.pi)
    
    # Encode weight
    circuit.ry(2, weight[0] * math.pi)
    circuit.ry(3, weight[1] * math.pi)
    
    # Cargo optimization
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

cargo_types = [
    ("Container", [0.7, 0.3], [0.6, 0.4]),
    ("Bulk", [0.5, 0.5], [0.8, 0.2]),
    ("Tanker", [0.4, 0.6], [0.9, 0.1]),
    ("Ro-Ro", [0.6, 0.4], [0.5, 0.5]),
]

for name, volume, weight in cargo_types:
    circuit = quantum_cargo_management(volume, weight)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

print("\nApplications:")
print("  - Shipping routes")
print("  - Port efficiency")
print("  - Cargo optimization")
print("  - Fleet management")
```
