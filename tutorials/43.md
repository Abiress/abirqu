```python
import math
from abirqu import Circuit, H, X, Y, Z
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Transversal gates
# ============================================

print("Fault-Tolerant Circuits:")
print("=" * 50)

def transversal_cnot(encoded_qubit_1, encoded_qubit_2):
    """Apply CNOT transversally between encoded qubits."""
    circuit = Circuit(6, name="TransCNOT")
    
    # Apply CNOT to each pair
    for i in range(3):
        circuit.cnot(i, i + 3)
    
    return circuit

print("Transversal Gates:")
print("-" * 50)
print("  Transversal CNOT: Apply CNOT to each qubit pair")
print("  Advantage: Errors don't spread between blocks")

# ============================================
# Part 2: Fault-tolerant measurement
# ============================================

print("\n\nFault-Tolerant Measurement:")
print("-" * 50)

def ft_measure_z(circuit, data_qubits, syndrome_qubit):
    """Fault-tolerant Z measurement."""
    # Use cat state for fault-tolerant measurement
    circuit.h(syndrome_qubit)
    
    for q in data_qubits:
        circuit.cnot(syndrome_qubit, q)
    
    circuit.h(syndrome_qubit)
    return circuit

circuit = Circuit(4, name="FTMeasure")
ft_measure_z(circuit, [0, 1, 2], 3)
circuit.measure_all()

result = QuantumRun(circuit, shots=100)
print(f"Fault-tolerant measurement: {result.counts}")

# ============================================
# Part 3: Magic state injection
# ============================================

print("\n\nMagic State Injection:")
print("-" * 50)

def magic_state_injection(circuit, data_qubit, magic_qubit):
    """Inject magic state for T gate."""
    circuit.cnot(magic_qubit, data_qubit)
    circuit.t(data_qubit)
    circuit.h(magic_qubit)
    circuit.measure(magic_qubit)
    return circuit

circuit = Circuit(2, name="MagicInject")
circuit.h(0)  # Data qubit

# Prepare magic state
circuit.h(1)
circuit.s(1)
circuit.t(1)

# Inject
magic_state_injection(circuit, 0, 1)

print("Magic state injection enables fault-tolerant T gate")

# ============================================
# Part 4: Resource overhead
# ============================================

print("\n\nResource Overhead:")
print("-" * 50)

print("  Logical qubit: ~1000 physical qubits (surface code)")
print("  T gate: ~100 physical gates (magic state)")
print("  CNOT: ~100 physical gates (transversal)")
print("  Total overhead: 1000-10000x for fault tolerance")
```

## Key Concepts

### Transversal Gates
- Apply gate independently to each qubit in code block
- Errors don't propagate between blocks
- Limited set available (Eastin-Knill theorem)

### Fault-Tolerant Measurement
- Measure syndrome without destroying encoded information
- Use ancilla qubits with error detection
- Repeat for reliability

### Applications
- **Fault-tolerant computing**: Reliable quantum computers
- **Quantum algorithms**: Shor, Grover at scale
- **Quantum simulation**: Large molecular systems
