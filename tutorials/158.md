import math
import numpy as np
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Content delivery
# ============================================

print("Quantum Media & Broadcasting:")
print("=" * 50)

def quantum_content_delivery(content, audience):
    """Quantum content delivery optimization."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QContent")
    
    # Encode content
    circuit.ry(0, content[0] * math.pi)
    circuit.ry(1, content[1] * math.pi)
    
    # Encode audience
    circuit.ry(2, audience[0] * math.pi)
    circuit.ry(3, audience[1] * math.pi)
    
    # Delivery optimization
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

print("Quantum Media & Broadcasting:")
print("-" * 50)

media_types = [
    ("Live TV", [0.8, 0.2], [0.7, 0.3]),
    ("On-Demand", [0.5, 0.5], [0.5, 0.5]),
    ("Streaming", [0.6, 0.4], [0.6, 0.4]),
    ("Podcast", [0.3, 0.7], [0.4, 0.6]),
]

for name, content, audience in media_types:
    circuit = quantum_content_delivery(content, audience)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 2: Audience measurement
# ============================================

print("\n\nAudience Measurement:")
print("-" * 50)

def quantum_audience_measurement(demographics, behavior):
    """Quantum audience measurement."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QAudience")
    
    # Encode demographics
    circuit.ry(0, demographics[0] * math.pi)
    circuit.ry(1, demographics[1] * math.pi)
    
    # Encode behavior
    circuit.ry(2, behavior[0] * math.pi)
    circuit.ry(3, behavior[1] * math.pi)
    
    # Measurement
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    
    circuit.measure_all()
    return circuit

audience_types = [
    ("Youth", [0.2, 0.8], [0.8, 0.2]),
    ("Adult", [0.5, 0.5], [0.5, 0.5]),
    ("Senior", [0.8, 0.2], [0.3, 0.7]),
    ("Family", [0.4, 0.6], [0.6, 0.4]),
]

for name, demographics, behavior in audience_types:
    circuit = quantum_audience_measurement(demographics, behavior)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 3: Ad optimization
# ============================================

print("\n\nAd Optimization:")
print("-" * 50)

def quantum_ad_optimization(campaign, targeting):
    """Quantum advertising optimization."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QAd")
    
    # Encode campaign
    circuit.ry(0, campaign[0] * math.pi)
    circuit.ry(1, campaign[1] * math.pi)
    
    # Encode targeting
    circuit.ry(2, targeting[0] * math.pi)
    circuit.ry(3, targeting[1] * math.pi)
    
    # Optimization
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

ad_scenarios = [
    ("Brand", [0.6, 0.4], [0.5, 0.5]),
    ("Performance", [0.7, 0.3], [0.7, 0.3]),
    ("Retargeting", [0.4, 0.6], [0.6, 0.4]),
    ("Programmatic", [0.5, 0.5], [0.8, 0.2]),
]

for name, campaign, targeting in ad_scenarios:
    circuit = quantum_ad_optimization(campaign, targeting)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

print("\nApplications:")
print("  - Content distribution")
print("  - Audience analytics")
print("  - Ad targeting")
print("  - Revenue optimization")
