import math
import numpy as np
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Case prediction
# ============================================

print("Quantum Legal Analytics:")
print("=" * 50)

def quantum_case_prediction(precedents, facts):
    """Quantum case prediction."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QCase")
    
    # Encode precedents
    circuit.ry(0, precedents[0] * math.pi)
    circuit.ry(1, precedents[1] * math.pi)
    
    # Encode facts
    circuit.ry(2, facts[0] * math.pi)
    circuit.ry(3, facts[1] * math.pi)
    
    # Prediction
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

print("Quantum Legal Analytics:")
print("-" * 50)

case_types = [
    ("Civil", [0.5, 0.5], [0.5, 0.5]),
    ("Criminal", [0.7, 0.3], [0.6, 0.4]),
    ("Commercial", [0.6, 0.4], [0.7, 0.3]),
    ("Administrative", [0.4, 0.6], [0.4, 0.6]),
]

for name, precedents, facts in case_types:
    circuit = quantum_case_prediction(precedents, facts)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 2: Contract analytics
# ============================================

print("\n\nContract Analytics:")

def quantum_contract_analytics(clauses, risks):
    """Quantum contract analytics."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QContract")
    
    # Encode clauses
    circuit.ry(0, clauses[0] * math.pi)
    circuit.ry(1, clauses[1] * math.pi)
    
    # Encode risks
    circuit.ry(2, risks[0] * math.pi)
    circuit.ry(3, risks[1] * math.pi)
    
    # Analytics
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

contract_types = [
    ("Sales", [0.6, 0.4], [0.4, 0.6]),
    ("Service", [0.5, 0.5], [0.5, 0.5]),
    ("Employment", [0.4, 0.6], [0.6, 0.4]),
    ("Partnership", [0.7, 0.3], [0.3, 0.7]),
]

for name, clauses, risks in contract_types:
    circuit = quantum_contract_analytics(clauses, risks)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 3: Legal research
# ============================================

print("\n\nLegal Research:")
print("-" * 50)

def quantum_legal_research(query, database):
    """Quantum legal research."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QLegal")
    
    # Encode query
    circuit.ry(0, query[0] * math.pi)
    circuit.ry(1, query[1] * math.pi)
    
    # Encode database
    circuit.ry(2, database[0] * math.pi)
    circuit.ry(3, database[1] * math.pi)
    
    # Research
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    
    circuit.measure_all()
    return circuit

research_scenarios = [
    ("Case Law", [0.7, 0.3], [0.6, 0.4]),
    ("Statutes", [0.5, 0.5], [0.5, 0.5]),
    ("Regulations", [0.6, 0.4], [0.7, 0.3]),

for name, query, database in research_scenarios:

    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

print("\nApplications:")

print("  - Contract review")
