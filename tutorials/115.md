import math
import numpy as np
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Many-body physics
# ============================================

print("Advanced Quantum Simulation:")
print("=" * 50)

def heisenberg_model_circuit(num_sites, coupling_constants):
    """Simulate Heisenberg model."""
    circuit = Circuit(num_sites, name="Heisenberg")
    
    # Initialize superposition
    for i in range(num_sites):
        circuit.h(i)
    
    # XX interactions
    for i in range(num_sites - 1):
        circuit.cnot(i, i + 1)
        circuit.rz(i + 1, coupling_constants[0])
        circuit.cnot(i, i + 1)
    
    # YY interactions
    for i in range(num_sites - 1):
        circuit.ry(i, math.pi / 2)
        circuit.ry(i + 1, math.pi / 2)
        circuit.cnot(i, i + 1)
        circuit.rz(i + 1, coupling_constants[1])
        circuit.cnot(i, i + 1)
        circuit.ry(i, -math.pi / 2)
        circuit.ry(i + 1, -math.pi / 2)
    
    # ZZ interactions
    for i in range(num_sites - 1):
        circuit.cnot(i, i + 1)
        circuit.rz(i + 1, coupling_constants[2])
        circuit.cnot(i, i + 1)
    
    circuit.measure_all()
    return circuit

print("Many-Body Physics:")
print("-" * 50)

circuit = heisenberg_model_circuit(4, [0.5, 0.5, 0.5])
result = QuantumRun(circuit, shots=1000)
print(f"  Heisenberg model: {len(result.counts)} states")

# ============================================
# Part 2: Lattice gauge theory
# ============================================

print("\n\nLattice Gauge Theory:")
print("-" * 50)

def lattice_gauge_circuit(num_links, coupling):
    """Simulate lattice gauge theory."""
    circuit = Circuit(num_links, name="LatticeGauge")
    
    # Initialize links
    for i in range(num_links):
        circuit.h(i)
    
    # Gauge transformations
    for i in range(num_links):
        circuit.ry(i, coupling)
    
    # Plaquette terms
    for i in range(0, num_links - 2, 2):
        circuit.cnot(i, i + 1)
        circuit.cnot(i + 1, i + 2)
        circuit.rz(i + 1, coupling)
        circuit.cnot(i + 1, i + 2)
        circuit.cnot(i, i + 1)
    
    circuit.measure_all()
    return circuit

circuit = lattice_gauge_circuit(6, 0.5)
result = QuantumRun(circuit, shots=1000)
print(f"  Lattice gauge theory: {len(result.counts)} states")

# ============================================
# Part 3: Quantum field theory
# ============================================

print("\n\nQuantum Field Theory:")
print("-" * 50)

def qft_field_circuit(num_modes, field_amplitudes):
    """Simulate QFT on lattice."""
    circuit = Circuit(num_modes, name="QFTField")
    
    # Initialize field modes
    for i in range(num_modes):
        circuit.ry(i, field_amplitudes[i])
    
    # Field interactions
    for i in range(num_modes - 1):
        circuit.cnot(i, i + 1)
        circuit.rz(i + 1, 0.1)
        circuit.cnot(i, i + 1)
    
    circuit.measure_all()
    return circuit

field_amps = np.random.rand(4)
circuit = qft_field_circuit(4, field_amps)
result = QuantumRun(circuit, shots=1000)
print(f"  Quantum field theory: {len(result.counts)} states")

print("\nApplications:")
print("  - Condensed matter physics")
print("  - High-energy physics")
print("  - Cosmology")
print("  - Materials science")
