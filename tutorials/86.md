```python
import math
import numpy as np
from abirqu import Circuit, H, X, Y, Z
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Hamiltonian simulation
# ============================================

print("Expert Quantum Simulation (Part 2):")
print("=" * 50)

def hamiltonian_simulation_circuit(terms, time, num_qubits=4):
    """Simulate Hamiltonian evolution."""
    circuit = Circuit(num_qubits, name="HamSim")
    
    # Initial state
    circuit.h(0)
    
    # Trotter decomposition
    num_steps = 5
    dt = time / num_steps
    
    for _ in range(num_steps):
        for coeff, pauli_string in terms:
            # Apply exp(-i*coeff*pauli*dt)
            for i, pauli in enumerate(pauli_string):
                if pauli == 'X':
                    circuit.h(i)
                elif pauli == 'Y':
                    circuit.s(i)
                    circuit.h(i)
            
            # ZZ interactions
            for i in range(num_qubits - 1):
                if pauli_string[i] == 'Z' and pauli_string[i+1] == 'Z':
                    circuit.cnot(i, i + 1)
                    circuit.rz(i + 1, 2 * coeff * dt)
                    circuit.cnot(i, i + 1)
            
            # Undo basis change
            for i, pauli in enumerate(pauli_string):
                if pauli == 'X':
                    circuit.h(i)
                elif pauli == 'Y':
                    circuit.h(i)
                    circuit.s_dag(i)
    
    return circuit

# Example Hamiltonian
hamiltonian = [
    (1.0, 'ZZII'),
    (0.5, 'IZZI'),
    (0.3, 'IIZZ'),
]

print("Hamiltonian Simulation:")
print("-" * 50)

circuit = hamiltonian_simulation_circuit(hamiltonian, time=1.0)
result = QuantumRun(circuit, shots=1000)

print(f"  Terms: {len(hamiltonian)}")
print(f"  Circuit depth: {circuit.depth()}")

# ============================================
# Part 2: Simulation algorithms
# ============================================

print("\n\nSimulation Algorithms:")
print("-" * 50)

algorithms = [
    ("Trotter-Suzuki", "Product formulas", "O(poly(t))"),
    ("Randomized", "Random ordering", "O(poly(t))"),
    ("Qubitization", "Block encoding + QPE", "O(t)"),
    ("LCU", "Linear combination", "O(t)"),
    ("Quantum walk", "Walk-based", "O(t)"),
]

for algorithm, description, complexity in algorithms:
    print(f"  {algorithm:20s}: {description:25s} [{complexity}]")

# ============================================
# Part 3: Simulation complexity
# ============================================

print("\n\nSimulation Complexity:")
print("-" * 50)

print("  Hamiltonian simulation is BQP-complete")
print("  Quantum computers can simulate quantum systems efficiently")
print("  Classical simulation is #P-hard in general")

print("\nApplications:")
print("  - Quantum chemistry")
print("  - Condensed matter physics")
print("  - High-energy physics")
print("  - Quantum field theory")
```

- Decompose into quantum gates
