## Step-by-Step Code

```python
import math
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Algorithm design principles
# ============================================

print("Expert Quantum Algorithms (Part 2):")
print("=" * 50)

principles = [
    ("Superposition", "Explore all possibilities simultaneously"),
    ("Interference", "Amplify correct answers"),
    ("Entanglement", "Correlate qubits for computation"),
    ("Phase kickback", "Transfer phase to control qubit"),
    ("Amplitude amplification", "Boost success probability"),
]

for principle, description in principles:
    print(f"  {principle:25s}: {description}")

# ============================================
# Part 2: Quantum speedups
# ============================================

print("\n\nQuantum Speedups:")
print("-" * 50)

speedups = [
    ("Shor's Algorithm", "Factoring", "Exponential"),
    ("Grover's Search", "Unstructured search", "Quadratic"),
    ("HHL Algorithm", "Linear systems", "Exponential"),
    ("Quantum Simulation", "Hamiltonian simulation", "Exponential"),
    ("Quantum Walk", "Graph problems", "Polynomial"),
    ("Amplitude Estimation", "Monte Carlo", "Quadratic"),
]

for algorithm, problem, speedup in speedups:
    print(f"  {algorithm:25s}: {problem:20s} [{speedup}]")

# ============================================
# Part 3: Algorithmic techniques
# ============================================

print("\n\nAlgorithmic Techniques:")
print("-" * 50)

techniques = [
    ("Phase Estimation", "Extract eigenvalues"),
    ("Amplitude Amplification", "Boost success probability"),
    ("Quantum Walk", "Graph-based search"),
    ("Hamiltonian Simulation", "Simulate quantum systems"),
    ("Variational Methods", "NISQ-friendly algorithms"),
    ("Quantum Counting", "Count solutions"),
]

for technique, description in techniques:
    print(f"  {technique:30s}: {description}")

# Final demonstration
print("\n\nFinal Demonstration:")
print("-" * 50)

circuit = Circuit(3, name="Final")
circuit.h(0)
circuit.cnot(0, 1)
circuit.cnot(1, 2)
circuit.measure_all()

result = QuantumRun(circuit, shots=1000)
print(f"GHZ state: {result.counts}")
```
