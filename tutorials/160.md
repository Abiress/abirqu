```python
import math
import numpy as np
from abirqu import Circuit, H, X, Y, Z
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Campaign optimization
# ============================================

print("Quantum Advertising:")
print("=" * 50)

def quantum_campaign_optimization(budget, channels):
    """Quantum campaign optimization."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QCampaign")
    
    # Encode budget
    circuit.ry(0, budget[0] * math.pi)
    circuit.ry(1, budget[1] * math.pi)
    
    # Encode channels
    circuit.ry(2, channels[0] * math.pi)
    circuit.ry(3, channels[1] * math.pi)
    
    # Optimization
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

print("Quantum Advertising:")
print("-" * 50)

campaigns = [
    ("Brand", [0.6, 0.4], [0.5, 0.5]),
    ("Performance", [0.7, 0.3], [0.7, 0.3]),
    ("Launch", [0.8, 0.2], [0.8, 0.2]),
    ("Seasonal", [0.5, 0.5], [0.4, 0.6]),
]

for name, budget, channels in campaigns:
    circuit = quantum_campaign_optimization(budget, channels)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 2: Audience targeting
# ============================================

print("\n\nAudience Targeting:")
print("-" * 50)

def quantum_audience_targeting(product, audience):
    """Quantum audience targeting."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QTargeting")
    
    # Encode product
    circuit.ry(0, product[0] * math.pi)
    circuit.ry(1, product[1] * math.pi)
    
    # Encode audience
    circuit.ry(2, audience[0] * math.pi)
    circuit.ry(3, audience[1] * math.pi)
    
    # Targeting
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    
    circuit.measure_all()
    return circuit

products = [
    ("Tech", [0.8, 0.2]),
    ("Fashion", [0.5, 0.5]),
    ("Food", [0.4, 0.6]),
    ("Finance", [0.7, 0.3]),
]

audiences = [
    ("Millennials", [0.3, 0.7]),
    ("Gen Z", [0.2, 0.8]),
    ("Boomers", [0.7, 0.3]),
    ("Professionals", [0.6, 0.4]),
]

for product_name, product in products:
    for audience_name, audience in audiences:
        circuit = quantum_audience_targeting(product, audience)
        result = QuantumRun(circuit, shots=100)
        print(f"  {product_name} + {audience_name}: {result.counts}")

# ============================================
# Part 3: ROI measurement
# ============================================

print("\n\nROI Measurement:")
print("-" * 50)

def quantum_roi_measurement(spend, results):
    """Quantum ROI measurement."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QROI")
    
    # Encode spend
    circuit.ry(0, spend[0] * math.pi)
    circuit.ry(1, spend[1] * math.pi)
    
    # Encode results
    circuit.ry(2, results[0] * math.pi)
    circuit.ry(3, results[1] * math.pi)
    
    # ROI calculation
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

roi_scenarios = [
    ("High ROI", [0.3, 0.7], [0.8, 0.2]),
    ("Medium ROI", [0.5, 0.5], [0.5, 0.5]),
    ("Low ROI", [0.7, 0.3], [0.3, 0.7]),
    ("Breakeven", [0.5, 0.5], [0.5, 0.5]),
]

for name, spend, results in roi_scenarios:
    circuit = quantum_roi_measurement(spend, results)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

print("\nApplications:")
print("  - Campaign management")
print("  - Audience targeting")
print("  - Performance tracking")
print("  - Budget optimization")
```
