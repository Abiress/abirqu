## Step-by-Step Code

```python
import math
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Quantum policy network
# ============================================

print("Quantum Reinforcement Learning:")
print("=" * 50)

def quantum_policy(state, params):
    """Quantum policy: state → action probabilities."""
    circuit = Circuit(2, name="Policy")
    
    # Encode state
    for i, s in enumerate(state):
        circuit.ry(i, s * math.pi)
    
    # Parameterized policy layer
    circuit.ry(0, params[0])
    circuit.ry(1, params[1])
    
    # Entanglement
    circuit.cnot(0, 1)
    
    circuit.measure_all()
    return circuit

def get_action_probabilities(state, params):
    """Get action probabilities from quantum policy."""
    circuit = quantum_policy(state, params)
    result = QuantumRun(circuit, shots=100)
    
    # Convert to probabilities
    probs = {'left': 0, 'right': 0, 'up': 0, 'down': 0}
    for bitstring, count in result.counts.items():
        action = int(bitstring, 2) % 4
        probs[list(probs.keys())[action]] = count / 100
    
    return probs

# ============================================
# Part 2: Simple grid world
# ============================================

print("\nSimple Grid World:")
print("-" * 50)

class SimpleGridWorld:
    def __init__(self):
        self.pos = [0, 0]
        self.goal = [2, 2]
    
    def reset(self):
        self.pos = [0, 0]
        return self.pos
    
    def step(self, action):
        # Actions: 0=left, 1=right, 2=up, 3=down
        if action == 0: self.pos[0] = max(0, self.pos[0] - 1)
        elif action == 1: self.pos[0] = min(2, self.pos[0] + 1)
        elif action == 2: self.pos[1] = min(2, self.pos[1] + 1)
        elif action == 3: self.pos[1] = max(0, self.pos[1] - 1)
        
        done = self.pos == self.goal
        reward = 1.0 if done else -0.01
        return self.pos, reward, done

env = SimpleGridWorld()

# ============================================
# Part 3: Training loop
# ============================================

print("\nTraining:")
print("-" * 50)

params = [0.0, 0.0]  # Initial parameters
total_rewards = []

for episode in range(20):
    state = env.reset()
    episode_reward = 0
    
    for step in range(50):
        # Get action from quantum policy
        probs = get_action_probabilities(state, params)
        
        # Simple action selection
        import random
        actions = list(range(4))
        probs_list = [probs[a] for a in ['left', 'right', 'up', 'down']]
        action = random.choices(actions, weights=probs_list)[0]
        
        # Take step
        next_state, reward, done = env.step(action)
        episode_reward += reward
        
        # Simple policy update
        if reward > 0:
            params[action % 2] += 0.1
        else:
            params[action % 2] -= 0.01
        
        state = next_state
        if done:
            break
    
    total_rewards.append(episode_reward)
    if (episode + 1) % 5 == 0:
        avg = sum(total_rewards[-5:]) / 5
        print(f"  Episode {episode+1}: Avg reward = {avg:.3f}")

print(f"\nFinal parameters: {params}")
```

## Key Concepts

### Quantum RL Components
1. **Quantum Policy**: Parameterized circuit mapping states to actions
2. **Quantum Value**: Circuit estimating state values
3. **Quantum Advantage**: Superposition enables efficient exploration

### Quantum Exploration
- Superposition allows exploring multiple actions simultaneously
- Entanglement captures state-action correlations
- Potential speedup in learning

### Applications
- **Game playing**: Quantum agents for complex games
- **Robotics**: Quantum control policies
- **Finance**: Quantum trading strategies
- **Optimization**: Resource allocation
