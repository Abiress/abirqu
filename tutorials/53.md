```python
import math
import numpy as np
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Multi-objective optimization
# ============================================

print("Advanced Quantum Optimization:")
print("=" * 50)

def multi_objective_qaoa(objectives, p=2):
    """QAOA for multi-objective optimization."""
    num_qubits = 2
    circuit = Circuit(num_qubits, name="MultiObjQAOA")
    
    # Initial superposition
    circuit.h(0)
    circuit.h(1)
    
    # QAOA layers
    for layer in range(p):
        gamma = layer * math.pi / (2 * p)
        beta = (p - layer) * math.pi / (4 * p)
        
        # Problem Hamiltonian
        circuit.cnot(0, 1)
        circuit.rz(1, gamma)
        circuit.cnot(0, 1)
        
        # Mixer
        circuit.rx(0, beta)
        circuit.rx(1, beta)
    
    circuit.measure_all()
    return circuit

print("Multi-Objective Optimization:")
print("-" * 50)

circuit = multi_objective_qaoa(None, p=2)
result = QuantumRun(circuit, shots=1000)
print(f"Optimization results: {result.counts}")

# ============================================
# Part 2: Constrained optimization
# ============================================

print("\n\nConstrained Optimization:")
print("-" * 50)

def constrained_qaoa(constraints, num_qubits=3):
    """QAOA with constraints."""
    circuit = Circuit(num_qubits, name="ConstrainedQAOA")
    
    # Initial state satisfying constraints
    circuit.x(0)  # Fix first qubit
    
    # Superposition of valid states
    for i in range(1, num_qubits):
        circuit.h(i)
    
    # Constraint enforcement
    circuit.cnot(0, 1)
    circuit.cnot(1, 2)
    
    circuit.measure_all()
    return circuit

circuit = constrained_qaoa(None)
result = QuantumRun(circuit, shots=1000)
print(f"Constrained results: {result.counts}")

# ============================================
# Part 3: Quantum annealing schedule
# ============================================

print("\n\nQuantum Annealing Schedule:")
print("-" * 50)

def annealing_schedule(num_steps=10):
    """Simulate quantum annealing schedule."""
    schedule = []
    for i in range(num_steps):
        s = i / (num_steps - 1)  # Annealing parameter
        schedule.append(s)
    return schedule

schedule = annealing_schedule(num_steps=10)
print("Annealing schedule (s from 0 to 1):")
for i, s in enumerate(schedule):
    print(f"  Step {i}: s={s:.2f}")

print("\nAnnealing: H(s) = (1-s)H_initial + sH_problem")

# ============================================
# Part 4: Optimization landscape
# ============================================

print("\n\nOptimization Landscape:")
print("-" * 50)

def objective_function(params):
    """Example objective function."""
    x, y = params
    return (x - 1) ** 2 + (y - 2) ** 2

# Grid search
best_val = float('inf')
best_params = None

for x in np.linspace(-2, 4, 7):
    for y in np.linspace(-2, 6, 9):
        val = objective_function([x, y])
        if val < best_val:
            best_val = val
            best_params = [x, y]

print(f"Minimum found at: ({best_params[0]:.1f}, {best_params[1]:.1f})")
print(f"Minimum value: {best_val:.4f}")
print(f"Exact minimum: (1, 2) with value 0")
```

## Key Concepts

### Advanced QAOA
- **Multi-objective**: Balance multiple goals
- **Constrained**: Handle constraints
- **Higher-order**: Use higher p values

### Quantum Annealing
- **Schedule**: Slowly change Hamiltonian
- **Adiabatic theorem**: Stay in ground state
- **Tunneling**: Escape local minima

### Applications
- **Portfolio optimization**: Financial returns
- **Supply chain**: Logistics optimization
- **Drug discovery**: Molecular optimization
- **Machine learning**: Hyperparameter tuning
