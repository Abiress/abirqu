```python
import math
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Heisenberg-limited sensing
# ============================================

print("Advanced Quantum Sensing:")
print("=" * 50)

def heisenberg_sensing_circuit(num_qubits=4, phase=0.5):
    """Heisenberg-limited sensing using GHZ state."""
    circuit = Circuit(num_qubits, name="HeisenbergSense")
    
    # Create GHZ state
    circuit.h(0)
    for i in range(num_qubits - 1):
        circuit.cnot(i, i + 1)
    
    # Phase accumulation
    for i in range(num_qubits):
        circuit.rz(i, phase)
    
    # Interference
    for i in range(num_qubits - 1, -1, -1):
        if i < num_qubits - 1:
            circuit.cnot(i, i + 1)
    circuit.h(0)
    
    circuit.measure_all()
    return circuit

print("Heisenberg-Limited Sensing:")
print("-" * 50)

phase = 0.5
for n in [1, 4, 9, 16]:
    circuit = heisenberg_sensing_circuit(num_qubits=n, phase=phase)
    result = QuantumRun(circuit, shots=1000)
    
    # Estimate phase
    p0 = result.probabilities.get('0', 0.5)
    estimated = 2 * math.acos(math.sqrt(p0))
    error = abs(phase - estimated)
    
    print(f"  N={n:2d}: Error = {error:.4f} (SQL: {1/math.sqrt(n):.4f})")

# ============================================
# Part 2: Entanglement-enhanced magnetometry
# ============================================

print("\n\nEntanglement-Enhanced Magnetometry:")
print("-" * 50)

def magnetometry_circuit(num_sensors=4, field_strength=0.3):
    """Quantum magnetometry using entangled sensors."""
    circuit = Circuit(num_sensors + 1, name="Magnetometry")
    
    # Create GHZ state
    circuit.h(0)
    for i in range(num_sensors):
        circuit.cnot(i, i + 1)
    
    # Magnetic field interaction
    for i in range(num_sensors + 1):
        circuit.rz(i, field_strength)
    
    # Reverse entanglement
    for i in range(num_sensors - 1, -1, -1):
        circuit.cnot(i, i + 1)
    circuit.h(0)
    
    circuit.measure_all()
    return circuit

circuit = magnetometry_circuit(num_sensors=4, field_strength=0.3)
result = QuantumRun(circuit, shots=1000)

p0 = result.probabilities.get('0', 0.5)
estimated_field = 2 * math.acos(math.sqrt(p0))
print(f"  True field: 0.300")
print(f"  Estimated field: {estimated_field:.3f}")

# ============================================
# Part 3: Quantum sensor networks
# ============================================

print("\n\nQuantum Sensor Networks:")
print("-" * 50)

def quantum_sensor_network(num_sensors=4):
    """Distributed quantum sensing."""
    circuit = Circuit(num_sensors, name="SensorNetwork")
    
    # Create entangled sensor network
    circuit.h(0)
    for i in range(num_sensors - 1):
        circuit.cnot(i, i + 1)
    
    # Each sensor measures local field
    for i in range(num_sensors):
        circuit.rz(i, 0.1 * (i + 1))
    
    circuit.measure_all()
    return circuit

circuit = quantum_sensor_network(num_sensors=4)
result = QuantumRun(circuit, shots=1000)

print(f"  Network size: 4 sensors")
print(f"  Measurement outcomes: {len(result.counts)} unique")

print("\nApplications:")
print("  - Gravitational wave detection")
print("  - Magnetic resonance imaging")
print("  - Navigation (quantum gyroscopes)")
print("  - Geophysical exploration")
```

## Key Concepts

### Heisenberg Limit
- Best possible precision: δφ ~ 1/N
- Requires entanglement
- N-fold improvement over SQL

### Entanglement Enhancement
- GHZ states for phase sensing
- Spin-squeezed states
- Cluster states

### Sensor Networks
- Distributed quantum sensing
- Correlated measurements
- Network advantage

### Applications
- **Gravitational waves**: LIGO, LISA
- **Medical imaging**: MRI, MEG
- **Navigation**: Quantum gyroscopes
- **Exploration**: Mineral detection
