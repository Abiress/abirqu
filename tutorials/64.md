import math
import random
from abirqu import Circuit, H, X, CNOT, Ry, Rz, Measure
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Quantum model-based RL
# ============================================

print("Expert Quantum RL:")
print("=" * 50)

class QuantumModelBasedRL:
    """Model-based RL with quantum world model."""
    
    def __init__(self, state_dim=2, action_dim=2):
        self.state_dim = state_dim
        self.action_dim = action_dim
        self.world_model_params = [0.0] * 4
        self.policy_params = [0.0] * 2
    
    def predict_next_state(self, state, action):
        """Quantum world model prediction."""
        circuit = Circuit(3, name="WorldModel")
        
        # Encode state and action
        circuit.ry(0, state[0] * math.pi)
        circuit.ry(1, state[1] * math.pi)
        circuit.ry(2, action * math.pi)
        
        # Predict
        circuit.cnot(0, 1)
        circuit.cnot(1, 2)
        
        circuit.measure([0, 1])
        result = QuantumRun(circuit, shots=100)
        
        # Parse prediction
        most_common = max(result.counts, key=result.counts.get)
        return [int(b) * 0.5 for b in most_common[:2]]
    
    def get_action(self, state):
        """Get action from quantum policy."""
        circuit = Circuit(2, name="Policy")
        circuit.ry(0, state[0] * math.pi)
        circuit.ry(1, self.policy_params[0])
        circuit.cnot(0, 1)
        circuit.measure([0, 1])
        
        result = QuantumRun(circuit, shots=100)
        return int(max(result.counts, key=result.counts.get)[0])

agent = QuantumModelBasedRL()

# Test prediction
state = [0.5, 0.3]
action = 1
next_state = agent.predict_next_state(state, action)
print(f"Predicted next state: {next_state}")

def quantum_offline_circuit(dataset_size=4, num_qubits=2):
    """Quantum offline RL using dataset."""
    circuit = Circuit(num_qubits + 1, name="OfflineRL")
    
    # Encode dataset (simplified)
    circuit.h(num_qubits)
    
    # Conservative update
    for i in range(num_qubits):
        circuit.cnot(num_qubits, i)
        circuit.ry(i, 0.2)
    
    circuit.measure(list(range(num_qubits)))
    return circuit

circuit = quantum_offline_circuit()
result = QuantumRun(circuit, shots=100)
print(f"Offline RL result: {result.counts}")

# Part 3: Quantum hierarchical RL

print("\n\nQuantum Hierarchical RL:")

class QuantumHierarchicalAgent:
    """Hierarchical agent with quantum sub-policies."""
    
    def __init__(self, num_options=3):
        self.num_options = num_options
        self.option_params = [[0.0, 0.0] for _ in range(num_options)]
    
    def select_option(self, state):
        """High-level option selection."""
        circuit = Circuit(2, name="OptionSelect")
        circuit.ry(0, state[0] * math.pi)
        circuit.ry(1, 0.5)
        circuit.measure([0, 1])
        
        result = QuantumRun(circuit, shots=100)
        most_common = max(result.counts, key=result.counts.get)
        return int(most_common[0]) % self.num_options
    
    def get_action(self, state, option):
        """Low-level action selection."""
        circuit = Circuit(2, name=f"Option{option}")
        circuit.ry(0, state[0] * math.pi)
        circuit.ry(1, self.option_params[option][0])
        circuit.cnot(0, 1)
        circuit.measure([0, 1])
        
        result = QuantumRun(circuit, shots=100)
        return int(max(result.counts, key=result.counts.get)[0])

h_agent = QuantumHierarchicalAgent()
state = [0.6, 0.4]
option = h_agent.select_option(state)
action = h_agent.get_action(state, option)
print(f"Selected option: {option}, Action: {action}")

print("\nApplications:")

print("  - Resource management")

## Key Concepts

### Quantum Model-Based RL
- Learn quantum world model
- Plan using quantum simulation
- More sample efficient

### Quantum Offline RL
- Learn from fixed dataset
- No online interaction needed
- Safer for real-world applications

### Quantum Hierarchical RL
- High-level options (skills)
- Low-level actions
- Temporal abstraction
