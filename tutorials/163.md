import math
import numpy as np
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Tax optimization
# ============================================

print("Quantum Tax & Accounting:")
print("=" * 50)

def quantum_tax_optimization(income, deductions):
    """Quantum tax optimization."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QTax")
    
    # Encode income
    circuit.ry(0, income[0] * math.pi)
    circuit.ry(1, income[1] * math.pi)
    
    # Encode deductions
    circuit.ry(2, deductions[0] * math.pi)
    circuit.ry(3, deductions[1] * math.pi)
    
    # Optimization
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

print("Quantum Tax & Accounting:")
print("-" * 50)

tax_scenarios = [
    ("Individual", [0.5, 0.5], [0.4, 0.6]),
    ("Business", [0.7, 0.3], [0.6, 0.4]),
    ("Corporate", [0.8, 0.2], [0.7, 0.3]),
    ("Non-Profit", [0.3, 0.7], [0.8, 0.2]),
]

for name, income, deductions in tax_scenarios:
    circuit = quantum_tax_optimization(income, deductions)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")
