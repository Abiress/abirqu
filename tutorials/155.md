```python
import math
import numpy as np
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Reservoir simulation
# ============================================

print("Quantum Oil & Gas:")
print("=" * 50)

def quantum_reservoir_simulation(porosity, permeability):
    """Quantum reservoir simulation."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QReservoir")
    
    # Encode porosity
    circuit.ry(0, porosity * math.pi)
    
    # Encode permeability
    circuit.ry(1, permeability[0] * math.pi)
    circuit.ry(2, permeability[1] * math.pi)
    
    # Simulation
    circuit.cnot(0, 1)
    circuit.cnot(1, 2)
    
    circuit.measure_all()
    return circuit

print("Quantum Oil & Gas:")
print("-" * 50)

reservoirs = [
    ("Sandstone", 0.25, [0.8, 0.2]),
    ("Limestone", 0.15, [0.5, 0.5]),
    ("Shale", 0.05, [0.2, 0.8]),
    ("Carbonate", 0.10, [0.4, 0.6]),
]

for name, porosity, permeability in reservoirs:
    circuit = quantum_reservoir_simulation(porosity, permeability)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 2: Drilling optimization
# ============================================

print("\n\nDrilling Optimization:")
print("-" * 50)

def quantum_drilling_optimization(depth, formation):
    """Quantum drilling optimization."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QDrilling")
    
    # Encode depth
    circuit.ry(0, depth * math.pi)
    
    # Encode formation
    circuit.ry(1, formation[0] * math.pi)
    circuit.ry(2, formation[1] * math.pi)
    
    # Optimization
    circuit.cnot(0, 1)
    circuit.cnot(1, 2)
    circuit.rz(1, 0.5)
    circuit.rz(2, 0.5)
    
    circuit.measure_all()
    return circuit

drilling_scenarios = [
    ("Shallow", 0.3, [0.6, 0.4]),
    ("Medium", 0.5, [0.5, 0.5]),
    ("Deep", 0.7, [0.4, 0.6]),
    ("Ultra-Deep", 0.9, [0.3, 0.7]),
]

for name, depth, formation in drilling_scenarios:
    circuit = quantum_drilling_optimization(depth, formation)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 3: Production forecasting
# ============================================

print("\n\nProduction Forecasting:")
print("-" * 50)

def quantum_production_forecast(reservoir, operations):
    """Quantum production forecasting."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QProduction")
    
    # Encode reservoir
    circuit.ry(0, reservoir[0] * math.pi)
    circuit.ry(1, reservoir[1] * math.pi)
    
    # Encode operations
    circuit.ry(2, operations[0] * math.pi)
    circuit.ry(3, operations[1] * math.pi)
    
    # Forecasting
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

production_scenarios = [
    ("Primary", [0.8, 0.2], [0.5, 0.5]),
    ("Secondary", [0.6, 0.4], [0.7, 0.3]),
    ("Tertiary", [0.4, 0.6], [0.8, 0.2]),
    ("Enhanced", [0.3, 0.7], [0.9, 0.1]),
]

for name, reservoir, operations in production_scenarios:
    circuit = quantum_production_forecast(reservoir, operations)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

print("\nApplications:")
print("  - Reservoir modeling")
print("  - Drilling planning")
print("  - Production optimization")
print("  - Cost reduction")
```
