```python
import math
import numpy as np
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Lattice gas concept
# ============================================

print("Quantum Lattice Gas Automata:")
print("=" * 50)

class LatticeGas:
    """Simple lattice gas automaton."""
    
    def __init__(self, size=8):
        self.size = size
        self.grid = np.zeros((size, size, 4))  # 4 directions
        self.grid[size//2, size//2, :] = 1  # Initial particle
    
    def step(self):
        """One time step of evolution."""
        new_grid = np.zeros_like(self.grid)
        
        for i in range(self.size):
            for j in range(self.size):
                for d in range(4):
                    if self.grid[i, j, d] > 0:
                        # Move in direction d
                        di, dj = [(0, 1), (1, 0), (0, -1), (-1, 0)][d]
                        ni, nj = (i + di) % self.size, (j + dj) % self.size
                        new_grid[ni, nj, d] += 1
        
        # Collision (simplified)
        self.grid = new_grid
    
    def density(self):
        """Compute density field."""
        return self.grid.sum(axis=2)

print("Lattice Gas Properties:")
print("-" * 50)
print("  Grid: 8x8 with 4 directions")
print("  Particles move and collide")
print("  Macroscopic behavior: fluid dynamics")

# ============================================
# Part 2: Quantum implementation
# ============================================

print("\n\nQuantum Lattice Gas Circuit:")
print("-" * 50)

def quantum_lattice_gas_circuit(num_sites=4, steps=2):
    """Build quantum circuit for lattice gas."""
    # 2 qubits per site (position + direction)
    circuit = Circuit(2 * num_sites, name="QLGA")
    
    # Initialize
    circuit.x(0)  # Particle at site 0
    
    for _ in range(steps):
        # Streaming step (simplified)
        for i in range(num_sites - 1):
            circuit.cnot(2 * i, 2 * (i + 1))
        
        # Collision step
        for i in range(num_sites):
            circuit.h(2 * i)
    
    return circuit

circuit = quantum_lattice_gas_circuit(num_sites=4, steps=2)
result = QuantumRun(circuit, shots=1000)

print(f"Quantum lattice gas result: {result.counts}")

# ============================================
# Part 3: Fluid dynamics connection
# ============================================

print("\n\nFluid Dynamics Connection:")
print("-" * 50)

print("Lattice gas → Navier-Stokes equations")
print("  Density: ρ = Σ f_i")
print("  Momentum: ρu = Σ f_i c_i")
print("  Pressure: p = ρc_s²")

print("\nApplications:")
print("  - Weather simulation")
print("  - Aerodynamics")
print("  - Blood flow modeling")
print("  - Ocean currents")
```

## Key Concepts

### Lattice Gas Automata
1. **Particles**: Move on lattice sites
2. **Streaming**: Particles move to neighbors
3. **Collision**: Particles interact at sites
4. **Macroscopic limit**: Recovers fluid equations

### Quantum Advantage
- Exponential speedup for certain flows
- Natural for quantum turbulence
- Potential for quantum-enhanced CFD

### Applications
- **Weather prediction**: Atmospheric modeling
