```python
import math
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Quantum software engineering
# ============================================

print("Expert Quantum Software (Part 2):")
print("=" * 50)

practices = [
    ("Version Control", "Git for quantum circuits"),
    ("Testing", "Unit tests, integration tests"),
    ("Documentation", "Circuit documentation"),
    ("Code Review", "Peer review of quantum code"),
    ("CI/CD", "Automated testing and deployment"),
]

for practice, description in practices:
    print(f"  {practice:20s}: {description}")

# ============================================
# Part 2: Testing strategies
# ============================================

print("\n\nTesting Strategies:")
print("-" * 50)

test_types = [
    ("Unit Tests", "Test individual gates"),
    ("Integration Tests", "Test circuit combinations"),
    ("Simulation Tests", "Test on simulator"),
    ("Hardware Tests", "Test on real hardware"),
    ("Regression Tests", "Ensure no regressions"),
]

for test_type, description in test_types:
    print(f"  {test_type:20s}: {description}")

# ============================================
# Part 3: Deployment strategies
# ============================================

print("\n\nDeployment Strategies:")
print("-" * 50)

strategies = [
    ("Local", "Run on local simulator"),
    ("Cloud", "Run on cloud quantum computer"),
    ("Hybrid", "Combine local and cloud"),
    ("Edge", "Run on edge devices"),
    ("On-premise", "Run on local quantum hardware"),
]

for strategy, description in strategies:
    print(f"  {strategy:15s}: {description}")

# Final demonstration
print("\n\nFinal Demonstration:")
print("-" * 50)

circuit = Circuit(3, name="Final")
circuit.h(0)
circuit.cnot(0, 1)
circuit.cnot(1, 2)
circuit.measure_all()

result = QuantumRun(circuit, shots=1000)
print(f"GHZ state: {result.counts}")
```
