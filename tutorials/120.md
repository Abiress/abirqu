import math
import numpy as np
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Hamiltonian simulation
# ============================================

print("Cutting-Edge Quantum Simulation:")
print("=" * 50)

def hamiltonian_simulation_circuit(H_params, time, steps):
    """Trotterized Hamiltonian simulation."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="HamiltonSim")
    
    dt = time / steps
    
    for _ in range(steps):
        # XX interactions
        for i in range(num_qubits - 1):
            circuit.cnot(i, i + 1)
            circuit.rz(i + 1, H_params[0] * dt)
            circuit.cnot(i, i + 1)
        
        # YY interactions
        for i in range(num_qubits - 1):
            circuit.ry(i, math.pi / 2)
            circuit.ry(i + 1, math.pi / 2)
            circuit.cnot(i, i + 1)
            circuit.rz(i + 1, H_params[1] * dt)
            circuit.cnot(i, i + 1)
            circuit.ry(i, -math.pi / 2)
            circuit.ry(i + 1, -math.pi / 2)
        
        # ZZ interactions
        for i in range(num_qubits - 1):
            circuit.cnot(i, i + 1)
            circuit.rz(i + 1, H_params[2] * dt)
            circuit.cnot(i, i + 1)
    
    circuit.measure_all()
    return circuit

print("Hamiltonian Simulation:")
print("-" * 50)

H_params = [1.0, 0.5, 0.3]
time = 1.0
steps = [1, 5, 10]

for s in steps:
    circuit = hamiltonian_simulation_circuit(H_params, time, s)
    result = QuantumRun(circuit, shots=100)
    print(f"  Steps={s}: {len(result.counts)} states")

# ============================================
# Part 2: Open quantum systems
# ============================================

print("\n\nOpen Quantum Systems:")
print("-" * 50)

def lindblad_circuit(gamma, steps):
    """Lindblad master equation simulation."""
    num_qubits = 2
    circuit = Circuit(num_qubits, name="Lindblad")
    
    # Initial state
    circuit.h(0)
    circuit.x(1)
    
    # Dissipative dynamics
    for _ in range(steps):
        # Hamiltonian part
        circuit.cnot(0, 1)
        circuit.rz(1, 0.1)
        circuit.cnot(0, 1)
        
        # Dissipation
        for i in range(num_qubits):
            circuit.ry(i, gamma * 0.1)
    
    circuit.measure_all()
    return circuit

gammas = [0.1, 0.5, 1.0]
steps = 5

for gamma in gammas:
    circuit = lindblad_circuit(gamma, steps)
    result = QuantumRun(circuit, shots=100)
    print(f"  γ={gamma}: {result.counts}")

# ============================================
# Part 3: Quantum thermodynamics
# ============================================

print("\n\nQuantum Thermodynamics:")
print("-" * 50)

def quantum_heat_engine(hot_temp, cold_temp, cycles):
    """Quantum heat engine."""
    num_qubits = 2
    circuit = Circuit(num_qubits, name="HeatEngine")
    
    for cycle in range(cycles):
        # Hot bath
        circuit.ry(0, hot_temp * 0.1)
        
        # Work extraction
        circuit.cnot(0, 1)
        circuit.rz(1, (hot_temp - cold_temp) * 0.1)
        circuit.cnot(0, 1)
        
        # Cold bath
        circuit.ry(0, cold_temp * 0.1)
    
    circuit.measure_all()
    return circuit

temps = [(100, 300), (50, 300), (10, 300)]

for hot, cold in temps:
    circuit = quantum_heat_engine(hot, cold, cycles=3)
    result = QuantumRun(circuit, shots=100)
    print(f"  T_hot={hot}K: {result.counts}")

print("\nApplications:")
print("  - Materials simulation")
print("  - Chemical reactions")
print("  - Energy systems")
print("  - Quantum devices")
