```python
import math
from abirqu import Circuit, H, X, CNOT, Measure
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Quantum hardware engineering
# ============================================

print("Expert Quantum Hardware (Part 2):")
print("=" * 50)

engineering = {
    'Fabrication': {
        'techniques': 'Lithography, etching',
        'materials': 'Aluminum, niobium, silicon',
        'challenges': 'Precision, yield',
    },
    'Cryogenics': {
        'temperature': '10-20 mK (dilution fridge)',
        'cooling_power': '~1 μW',
        'challenges': 'Heat load, vibration',
    },
    'Control': {
        'electronics': 'FPGA, AWG, ADC',
        'frequency': '5-10 GHz (microwave)',
        'challenges': 'Latency, precision',
    },
    'Packaging': {
        'materials': 'Copper, aluminum',
        'shielding': 'Mu-metal, superconducting',
        'challenges': 'Wire bonding, connectivity',
    },
}

for component, properties in engineering.items():
    print(f"\n{component}:")
    for prop, value in properties.items():
        print(f"  {prop}: {value}")

# ============================================
# Part 2: Scalability challenges
# ============================================

print("\n\nScalability Challenges:")
print("-" * 50)

challenges = [
    ("Wiring", "More qubits = more cables"),
    ("Heat load", "Control signals generate heat"),
    ("Crosstalk", "Neighboring qubits interfere"),
    ("Calibration", "More qubits = longer calibration"),
    ("Yield", "Defect rate increases with size"),
]

for challenge, description in challenges:
    print(f"  {challenge:15s}: {description}")

# ============================================
# Part 3: Future directions
# ============================================

print("\n\nFuture Directions:")
print("-" * 50)

directions = [
    ("Modular architectures", "Connect multiple chips"),
    ("3D integration", "Vertical qubit stacking"),
    ("Cryogenic CMOS", "Control electronics at mK"),
    ("Integrated photonics", "On-chip optical control"),
    ("Quantum error correction", "Fault-tolerant systems"),
]

for direction, description in directions:
    print(f"  {direction:25s}: {description}")

# Final demonstration
print("\n\nFinal Demonstration:")
print("-" * 50)

circuit = Circuit(3, name="Final")
circuit.h(0)
circuit.cnot(0, 1)
circuit.cnot(1, 2)
circuit.measure([0, 1, 2])

result = QuantumRun(circuit, shots=1000)
print(f"GHZ state: {result.counts}")
```
