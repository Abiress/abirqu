import math
import numpy as np
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Blockchain consensus
# ============================================

print("Quantum Cryptocurrency:")
print("=" * 50)

def quantum_blockchain_consensus(nodes, transactions):
    """Quantum blockchain consensus."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QConsensus")
    
    # Encode nodes
    circuit.ry(0, nodes[0] * math.pi)
    circuit.ry(1, nodes[1] * math.pi)
    
    # Encode transactions
    circuit.ry(2, transactions[0] * math.pi)
    circuit.ry(3, transactions[1] * math.pi)
    
    # Consensus
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

print("Quantum Cryptocurrency:")
print("-" * 50)

consensus_types = [
    ("PoW", [0.5, 0.5], [0.5, 0.5]),
    ("PoS", [0.6, 0.4], [0.6, 0.4]),
    ("DPoS", [0.7, 0.3], [0.4, 0.6]),
    ("BFT", [0.8, 0.2], [0.7, 0.3]),
]

for name, nodes, transactions in consensus_types:
    circuit = quantum_blockchain_consensus(nodes, transactions)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 2: Transaction validation
# ============================================

print("\n\nTransaction Validation:")

def quantum_transaction_validation(sender, receiver):
    """Quantum transaction validation."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QTransaction")
    
    # Encode sender
    circuit.ry(0, sender[0] * math.pi)
    circuit.ry(1, sender[1] * math.pi)
    
    # Encode receiver
    circuit.ry(2, receiver[0] * math.pi)
    circuit.ry(3, receiver[1] * math.pi)
    
    # Validation
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

validation_scenarios = [
    ("Standard", [0.5, 0.5], [0.5, 0.5]),
    ("Multi-Sig", [0.7, 0.3], [0.6, 0.4]),
    ("Atomic", [0.6, 0.4], [0.4, 0.6]),
    ("Privacy", [0.4, 0.6], [0.3, 0.7]),
]

for name, sender, receiver in validation_scenarios:
    circuit = quantum_transaction_validation(sender, receiver)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 3: Wallet security
# ============================================

print("-" * 50)

def quantum_wallet_security(keys, access):
    """Quantum wallet security."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QWallet")
    
    # Encode keys
    circuit.ry(0, keys[0] * math.pi)
    circuit.ry(1, keys[1] * math.pi)
    
    # Encode access
    circuit.ry(2, access[0] * math.pi)
    circuit.ry(3, access[1] * math.pi)
    
    # Security
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

wallet_types = [
    ("Hot", [0.3, 0.7], [0.7, 0.3]),
    ("Cold", [0.8, 0.2], [0.2, 0.8]),
    ("Hardware", [0.9, 0.1], [0.1, 0.9]),
    ("Paper", [0.5, 0.5], [0.5, 0.5]),
]

for name, keys, access in wallet_types:
    circuit = quantum_wallet_security(keys, access)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")
