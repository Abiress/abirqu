```python
import math
import numpy as np
from abirqu import Circuit, H, X, CNOT, Ry, Rz, Measure
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Route optimization
# ============================================

print("Quantum Supply Chain:")
print("=" * 50)

def quantum_route_optimization(distances, demands):
    """Quantum route optimization (TSP)."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QRoute")
    
    # Encode distances
    for i in range(num_qubits):
        circuit.ry(i, distances[i] * math.pi)
    
    # TSP constraints
    circuit.cnot(0, 1)
    circuit.cnot(1, 2)
    circuit.cnot(2, 3)
    circuit.cnot(3, 0)
    
    # Demand constraints
    for i in range(num_qubits):
        circuit.rz(i, demands[i] * 0.5)
    
    circuit.measure(list(range(num_qubits)))
    return circuit

print("Quantum Supply Chain:")
print("-" * 50)

distances = [0.3, 0.5, 0.7, 0.2]
demands = [0.4, 0.6, 0.3, 0.5]

circuit = quantum_route_optimization(distances, demands)
result = QuantumRun(circuit, shots=1000)
print(f"  Route optimization: {result.counts}")

# ============================================
# Part 2: Inventory management
# ============================================

print("\n\nInventory Management:")
print("-" * 50)

def quantum_inventory(demand, lead_time, costs):
    """Quantum inventory optimization."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QInventory")
    
    # Encode demand
    circuit.ry(0, demand * math.pi)
    
    # Encode lead time
    circuit.ry(1, lead_time * math.pi)
    
    # Encode costs
    circuit.ry(2, costs[0] * math.pi)
    circuit.ry(3, costs[1] * math.pi)
    
    # Optimization
    circuit.cnot(0, 1)
    circuit.cnot(2, 3)
    circuit.rz(1, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure([0, 1])
    return circuit

products = [
    ("Product A", 0.8, 0.2, [0.3, 0.7]),
    ("Product B", 0.5, 0.5, [0.5, 0.5]),
    ("Product C", 0.3, 0.8, [0.7, 0.3]),
]

for name, demand, lead_time, costs in products:
    circuit = quantum_inventory(demand, lead_time, costs)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 3: Demand forecasting
# ============================================

print("\n\nDemand Forecasting:")
print("-" * 50)

def quantum_demand_forecast(historical_data, seasonality):
    """Quantum demand forecasting."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QForecast")
    
    # Encode historical data
    for i in range(4):
        circuit.ry(i, historical_data[i] * math.pi)
    
    # Seasonality
    circuit.cnot(0, 1)
    circuit.cnot(2, 3)
    
    # Trend
    circuit.rz(1, seasonality * 0.5)
    circuit.rz(3, (1 - seasonality) * 0.5)
    
    circuit.measure([0, 1])
    return circuit

historical_data = [0.6, 0.5, 0.7, 0.4]
seasonality = 0.3

circuit = quantum_demand_forecast(historical_data, seasonality)
result = QuantumRun(circuit, shots=1000)
print(f"  Demand forecast: {result.counts}")

print("\nApplications:")
print("  - Logistics planning")
print("  - Warehouse management")
print("  - Procurement optimization")
print("  - Supplier selection")
```
