```python
import math
import numpy as np
from abirqu import Circuit, H, X, CNOT, Ry, Rz, Measure
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Threat detection
# ============================================

print("Quantum Cybersecurity:")
print("=" * 50)

def quantum_threat_detection(traffic, signatures):
    """Quantum threat detection."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QThreat")
    
    # Encode traffic
    circuit.ry(0, traffic[0] * math.pi)
    circuit.ry(1, traffic[1] * math.pi)
    
    # Encode signatures
    circuit.ry(2, signatures[0] * math.pi)
    circuit.ry(3, signatures[1] * math.pi)
    
    # Pattern matching
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure([2, 3])
    return circuit

print("Quantum Cybersecurity:")
print("-" * 50)

traffic_patterns = [
    ("Normal", [0.3, 0.7]),
    ("Suspicious", [0.7, 0.3]),
    ("Malicious", [0.9, 0.1]),
    ("DDoS", [0.8, 0.2]),
]

signatures = [0.5, 0.5]

for name, traffic in traffic_patterns:
    circuit = quantum_threat_detection(traffic, signatures)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 2: Vulnerability analysis
# ============================================

print("\n\nVulnerability Analysis:")
print("-" * 50)

def quantum_vulnerability(system, attack_vectors):
    """Quantum vulnerability analysis."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QVuln")
    
    # Encode system
    circuit.ry(0, system[0] * math.pi)
    circuit.ry(1, system[1] * math.pi)
    
    # Encode attack vectors
    circuit.ry(2, attack_vectors[0] * math.pi)
    circuit.ry(3, attack_vectors[1] * math.pi)
    
    # Vulnerability assessment
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    
    circuit.measure([2, 3])
    return circuit

systems = [
    ("Web App", [0.6, 0.4], [0.5, 0.5]),
    ("Database", [0.4, 0.6], [0.4, 0.6]),
    ("Network", [0.5, 0.5], [0.6, 0.4]),
    ("Cloud", [0.3, 0.7], [0.7, 0.3]),
]

for name, system, attack_vectors in systems:
    circuit = quantum_vulnerability(system, attack_vectors)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 3: Security optimization
# ============================================

print("\n\nSecurity Optimization:")
print("-" * 50)

def quantum_security_optimization(risks, controls):
    """Quantum security optimization."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QSecurity")
    
    # Encode risks
    circuit.ry(0, risks[0] * math.pi)
    circuit.ry(1, risks[1] * math.pi)
    
    # Encode controls
    circuit.ry(2, controls[0] * math.pi)
    circuit.ry(3, controls[1] * math.pi)
    
    # Optimization
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure([2])
    return circuit

security_scenarios = [
    ("High Risk", [0.8, 0.2], [0.3, 0.7]),
    ("Medium Risk", [0.5, 0.5], [0.5, 0.5]),
    ("Low Risk", [0.2, 0.8], [0.7, 0.3]),
    ("Compliant", [0.1, 0.9], [0.9, 0.1]),
]

for name, risks, controls in security_scenarios:
    circuit = quantum_security_optimization(risks, controls)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

print("\nApplications:")
print("  - Network security")
print("  - Application security")
print("  - Data protection")
print("  - Compliance management")
```
