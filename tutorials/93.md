class EntanglementManager:
    """Manage entanglement across network."""
    
    def __init__(self):
        self.pools = {}
        self.stats = {'created': 0, 'consumed': 0, 'decayed': 0}
    
    def create(self, node1, node2, fidelity=0.9):
        """Create entanglement."""
        key = (node1, node2)
        if key not in self.pools:
            self.pools[key] = []
        self.pools[key].append({'fidelity': fidelity, 'age': 0})
        self.stats['created'] += 1
    
    def consume(self, node1, node2):
        """Consume entanglement."""
        key = (node1, node2)
        if key in self.pools and self.pools[key]:
            self.pools[key].pop()
            self.stats['consumed'] += 1
            return True
        return False
    
    def get_stats(self):
        """Get statistics."""
        return self.stats

manager = EntanglementManager()

# Create entanglement
for i in range(10):
    manager.create(0, 1)
    manager.create(1, 2)
    manager.create(2, 3)

print(f"Entanglement stats: {manager.get_stats()}")
