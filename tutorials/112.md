import math
import numpy as np
from abirqu import Circuit, H, X, CNOT, Ry, Rz, Measure
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Portfolio optimization
# ============================================

print("Advanced Quantum Optimization:")
print("=" * 50)

def portfolio_qaoa(returns, risks, correlations, budget=1.0):
    """QAOA for portfolio optimization."""
    num_assets = len(returns)
    circuit = Circuit(num_assets, name="Portfolio")
    
    # Initial superposition
    for i in range(num_assets):
        circuit.h(i)
    
    # Risk term
    for i in range(num_assets):
        for j in range(i + 1, num_assets):
            if correlations[i][j] > 0.5:
                circuit.cnot(i, j)
                circuit.rz(j, risks[i] * risks[j] * 0.1)
                circuit.cnot(i, j)
    
    # Return term
    for i in range(num_assets):
        circuit.ry(i, returns[i])
    
    circuit.measure(list(range(num_assets)))
    return circuit

returns = [0.1, 0.15, 0.12, 0.08]
risks = [0.2, 0.3, 0.25, 0.15]
correlations = np.random.rand(4, 4)
correlations = (correlations + correlations.T) / 2

print("Portfolio Optimization:")
print("-" * 50)

circuit = portfolio_qaoa(returns, risks, correlations)
result = QuantumRun(circuit, shots=1000)

print(f"  Assets: 4")
print(f"  Returns: {returns}")
print(f"  Risks: {risks}")

# ============================================
# Part 2: Supply chain optimization
# ============================================

print("\n\nSupply Chain Optimization:")
print("-" * 50)

def supply_chain_qaoa(num_locations, distances):
    """QAOA for supply chain."""
    num_qubits = num_locations * num_locations
    circuit = Circuit(num_qubits, name="SupplyChain")
    
    # Initial state
    for i in range(num_qubits):
        circuit.h(i)
    
    # Distance constraints
    for i in range(num_locations):
        for j in range(num_locations):
            if i != j:
                idx = i * num_locations + j
                circuit.ry(idx, distances[i][j] * 0.1)
    
    circuit.measure(list(range(num_qubits)))
    return circuit

distances = np.random.rand(4, 4)
distances = (distances + distances.T) / 2

circuit = supply_chain_qaoa(num_locations=4, distances=distances)
print(f"  Locations: 4")
print(f"  Qubits: {circuit.num_qubits}")

# ============================================
# Part 3: Resource allocation
# ============================================

print("\n\nResource Allocation:")
print("-" * 50)

def resource_allocation_qaoa(num_tasks, num_resources, requirements):
    """QAOA for resource allocation."""
    num_qubits = num_tasks * num_resources
    circuit = Circuit(num_qubits, name="ResourceAlloc")
    
    # Initial state
    for i in range(num_qubits):
        circuit.h(i)
    
    # Task requirements
    for i in range(num_tasks):
        for j in range(num_resources):
            idx = i * num_resources + j
            circuit.ry(idx, requirements[i][j] * 0.1)
    
    circuit.measure(list(range(num_qubits)))
    return circuit

requirements = np.random.rand(3, 4)

circuit = resource_allocation_qaoa(num_tasks=3, num_resources=4, requirements=requirements)
print(f"  Tasks: 3")
print(f"  Resources: 4")
print(f"  Qubits: {circuit.num_qubits}")

print("\nApplications:")
print("  - Financial portfolio")
print("  - Logistics optimization")
print("  - Cloud resource allocation")
print("  - Manufacturing scheduling")
