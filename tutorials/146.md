import math
import numpy as np
from abirqu import Circuit, H, X, CNOT, Ry, Rz, Measure
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Demand forecasting
# ============================================

print("Quantum Retail:")
print("=" * 50)

def quantum_demand_forecasting(history, seasonality):
    """Quantum demand forecasting."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QDemand")
    
    # Encode history
    circuit.ry(0, history[0] * math.pi)
    circuit.ry(1, history[1] * math.pi)
    
    # Encode seasonality
    circuit.ry(2, seasonality[0] * math.pi)
    circuit.ry(3, seasonality[1] * math.pi)
    
    # Forecasting
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure([2, 3])
    return circuit

print("Quantum Retail:")
print("-" * 50)

products = [
    ("Electronics", [0.8, 0.2], [0.7, 0.3]),
    ("Clothing", [0.6, 0.4], [0.5, 0.5]),
    ("Groceries", [0.5, 0.5], [0.3, 0.7]),
    ("Furniture", [0.3, 0.7], [0.4, 0.6]),
]

for name, history, seasonality in products:
    circuit = quantum_demand_forecasting(history, seasonality)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 2: Inventory optimization
# ============================================

print("\n\nInventory Optimization:")
print("-" * 50)

def quantum_inventory_optimization(stock, turnover):
    """Quantum inventory optimization."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QInventory")
    
    # Encode stock levels
    circuit.ry(0, stock[0] * math.pi)
    circuit.ry(1, stock[1] * math.pi)
    
    # Encode turnover rates
    circuit.ry(2, turnover[0] * math.pi)
    circuit.ry(3, turnover[1] * math.pi)
    
    # Optimization
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure([2, 3])
    return circuit

inventory_scenarios = [
    ("High Stock", [0.8, 0.2], [0.3, 0.7]),
    ("Optimal", [0.5, 0.5], [0.5, 0.5]),
    ("Low Stock", [0.2, 0.8], [0.7, 0.3]),
    ("Dropship", [0.1, 0.9], [0.9, 0.1]),
]

for name, stock, turnover in inventory_scenarios:
    circuit = quantum_inventory_optimization(stock, turnover)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 3: Customer segmentation
# ============================================

print("\n\nCustomer Segmentation:")
print("-" * 50)

def quantum_customer_segmentation(behavior, demographics):
    """Quantum customer segmentation."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QSegment")
    
    # Encode behavior
    circuit.ry(0, behavior[0] * math.pi)
    circuit.ry(1, behavior[1] * math.pi)
    
    # Encode demographics
    circuit.ry(2, demographics[0] * math.pi)
    circuit.ry(3, demographics[1] * math.pi)
    
    # Segmentation
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    
    circuit.measure([2, 3])
    return circuit

segments = [
    ("Premium", [0.9, 0.1], [0.8, 0.2]),
    ("Regular", [0.5, 0.5], [0.5, 0.5]),
    ("Budget", [0.2, 0.8], [0.3, 0.7]),
    ("New", [0.3, 0.7], [0.4, 0.6]),
]

for name, behavior, demographics in segments:
    circuit = quantum_customer_segmentation(behavior, demographics)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

print("\nApplications:")
print("  - Demand planning")
print("  - Stock management")
print("  - Marketing campaigns")
print("  - Customer experience")
