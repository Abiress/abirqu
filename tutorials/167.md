```python
import math
import numpy as np
from abirqu import Circuit, H, X, CNOT, Ry, Rz, Measure
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Resource allocation
# ============================================

print("Quantum Project Management:")
print("=" * 50)

def quantum_resource_allocation(resources, tasks):
    """Quantum resource allocation."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QResource")
    
    # Encode resources
    circuit.ry(0, resources[0] * math.pi)
    circuit.ry(1, resources[1] * math.pi)
    
    # Encode tasks
    circuit.ry(2, tasks[0] * math.pi)
    circuit.ry(3, tasks[1] * math.pi)
    
    # Allocation
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure([2, 3])
    return circuit

print("Quantum Project Management:")
print("-" * 50)

project_scenarios = [
    ("Software", [0.7, 0.3], [0.6, 0.4]),
    ("Construction", [0.5, 0.5], [0.5, 0.5]),
    ("Research", [0.6, 0.4], [0.7, 0.3]),
    ("Event", [0.4, 0.6], [0.4, 0.6]),
]

for name, resources, tasks in project_scenarios:
    circuit = quantum_resource_allocation(resources, tasks)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 2: Risk management
# ============================================

print("\n\nRisk Management:")
print("-" * 50)

def quantum_project_risk(risks, mitigations):
    """Quantum project risk management."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QRisk")
    
    # Encode risks
    circuit.ry(0, risks[0] * math.pi)
    circuit.ry(1, risks[1] * math.pi)
    
    # Encode mitigations
    circuit.ry(2, mitigations[0] * math.pi)
    circuit.ry(3, mitigations[1] * math.pi)
    
    # Risk assessment
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure([2])
    return circuit

risk_levels = [
    ("Low", [0.2, 0.8], [0.8, 0.2]),
    ("Medium", [0.5, 0.5], [0.5, 0.5]),
    ("High", [0.8, 0.2], [0.3, 0.7]),
    ("Critical", [0.9, 0.1], [0.2, 0.8]),
]

for name, risks, mitigations in risk_levels:
    circuit = quantum_project_risk(risks, mitigations)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 3: Schedule optimization
# ============================================

print("\n\nSchedule Optimization:")
print("-" * 50)

def quantum_schedule(dependencies, constraints):
    """Quantum schedule optimization."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QSchedule")
    
    # Encode dependencies
    circuit.ry(0, dependencies[0] * math.pi)
    circuit.ry(1, dependencies[1] * math.pi)
    
    # Encode constraints
    circuit.ry(2, constraints[0] * math.pi)
    circuit.ry(3, constraints[1] * math.pi)
    
    # Optimization
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure([2, 3])
    return circuit

schedule_scenarios = [
    ("Fast Track", [0.3, 0.7], [0.7, 0.3]),
    ("Critical Path", [0.7, 0.3], [0.8, 0.2]),
    ("Resource Constrained", [0.5, 0.5], [0.5, 0.5]),
    ("Agile", [0.4, 0.6], [0.4, 0.6]),
]

for name, dependencies, constraints in schedule_scenarios:
    circuit = quantum_schedule(dependencies, constraints)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

print("\nApplications:")
print("  - Project planning")
print("  - Resource management")
print("  - Risk mitigation")
print("  - Schedule optimization")
```
