## Step-by-Step Code

```python
import math
import numpy as np
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Quantum System Monitoring
# ============================================

print("Quantum SRE:")
print("=" * 50)

print("Quantum System Metrics:")
metrics = [
    ("Qubit Coherence Time", "T1", "100μs", "95%", "Healthy"),
    ("Gate Fidelity", "F", "99.9%", "98%", "Healthy"),
    ("Circuit Depth", "D", "50", "85%", "Warning"),
    ("Error Rate", "E", "0.1%", "92%", "Healthy"),
    ("Throughput", "TPS", "1000", "88%", "Healthy"),
    ("Queue Depth", "QD", "50", "75%", "Healthy"),
]

for metric, unit, value, sla, status in metrics:
    status_icon = "✓" if status == "Healthy" else "⚠"
    print(f"  [{status_icon}] {metric:25s}: {value:8s} (SLA: {sla})")

# Simulate monitoring check
circuit = Circuit(2, name="MonitoringCheck")
circuit.h(0)
circuit.cnot(0, 1)
circuit.measure_all()

result = QuantumRun(circuit, shots=200)
print(f"\nMonitoring check: {result.counts}")

# ============================================
# Part 2: Error Budget Management
# ============================================

print("\n\nError Budget Management:")
print("-" * 50)

error_budgets = [
    ("Qubit Errors", 0.01, 0.005, "50% remaining"),
    ("Gate Errors", 0.02, 0.012, "40% remaining"),
    ("Measurement Errors", 0.015, 0.01, "33% remaining"),
    ("Crosstalk Errors", 0.008, 0.003, "62% remaining"),
    ("Decoherence Errors", 0.012, 0.008, "33% remaining"),
]

print(f"{'Error Type':20s} {'Budget':8s} {'Used':8s} {'Status'}")
print("-" * 55)
for error_type, budget, used, status in error_budgets:
    print(f"{error_type:20s} {budget:8.3f} {used:8.3f} {status}")

total_budget = sum(b for _, b, _, _ in error_budgets)
total_used = sum(u for _, _, u, _ in error_budgets)
remaining = (total_budget - total_used) / total_budget * 100
print(f"\nOverall Error Budget: {remaining:.1f}% remaining")

circuit = Circuit(3, name="ErrorBudgetTest")
circuit.h(0)
circuit.cnot(0, 1)
circuit.cnot(1, 2)
circuit.measure_all()

result = QuantumRun(circuit, shots=300)
print(f"Error budget test: {result.counts}")

# ============================================
# Part 3: Incident Response Automation
# ============================================

print("\n\nIncident Response Automation:")
print("-" * 50)

incident_severity = [
    ("SEV-1", "Critical", "Immediate", "On-call team"),
    ("SEV-2", "High", "15 minutes", "SRE team"),
    ("SEV-3", "Medium", "1 hour", "Engineering team"),
    ("SEV-4", "Low", "Next business day", "Support team"),
]

for severity, level, response, team in incident_severity:
    print(f"  {severity:8s}: {level:10s} | Response: {response:15s} | Team: {team}")

runbooks = [
    ("High Error Rate", "Switch to backup backend"),
    ("Qubit Failure", "Isolate and recalibrate"),
    ("Queue Overflow", "Scale up quantum resources"),
    ("Coherence Drop", "Reduce circuit complexity"),
]

print("\nAutomated Runbooks:")
for trigger, action in runbooks:
    print(f"  {trigger:20s} → {action}")

circuit = Circuit(4, name="IncidentTest")
circuit.h(0)
circuit.h(1)
circuit.cnot(0, 1)
circuit.cnot(2, 3)
circuit.measure_all()

result = QuantumRun(circuit, shots=400)
print(f"\nIncident test result: {result.counts}")

print("\n" + "=" * 50)
print("Quantum SRE Complete!")
print("=" * 50)
```
