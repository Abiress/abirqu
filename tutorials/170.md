import math
import numpy as np
from abirqu import Circuit, H, X, CNOT, Ry, Rz, Measure
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Customer support
# ============================================

print("Quantum Customer Service:")
print("=" * 50)

def quantum_customer_support(issue, agent):
    """Quantum customer support."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QSupport")
    
    # Encode issue
    circuit.ry(0, issue[0] * math.pi)
    circuit.ry(1, issue[1] * math.pi)
    
    # Encode agent
    circuit.ry(2, agent[0] * math.pi)
    circuit.ry(3, agent[1] * math.pi)
    
    # Support optimization
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure([2, 3])
    return circuit

print("Quantum Customer Service:")
print("-" * 50)

support_scenarios = [
    ("Technical", [0.7, 0.3], [0.6, 0.4]),
    ("Billing", [0.5, 0.5], [0.5, 0.5]),
    ("General", [0.4, 0.6], [0.4, 0.6]),
    ("Complaint", [0.8, 0.2], [0.7, 0.3]),
]

for name, issue, agent in support_scenarios:
    circuit = quantum_customer_support(issue, agent)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================

satisfaction_levels = [
