```python
import math
from abirqu import Circuit, H, X, Y, Z
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Quantum channel capacity
# ============================================

print("Expert Quantum Information (Part 2):")
print("=" * 50)

def channel_capacity(channel_type, noise=0.1):
    """Compute channel capacity."""
    if channel_type == 'depolarizing':
        p = noise
        if p == 0 or p == 1:
            return 1
        H = -p * math.log2(p) - (1-p) * math.log2(1-p)
        return 1 - H
    elif channel_type == 'erasure':
        return 1 - noise
    elif channel_type == 'amplitude_damping':
        return 1 - noise * 2
    return 0

channels = [
    ('depolarizing', 0.01),
    ('depolarizing', 0.1),
    ('erasure', 0.1),
    ('amplitude_damping', 0.05),
]

print("Channel Capacities:")
for channel, noise in channels:
    C = channel_capacity(channel, noise)
    print(f"  {channel:20s} (p={noise:.2f}): {C:.4f} bits")

# ============================================
# Part 2: Quantum error correction theory
# ============================================

print("\n\nQuantum Error Correction Theory:")
print("-" * 50)

qec_theory = [
    ("Knill-Laflamme", "Necessary and sufficient conditions for QEC"),
    ("Quantum Singleton", "n - k ≥ 2(n - d) + 2"),
    ("Quantum Hamming", "Upper bound on code distance"),
    ("Quantum Gilbert-Varshamov", "Lower bound on code existence"),
    ("Quantum Plotkin", "Upper bound on code size"),
]

for theorem, description in qec_theory:
    print(f"  {theorem:25s}: {description}")

# ============================================
# Part 3: Quantum Shannon theory
# ============================================

print("\n\nQuantum Shannon Theory:")
print("-" * 50)

shannon_results = [
    ("Holevo bound", "Maximum classical information from quantum states"),
    ("Schumacher compression", "Quantum data compression"),
    ("Quantum channel capacity", "Maximum reliable transmission rate"),
    ("Entanglement assisted", "Capacity with shared entanglement"),
    ("Quantum rate-distortion", "Optimal compression with distortion"),
]

for result, description in shannon_results:
    print(f"  {result:25s}: {description}")

print("\nKey Formulas:")
print("  - Classical capacity: C = max[ I(ρ, ε) ]")
print("  - Quantum capacity: Q = max[ I_c(ρ, ε) ]")
print("  - Entanglement-assisted: C_E = 2 log d - S(ρ)")

# ============================================
# Part 4: Applications
# ============================================

print("\n\nApplications:")
print("-" * 50)

applications = [
    "Quantum cryptography: Security proofs",
    "Quantum communication: Optimal protocols",
    "Quantum error correction: Code design",
    "Quantum computing: Algorithm design",
    "Quantum networks: Routing optimization",
]

for app in applications:
    print(f"  - {app}")

# Final demonstration
print("\n\nFinal Demonstration:")
print("-" * 50)

circuit = Circuit(3, name="Final")
circuit.h(0)
circuit.cnot(0, 1)
circuit.cnot(1, 2)
circuit.measure_all()

result = QuantumRun(circuit, shots=1000)
print(f"GHZ state: {result.counts}")
```
