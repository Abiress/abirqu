import math
import numpy as np
from abirqu import Circuit, H, X, CNOT, Ry, Rz, Measure
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Digital quantum simulation
# ============================================

print("Expert Quantum Simulation:")
print("=" * 50)

def digital_simulation_circuit(hamiltonian_terms, time, num_qubits=4):
    """Digital simulation using Trotter."""
    circuit = Circuit(num_qubits, name="DigitalSim")
    
    # Initial state
    circuit.h(0)
    
    # Trotter steps
    num_steps = 10
    dt = time / num_steps
    
    for _ in range(num_steps):
        for coeff, (i, j), pauli in hamiltonian_terms:
            # Apply exp(-i*coeff*pauli*dt)
            if pauli == 'ZZ':
                circuit.cnot(i, j)
                circuit.rz(j, 2 * coeff * dt)
                circuit.cnot(i, j)
            elif pauli == 'XX':
                circuit.h(i)
                circuit.h(j)
                circuit.cnot(i, j)
                circuit.rz(j, 2 * coeff * dt)
                circuit.cnot(i, j)
                circuit.h(i)
                circuit.h(j)
    
    return circuit

# Example Hamiltonian
hamiltonian = [
    (1.0, (0, 1), 'ZZ'),
    (0.5, (1, 2), 'XX'),
    (0.3, (2, 3), 'ZZ'),
]

print("Digital Quantum Simulation:")
print("-" * 50)

circuit = digital_simulation_circuit(hamiltonian, time=1.0)
result = QuantumRun(circuit, shots=1000)
print(f"Circuit depth: {circuit.depth}")
print(f"Number of qubits: {circuit.num_qubits}")

# ============================================

print("\n\nAnalog Quantum Simulation:")

def analog_simulation_circuit(num_qubits=4, interaction_strength=1.0):
    """Analog simulation using natural interactions."""
    circuit = Circuit(num_qubits, name="AnalogSim")
    
    # Initialize
    for i in range(num_qubits):
        circuit.h(i)
    
    # Natural evolution (simplified)
    for _ in range(5):
        for i in range(num_qubits - 1):
            # ZZ interaction
            circuit.cnot(i, i + 1)
            circuit.rz(i + 1, interaction_strength * 0.1)
            circuit.cnot(i, i + 1)
        
        # Transverse field
        for i in range(num_qubits):
            circuit.rx(i, 0.05)
    
    return circuit

circuit = analog_simulation_circuit()
print(f"Analog simulation depth: {circuit.depth}")

print("\n\nHybrid Simulation:")

def hybrid_simulation_circuit(classical_part, quantum_part):
    """Combine classical and quantum simulation."""
    # Classical part: pre-processing
    # Quantum part: simulation
    # Classical part: post-processing
    
    circuit = Circuit(4, name="HybridSim")
    
    # Encode classical result
    for i in range(4):
        circuit.ry(i, 0.5)
    
    # Quantum simulation
    for i in range(3):
        circuit.cnot(i, i + 1)
        circuit.rz(i + 1, 0.2)
    
    return circuit

circuit = hybrid_simulation_circuit(None, None)
print(f"Hybrid circuit depth: {circuit.depth}")
