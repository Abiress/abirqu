import math
import numpy as np
from abirqu import Circuit, H, X, CNOT, Ry, Rz, Measure
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Power grid management
# ============================================

print("Quantum Utilities:")
print("=" * 50)

def quantum_power_grid(generation, demand):
    """Quantum power grid management."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QGrid")
    
    # Encode generation
    circuit.ry(0, generation[0] * math.pi)
    circuit.ry(1, generation[1] * math.pi)
    
    # Encode demand
    circuit.ry(2, demand[0] * math.pi)
    circuit.ry(3, demand[1] * math.pi)
    
    # Grid balancing
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure([2])
    return circuit

print("Quantum Utilities:")
print("-" * 50)

grid_scenarios = [
    ("Peak", [0.8, 0.2], [0.9, 0.1]),
    ("Off-Peak", [0.4, 0.6], [0.3, 0.7]),
    ("Renewable", [0.6, 0.4], [0.5, 0.5]),
    ("Mixed", [0.5, 0.5], [0.6, 0.4]),
]

for name, generation, demand in grid_scenarios:
    circuit = quantum_power_grid(generation, demand)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 2: Water distribution
# ============================================

print("\n\nWater Distribution:")
print("-" * 50)

def quantum_water_distribution(supply, quality):
    """Quantum water distribution."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QWater")
    
    # Encode supply
    circuit.ry(0, supply[0] * math.pi)
    circuit.ry(1, supply[1] * math.pi)
    
    # Encode quality
    circuit.ry(2, quality[0] * math.pi)
    circuit.ry(3, quality[1] * math.pi)
    
    # Distribution optimization
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure([2, 3])
    return circuit

water_scenarios = [
    ("Residential", [0.6, 0.4], [0.8, 0.2]),
    ("Industrial", [0.4, 0.6], [0.6, 0.4]),
    ("Agricultural", [0.7, 0.3], [0.5, 0.5]),
    ("Commercial", [0.5, 0.5], [0.7, 0.3]),
]

for name, supply, quality in water_scenarios:
    circuit = quantum_water_distribution(supply, quality)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 3: Waste management
# ============================================

print("\n\nWaste Management:")
print("-" * 50)

def quantum_waste_management(collection, processing):
    """Quantum waste management."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QWaste")
    
    # Encode collection
    circuit.ry(0, collection[0] * math.pi)
    circuit.ry(1, collection[1] * math.pi)
    
    # Encode processing
    circuit.ry(2, processing[0] * math.pi)
    circuit.ry(3, processing[1] * math.pi)
    
    # Waste optimization
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure([2, 3])
    return circuit

waste_scenarios = [
    ("Recycling", [0.6, 0.4], [0.7, 0.3]),
    ("Composting", [0.4, 0.6], [0.5, 0.5]),
    ("Landfill", [0.3, 0.7], [0.3, 0.7]),
    ("Incineration", [0.5, 0.5], [0.6, 0.4]),
]

for name, collection, processing in waste_scenarios:
    circuit = quantum_waste_management(collection, processing)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

print("\nApplications:")
print("  - Power distribution")
print("  - Water supply")
print("  - Waste processing")
print("  - Infrastructure")
