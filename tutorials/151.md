import math
import numpy as np
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Autonomous driving
# ============================================

print("Quantum Automotive:")
print("=" * 50)

def quantum_autonomous_driving(perception, planning):
    """Quantum autonomous driving."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QAutoDrive")
    
    # Encode perception
    circuit.ry(0, perception[0] * math.pi)
    circuit.ry(1, perception[1] * math.pi)
    
    # Encode planning
    circuit.ry(2, planning[0] * math.pi)
    circuit.ry(3, planning[1] * math.pi)
    
    # Decision making
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

print("Quantum Automotive:")
print("-" * 50)

driving_scenarios = [
    ("Highway", [0.8, 0.2], [0.6, 0.4]),
    ("City", [0.5, 0.5], [0.4, 0.6]),
    ("Parking", [0.3, 0.7], [0.2, 0.8]),
    ("Off-Road", [0.6, 0.4], [0.7, 0.3]),
]

for name, perception, planning in driving_scenarios:
    circuit = quantum_autonomous_driving(perception, planning)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 2: Vehicle diagnostics
# ============================================

print("\n\nVehicle Diagnostics:")
print("-" * 50)

def quantum_vehicle_diagnostics(sensors, maintenance):
    """Quantum vehicle diagnostics."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QDiagnostics")
    
    # Encode sensors
    circuit.ry(0, sensors[0] * math.pi)
    circuit.ry(1, sensors[1] * math.pi)
    
    # Encode maintenance
    circuit.ry(2, maintenance[0] * math.pi)
    circuit.ry(3, maintenance[1] * math.pi)
    
    # Diagnostic analysis
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

diagnostics = [
    ("Engine", [0.7, 0.3], [0.6, 0.4]),
    ("Transmission", [0.5, 0.5], [0.5, 0.5]),
    ("Brakes", [0.6, 0.4], [0.4, 0.6]),
    ("Battery", [0.4, 0.6], [0.7, 0.3]),
]

for name, sensors, maintenance in diagnostics:
    circuit = quantum_vehicle_diagnostics(sensors, maintenance)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 3: Fleet management
# ============================================

print("\n\nFleet Management:")
print("-" * 50)

def quantum_fleet_management(vehicles, routes):
    """Quantum fleet management."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QFleetMgmt")
    
    # Encode vehicles
    circuit.ry(0, vehicles[0] * math.pi)
    circuit.ry(1, vehicles[1] * math.pi)
    
    # Encode routes
    circuit.ry(2, routes[0] * math.pi)
    circuit.ry(3, routes[1] * math.pi)
    
    # Fleet optimization
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

fleet_scenarios = [
    ("Delivery", [0.6, 0.4], [0.7, 0.3]),
    ("Ride-Share", [0.5, 0.5], [0.5, 0.5]),
    ("Logistics", [0.7, 0.3], [0.6, 0.4]),
    ("Public Transit", [0.4, 0.6], [0.4, 0.6]),
]

for name, vehicles, routes in fleet_scenarios:
    circuit = quantum_fleet_management(vehicles, routes)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

print("\nApplications:")
print("  - Self-driving cars")
print("  - Predictive maintenance")
print("  - Fleet optimization")
print("  - Traffic management")
