```python
import math
import numpy as np
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Toric code
# ============================================

print("Cutting-Edge Quantum Error Correction:")
print("=" * 50)

def toric_code_circuit(num_x, num_y):
    """Toric code on lattice."""
    num_qubits = 2 * num_x * num_y
    circuit = Circuit(min(num_qubits, 16), name="ToricCode")
    
    # Initialize
    for i in range(min(num_qubits, 16)):
        circuit.h(i)
    
    # Stabilizer measurements
    for i in range(min(num_x * num_y, 8)):
        idx1 = (2 * i) % circuit.num_qubits
        idx2 = (2 * i + 1) % circuit.num_qubits
        circuit.cnot(idx1, idx2)
    
    circuit.measure_all()
    return circuit

print("Toric Code:")
print("-" * 50)

circuit = toric_code_circuit(4, 4)
result = QuantumRun(circuit, shots=100)
print(f"  Toric code: {circuit.num_qubits} qubits")

# ============================================
# Part 2: Color code
# ============================================

print("\n\nColor Code:")
print("-" * 50)

def color_code_circuit(num_faces):
    """Color code."""
    num_qubits = num_faces * 3
    circuit = Circuit(min(num_qubits, 12), name="ColorCode")
    
    # Initialize
    for i in range(circuit.num_qubits):
        circuit.h(i)
    
    # Face stabilizers
    for face in range(min(num_faces, 4)):
        base = (face * 3) % circuit.num_qubits
        circuit.cnot(base, (base + 1) % circuit.num_qubits)
        circuit.cnot(base, (base + 2) % circuit.num_qubits)
    
    circuit.measure_all()
    return circuit

circuit = color_code_circuit(4)
result = QuantumRun(circuit, shots=100)
print(f"  Color code: {circuit.num_qubits} qubits")

# ============================================
# Part 3: Decoding algorithms
# ============================================

print("\n\nDecoding Algorithms:")
print("-" * 50)

decoders = [
    ("MWPM", "Minimum Weight Perfect Matching"),
    ("Union-Find", "Graph-based decoder"),
    ("Neural Network", "ML-based decoder"),
    ("belief_propagation", "Iterative decoder"),
    ("Renorm", "Renormalization decoder"),
]

for name, description in decoders:
    print(f"  {name:20s}: {description}")

# Demo
def demo_decoding(syndrome):
    """Demo decoding."""
    circuit = Circuit(4, name="Decoding")
    
    # Encode syndrome
    for i, bit in enumerate(syndrome[:4]):
        if bit == 1:
            circuit.x(i)
    
    # Decode
    circuit.cnot(0, 1)
    circuit.cnot(2, 3)
    circuit.cnot(1, 3)
    
    circuit.measure_all()
    return circuit

syndromes = [[1, 0, 0, 0], [0, 1, 0, 1], [1, 1, 1, 0]]
for syn in syndromes:
    circuit = demo_decoding(syn)
    result = QuantumRun(circuit, shots=100)
    print(f"  Syndrome {syn}: {result.counts}")

print("\nApplications:")
print("  - Fault-tolerant computing")
print("  - Quantum memory")
print("  - Reliable computation")
print("  - Distributed quantum computing")
```

## Key Concepts

### Toric Code
- Topological protection
- Surface code on torus
- Efficient syndrome extraction

### Color Code
- Transversal gates
- Self-dual codes
- Higher-rate codes

### Decoding Algorithms
- **MWPM**: Optimal for depolarizing
- **Union-Find**: Near-linear complexity
- **Neural Network**: Fast decoding
- **Belief Propagation**: Iterative
- **Renorm**: Multi-scale

### Applications

- **Quantum memory**: Data storage
