import math
import numpy as np
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Risk assessment
# ============================================

print("Quantum Insurance:")
print("=" * 50)

def quantum_risk_assessment(profile, history):
    """Quantum risk assessment."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QRisk")
    
    # Encode profile
    circuit.ry(0, profile[0] * math.pi)
    circuit.ry(1, profile[1] * math.pi)
    
    # Encode history
    circuit.ry(2, history[0] * math.pi)
    circuit.ry(3, history[1] * math.pi)
    
    # Risk calculation
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

print("Quantum Insurance:")
print("-" * 50)

risk_profiles = [
    ("Low Risk", [0.2, 0.8], [0.1, 0.9]),
    ("Medium Risk", [0.5, 0.5], [0.5, 0.5]),
    ("High Risk", [0.8, 0.2], [0.9, 0.1]),
    ("Premium", [0.3, 0.7], [0.2, 0.8]),
]

for name, profile, history in risk_profiles:
    circuit = quantum_risk_assessment(profile, history)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 2: Claims processing
# ============================================

print("\n\nClaims Processing:")

    """Quantum claims processing."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QClaims")
    
    # Encode claim
    circuit.ry(0, claim[0] * math.pi)
    circuit.ry(1, claim[1] * math.pi)
    
    # Encode evidence
    circuit.ry(2, evidence[0] * math.pi)
    circuit.ry(3, evidence[1] * math.pi)
    
    # Validation
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

claims = [
    ("Auto", [0.6, 0.4], [0.7, 0.3]),
    ("Health", [0.5, 0.5], [0.5, 0.5]),
    ("Property", [0.4, 0.6], [0.3, 0.7]),
    ("Life", [0.3, 0.7], [0.4, 0.6]),
]

for name, claim, evidence in claims:
    circuit = quantum_claims_processing(claim, evidence)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 3: Policy optimization
# ============================================

print("\n\nPolicy Optimization:")
print("-" * 50)

def quantum_policy_optimization(demand, pricing):
    """Quantum policy optimization."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QPolicy")
    
    # Encode demand
    circuit.ry(0, demand[0] * math.pi)
    circuit.ry(1, demand[1] * math.pi)
    
    # Encode pricing
    circuit.ry(2, pricing[0] * math.pi)
    circuit.ry(3, pricing[1] * math.pi)
    
    # Optimization
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

policies = [
    ("Term Life", [0.6, 0.4], [0.5, 0.5]),
    ("Whole Life", [0.4, 0.6], [0.6, 0.4]),
    ("Auto", [0.7, 0.3], [0.4, 0.6]),
    ("Health", [0.5, 0.5], [0.7, 0.3]),
]

for name, demand, pricing in policies:
    circuit = quantum_policy_optimization(demand, pricing)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

print("\nApplications:")

print("  - Pricing models")
