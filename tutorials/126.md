```python
import math
import numpy as np
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Quantum cryptanalysis
# ============================================

print("Quantum Defense Applications:")
print("=" * 50)

def quantum_cryptanalysis(ciphertext, key_bits):
    """Quantum cryptanalysis demonstration."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QCryptanalysis")
    
    # Encode ciphertext
    circuit.ry(0, ciphertext[0] * math.pi)
    circuit.ry(1, ciphertext[1] * math.pi)
    
    # Encode key bits
    circuit.ry(2, key_bits[0] * math.pi)
    circuit.ry(3, key_bits[1] * math.pi)
    
    # Key recovery
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    
    circuit.measure_all()
    return circuit

print("Quantum Defense Applications:")
print("-" * 50)

ciphertexts = [
    ("Cipher1", [0.5, 0.3]),
    ("Cipher2", [0.8, 0.2]),
    ("Cipher3", [0.2, 0.7]),
]

key_bits = [0.6, 0.4]

for name, cipher in ciphertexts:
    circuit = quantum_cryptanalysis(cipher, key_bits)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 2: Secure communications
# ============================================

print("\n\nSecure Communications:")
print("-" * 50)

def quantum_secure_channel(message, key):
    """Quantum secure channel."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QSecureChannel")
    
    # Encode message
    circuit.ry(0, message[0] * math.pi)
    circuit.ry(1, message[1] * math.pi)
    
    # Encode key
    circuit.ry(2, key[0] * math.pi)
    circuit.ry(3, key[1] * math.pi)
    
    # Encryption
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

messages = [
    ("Secret1", [0.3, 0.7]),
    ("Secret2", [0.8, 0.2]),
    ("Secret3", [0.5, 0.5]),
]

keys = [
    ("Key1", [0.6, 0.4]),
    ("Key2", [0.4, 0.6]),
]

for msg_name, message in messages:
    for key_name, key in keys:
        circuit = quantum_secure_channel(message, key)
        result = QuantumRun(circuit, shots=100)
        print(f"  {msg_name} + {key_name}: {result.counts}")

# ============================================
# Part 3: Intelligence analysis
# ============================================

print("\n\nIntelligence Analysis:")
print("-" * 50)

def quantum_intelligence(sources, weights):
    """Quantum intelligence fusion."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QIntelligence")
    
    # Encode sources
    for i in range(4):
        circuit.ry(i, sources[i] * math.pi)
    
    # Source fusion
    circuit.cnot(0, 1)
    circuit.cnot(2, 3)
    circuit.cnot(1, 3)
    
    # Weighting
    for i in range(4):
        circuit.rz(i, weights[i])
    
    circuit.measure_all()
    return circuit

sources = [0.8, 0.6, 0.4, 0.2]
weights = [0.4, 0.3, 0.2, 0.1]

circuit = quantum_intelligence(sources, weights)
result = QuantumRun(circuit, shots=1000)
print(f"  Intelligence fusion: {result.counts}")

print("\nApplications:")
print("  - Code breaking")
print("  - Secure messaging")
print("  - Threat assessment")
print("  - Surveillance")
```
