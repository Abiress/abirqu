import math
import numpy as np
from abirqu import Circuit, H, X, CNOT, Ry, Rz, Measure
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Traveling salesman
# ============================================

print("Expert Quantum Optimization (Part 2):")
print("=" * 50)

def tsp_qaoa(distances, p=2):
    """QAOA for traveling salesman problem."""
    n = len(distances)
    num_qubits = n * n  # Binary encoding of tour
    
    circuit = Circuit(num_qubits, name="TSP")
    
    # Initial superposition
    for i in range(num_qubits):
        circuit.h(i)
    
    # QAOA layers
    for layer in range(p):
        gamma = layer * math.pi / (2 * p)
        beta = (p - layer) * math.pi / (4 * p)
        
        # Problem Hamiltonian (simplified)
        for i in range(num_qubits):
            circuit.ry(i, beta)
        
        # Mixer
        for i in range(num_qubits - 1):
            circuit.cnot(i, i + 1)
            circuit.rz(i + 1, gamma)
            circuit.cnot(i, i + 1)
    
    circuit.measure(list(range(num_qubits)))
    return circuit

# Example TSP
distances = [
    [0, 10, 15, 20],
    [10, 0, 35, 25],
    [15, 35, 0, 30],
    [20, 25, 30, 0],
]

print("Traveling Salesman Problem:")
print("-" * 50)

circuit = tsp_qaoa(distances, p=2)
result = QuantumRun(circuit, shots=1000)

print(f"  Cities: 4")
print(f"  Distance matrix: 4x4")
print(f"  Qubits used: {circuit.num_qubits}")
print(f"  Measurement outcomes: {len(result.counts)}")

def scheduling_qaoa(jobs, machines):
    """QAOA for job shop scheduling."""
    num_qubits = jobs * machines
    
    circuit = Circuit(num_qubits, name="Schedule")
    
    # Initial state
    for i in range(num_qubits):
        circuit.h(i)
    
    # Encoding constraints
    for i in range(jobs):
        for j in range(machines):
            idx = i * machines + j
            circuit.ry(idx, 0.3)
    
    circuit.measure(list(range(num_qubits)))
    return circuit

circuit = scheduling_qaoa(jobs=3, machines=2)
result = QuantumRun(circuit, shots=1000)

print(f"  Jobs: 3")
print(f"  Machines: 2")
print(f"  Qubits: {circuit.num_qubits}")

print("\n\nPortfolio Optimization:")
print("-" * 50)

def portfolio_optimization(returns, risks, correlations):
    """Quantum portfolio optimization."""
    n = len(returns)
    num_qubits = n
    
    circuit = Circuit(num_qubits, name="Portfolio")
    
    # Risk term
    for i in range(n):
        for j in range(i + 1, n):
            if correlations[i][j] > 0.5:
                circuit.cnot(i, j)
                circuit.rz(j, risks[i] * risks[j] * 0.1)
                circuit.cnot(i, j)
    
    # Return term
    for i in range(n):
        circuit.ry(i, returns[i])
    
    circuit.measure(list(range(num_qubits)))
    return circuit

returns = [0.1, 0.15, 0.12, 0.08]
risks = [0.2, 0.3, 0.25, 0.15]
correlations = np.random.rand(4, 4)
correlations = (correlations + correlations.T) / 2

circuit = portfolio_optimization(returns, risks, correlations)
print(f"  Assets: 4")
print(f"  Qubits: {circuit.num_qubits}")

print("\nApplications:")
print("  - Logistics optimization")
print("  - Manufacturing scheduling")
print("  - Financial portfolio")
print("  - Resource allocation")
