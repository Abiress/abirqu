```python
import math
import random
from abirqu import Circuit, H, X, Ry, Measure
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: BB84 Protocol
# ============================================

print("BB84 Quantum Key Distribution:")
print("=" * 50)

def bb84_alice(n_bits=8):
    """Alice's operations in BB84."""
    bits = [random.randint(0, 1) for _ in range(n_bits)]
    bases = [random.randint(0, 1) for _ in range(n_bits)]  # 0=Z, 1=X
    
    states = []
    for bit, base in zip(bits, bases):
        circuit = Circuit(1, name="BB84_Alice")
        if bit == 1:
            circuit.x(0)
        if base == 1:  # X basis
            circuit.h(0)
        states.append((circuit, bit, base))
    
    return states

def bb84_bob(states):
    """Bob's measurements in BB84."""
    bases = [random.randint(0, 1) for _ in range(len(states))]
    measurements = []
    
    for i, (circuit, _, _) in enumerate(states):
        measure_circuit = circuit.copy()
        if bases[i] == 1:  # X basis
            measure_circuit.h(0)
        measure_circuit.measure([0])
        
        result = QuantumRun(measure_circuit, shots=1)
        measured_bit = int(list(result.counts.keys())[0])
        measurements.append((measured_bit, bases[i]))
    
    return measurements

# Run BB84
print("Running BB84 protocol:")
alice_states = bb84_alice(n_bits=16)
bob_measurements = bb84_bob(alice_states)

# Sifting
print("\nSifting:")
key_alice = []
key_bob = []

for i, (_, alice_bit, alice_base) in enumerate(alice_states):
    bob_bit, bob_base = bob_measurements[i]
    
    if alice_base == bob_base:
        key_alice.append(alice_bit)
        key_bob.append(bob_bit)
        print(f"  Bit {i}: Alice basis={alice_base}, Bob basis={bob_base} "
              f"→ Match: {alice_bit == bob_bit}")

print(f"\nSifted key length: {len(key_alice)}")
print(f"Key match rate: {sum(a == b for a, b in zip(key_alice, key_bob)) / len(key_alice):.2%}")

# ============================================
# Part 2: Eavesdropping detection
# ============================================

print("\n\nEavesdropping Detection:")
print("-" * 50)

def simulate_eavesdropper(states, intercept_prob=0.1):
    """Simulate eavesdropper intercepting qubits."""
    intercepted = 0
    
    for i, (circuit, _, _) in enumerate(states):
        if random.random() < intercept_prob:
            # Eve measures in random basis
            eve_basis = random.randint(0, 1)
            measure_circuit = circuit.copy()
            if eve_basis == 1:
                measure_circuit.h(0)
            measure_circuit.measure([0])
            
            result = QuantumRun(measure_circuit, shots=1)
            intercepted += 1
    
    return intercepted

n_intercepted = simulate_eavesdropper(alice_states, intercept_prob=0.2)
print(f"Intercepted qubits: {n_intercepted}")
print("Eavesdropping introduces errors in key!")
```

## Key Concepts

### BB84 Protocol
1. **Alice** prepares qubits in random bases (Z or X)
2. **Bob** measures in random bases
3. **Sifting**: Keep only matching basis results
4. **Error estimation**: Check for eavesdropping
5. **Key distillation**: Generate final secure key

### Security
- **No-cloning theorem**: Eve cannot copy qubits
- **Measurement disturbance**: Eve's measurement introduces errors
- **Privacy amplification**: Remove partial information

### Applications
- **Quantum key distribution**: Secure communication
- **Quantum secure direct communication**: Direct message transmission
- **Quantum digital signatures**: Authentication
