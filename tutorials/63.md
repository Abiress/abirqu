def quantum_rnn_cell(input_qubit, hidden_qubits, params):
    """Quantum RNN cell."""
    circuit = Circuit(len(hidden_qubits) + 1, name="QRNN")
    
    # Input encoding
    circuit.ry(0, input_qubit * math.pi)
    
    # Hidden state update
    for i, hq in enumerate(hidden_qubits):
        circuit.cnot(0, hq)
        circuit.ry(hq, params[i])
    
    # Hidden-hidden connections
    for i in range(len(hidden_qubits) - 1):
        circuit.cnot(hidden_qubits[i], hidden_qubits[i + 1])
    
    return circuit

def quantum_encoder(sequence, num_qubits=4):
    """Quantum encoder for seq2seq."""
    circuit = Circuit(num_qubits, name="QEncoder")
    
    # Encode sequence
    for i, val in enumerate(sequence[:num_qubits]):
        circuit.ry(i, val * math.pi)
    
    # Encode dependencies
    for i in range(num_qubits - 1):
        circuit.cnot(i, i + 1)
        circuit.rz(i + 1, val * 0.1)
    
    return circuit

def quantum_decoder(encoded_state, output_length=3):

    circuit = Circuit(4, name="QDecoder")
    
    # Decode
    for i in range(output_length):
        circuit.ry(i, 0.3)
        circuit.cnot(i, (i + 1) % 4)
    
    circuit.measure([0, 1, 2])
    return circuit

def quantum_ts_anomaly(series_window, params):
    """Detect anomalies in time series window."""
    circuit = Circuit(4, name="TSAnomaly")
    
    # Encode window
    for i in range(4):
        circuit.ry(i, series_window[i] * math.pi)
    
    # Anomaly scoring
    circuit.cnot(0, 1)
    circuit.cnot(2, 3)
    circuit.cnot(1, 3)
    
    # Score
    circuit.ry(0, params[0])
    circuit.measure([0])
    return circuit
