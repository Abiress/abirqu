import math
import numpy as np
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Higher-order Trotter
# ============================================

print("Advanced Quantum Simulation:")
print("=" * 50)

def higher_order_trotter(num_qubits=4, time=0.5, order=2):
    """Implement higher-order Trotter decomposition."""
    circuit = Circuit(num_qubits, name="Trotter")
    
    # Initial state
    circuit.h(0)
    
    # Second-order Trotter
    for _ in range(order):
        # Half step of A
        for i in range(num_qubits - 1):
            circuit.cnot(i, i + 1)
            circuit.rz(i + 1, time / 2)
            circuit.cnot(i, i + 1)
        
        # Full step of B
        for i in range(num_qubits):
            circuit.rx(i, time)
        
        # Half step of A
        for i in range(num_qubits - 1):
            circuit.cnot(i, i + 1)
            circuit.rz(i + 1, time / 2)
            circuit.cnot(i, i + 1)
    
    return circuit

print("Higher-Order Trotter Decomposition:")
print("-" * 50)

for order in [1, 2, 3]:
    circuit = higher_order_trotter(order=order)
    print(f"  Order {order}: Depth = {circuit.depth()}, Error = O(t^{2*order+1})")

# ============================================
# Part 2: Linear combination of unitaries
# ============================================

print("\n\nLinear Combination of Unitaries (LCU):")
print("-" * 50)

def lcu_circuit(num_qubits=2, coefficients=None):
    """Implement LCU for Hamiltonian simulation."""
    if coefficients is None:
        coefficients = [0.5, 0.3, 0.2]
    
    circuit = Circuit(num_qubits + 1, name="LCU")
    
    # Prepare superposition of coefficients
    circuit.h(num_qubits)
    
    # Apply controlled unitaries
    for i, coeff in enumerate(coefficients):
        if coeff > 0:
            circuit.cnot(num_qubits, 0)
            circuit.ry(0, coeff * math.pi)
    
    # Uncompute
    circuit.h(num_qubits)
    
    return circuit

circuit = lcu_circuit()
print(f"  Coefficients: [0.5, 0.3, 0.2]")
print(f"  Circuit depth: {circuit.depth()}")

# ============================================

print("\n\nProduct Formulas:")

## Key Concepts

### Trotter-Suzuki
- Decompose exponential of sums into products
- Higher orders give better accuracy
- Trade depth for accuracy

### LCU
- Express Hamiltonian as linear combination
- Use ancilla to select terms
- Enables efficient simulation

### Product Formulas
- **Trotter**: Simple but low accuracy
- **Suzuki**: Symmetric, higher accuracy
- **Randomized**: Average over orderings

### Applications
- **Condensed matter**: Many-body physics
- **High-energy**: Particle physics
- **Chemistry**: Molecular dynamics
- **Finance**: Quantum Monte Carlo
