class EntanglementService:
    """Provide entanglement on demand."""
    
    def __init__(self):
        self.pools = {}
        self.requests = []
    
    def request_entanglement(self, node1, node2, fidelity=0.9):
        """Client requests entanglement."""
        self.requests.append({
            'nodes': (node1, node2),
            'fidelity': fidelity,
            'status': 'pending'
        })
        return len(self.requests) - 1
    
    def fulfill_request(self, request_id):
        """Fulfill entanglement request."""
        if request_id < len(self.requests):
            self.requests[request_id]['status'] = 'fulfilled'
            return True
        return False
    
    def get_status(self):
        """Get service status."""
        return {
            'pending': sum(1 for r in self.requests if r['status'] == 'pending'),
            'fulfilled': sum(1 for r in self.requests if r['status'] == 'fulfilled'),
        }

service = EntanglementService()

# Simulate requests
for i in range(5):
    service.request_entanglement(0, i + 1, fidelity=0.95 - i * 0.05)

for i in range(3):
    service.fulfill_request(i)

print(f"Service status: {service.get_status()}")
