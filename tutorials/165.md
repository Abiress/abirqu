```python
import math
import numpy as np
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Supplier management
# ============================================

print("Quantum Supply Chain Management:")
print("=" * 50)

def quantum_supplier_management(supplier, quality):
    """Quantum supplier management."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QSupplier")
    
    # Encode supplier
    circuit.ry(0, supplier[0] * math.pi)
    circuit.ry(1, supplier[1] * math.pi)
    
    # Encode quality
    circuit.ry(2, quality[0] * math.pi)
    circuit.ry(3, quality[1] * math.pi)
    
    # Evaluation
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

print("Quantum Supply Chain Management:")
print("-" * 50)

suppliers = [
    ("Strategic", [0.8, 0.2], [0.9, 0.1]),
    ("Tactical", [0.5, 0.5], [0.6, 0.4]),
    ("Operational", [0.3, 0.7], [0.4, 0.6]),
    ("Emergency", [0.7, 0.3], [0.5, 0.5]),
]

for name, supplier, quality in suppliers:
    circuit = quantum_supplier_management(supplier, quality)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 2: Logistics optimization
# ============================================

print("\n\nLogistics Optimization:")
print("-" * 50)

def quantum_logistics(routes, vehicles):
    """Quantum logistics optimization."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QLogistics")
    
    # Encode routes
    circuit.ry(0, routes[0] * math.pi)
    circuit.ry(1, routes[1] * math.pi)
    
    # Encode vehicles
    circuit.ry(2, vehicles[0] * math.pi)
    circuit.ry(3, vehicles[1] * math.pi)
    
    # Optimization
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

logistics_scenarios = [
    ("Last Mile", [0.7, 0.3], [0.6, 0.4]),
    ("Long Haul", [0.5, 0.5], [0.5, 0.5]),
    ("Intermodal", [0.6, 0.4], [0.7, 0.3]),
    ("Express", [0.8, 0.2], [0.8, 0.2]),
]

for name, routes, vehicles in logistics_scenarios:
    circuit = quantum_logistics(routes, vehicles)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 3: Risk mitigation
# ============================================

print("\n\nRisk Mitigation:")
print("-" * 50)

def quantum_risk_mitigation(risks, controls):
    """Quantum supply chain risk mitigation."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QRiskMit")
    
    # Encode risks
    circuit.ry(0, risks[0] * math.pi)
    circuit.ry(1, risks[1] * math.pi)
    
    # Encode controls
    circuit.ry(2, controls[0] * math.pi)
    circuit.ry(3, controls[1] * math.pi)
    
    # Mitigation
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

risk_scenarios = [
    ("Supplier Failure", [0.8, 0.2], [0.3, 0.7]),
    ("Demand Spike", [0.6, 0.4], [0.5, 0.5]),
    ("Transport Disruption", [0.7, 0.3], [0.4, 0.6]),
    ("Quality Issue", [0.5, 0.5], [0.6, 0.4]),
]

for name, risks, controls in risk_scenarios:
    circuit = quantum_risk_mitigation(risks, controls)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

print("\nApplications:")
print("  - Supplier selection")
print("  - Route optimization")
print("  - Risk management")
print("  - Cost reduction")
```
