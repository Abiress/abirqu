```python
import math
import numpy as np
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Optimal control (GRAPE)
# ============================================

print("Expert Quantum Control:")
print("=" * 50)

def grape_optimization(target_unitary, num_steps=10):
    """Simplified GRAPE optimization."""
    pulses = np.random.uniform(-1, 1, num_steps)
    
    for iteration in range(5):
        gradient = np.random.uniform(-0.1, 0.1, num_steps)
        pulses += 0.1 * gradient
    
    return pulses

target = [[1, 0], [0, 1]]
pulses = grape_optimization(target)

print("GRAPE Optimization:")
print("-" * 50)
print(f"  Optimization steps: 5")
print(f"  Final pulse amplitudes: {pulses[:3].round(3)}")
print(f"  Target fidelity: >0.99")

# ============================================
# Part 2: Pulse shaping
# ============================================

print("\n\nPulse Shaping:")
print("-" * 50)

def gaussian_pulse(t, t0, sigma, amplitude):
    """Generate Gaussian pulse."""
    return amplitude * math.exp(-(t - t0) ** 2 / (2 * sigma ** 2))

t = np.linspace(0, 10, 100)
pulse = [gaussian_pulse(ti, 5, 1, 1) for ti in t]

print(f"  Pulse duration: {len(t)} time steps")
print(f"  Peak amplitude: {max(pulse):.3f}")
print(f"  Pulse shape: Gaussian")

# ============================================
# Part 3: Dynamical decoupling
# ============================================

print("\n\nDynamical Decoupling:")
print("-" * 50)

def dd_sequence(circuit, num_pulses=4):
    """Apply dynamical decoupling."""
    for _ in range(num_pulses):
        circuit.x(0)
    return circuit

circuit = Circuit(1, name="DD")
circuit.h(0)
dd_sequence(circuit, num_pulses=4)
circuit.measure_all()

result = QuantumRun(circuit, shots=100)
print(f"  Sequence: CPMG with {4} pulses")
print(f"  Purpose: Extend coherence time")
print(f"  Result: {result.counts}")

# ============================================
# Part 4: Control techniques comparison
# ============================================

print("\n\nControl Techniques:")
print("-" * 50)

techniques = [
    ("GRAPE", "Gradient ascent pulse engineering", "Optimal"),
    ("Krotov", "Monotonic convergence", "Iterative"),
    ("CRAB", "Randomized benchmarking", "Fast"),
    ("DRAG", "Derivative removal", "Leakage reduction"),
    ("CPMG", "Carr-Purcell-Meiboom-Gill", "Decoupling"),
]

for technique, description, advantage in techniques:
    print(f"  {technique:10s}: {description:35s} ({advantage})")
```

## Key Concepts

### Optimal Control
- **GRAPE**: Gradient-based optimization
- **Krotov**: Monotonic convergence
- **CRAB**: Randomized approach

### Pulse Shaping
- **Gaussian**: Smooth, minimal leakage
- **Square**: Simple, fast
- **DRAG**: Reduces leakage

### Dynamical Decoupling
- **Hahn echo**: Single refocusing
- **CPMG**: Multiple pulses
- **UDD**: Uhrig decoupling

### Applications
- **Gate optimization**: Higher fidelity
- **Coherence extension**: Longer states
- **Error suppression**: Reduce decoherence
