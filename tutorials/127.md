```python
import math
import numpy as np
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Portfolio optimization
# ============================================

print("Quantum Financial Applications:")
print("=" * 50)

def quantum_portfolio(returns, risks, correlations):
    """Quantum portfolio optimization."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QPortfolio")
    
    # Encode returns
    for i in range(num_qubits):
        circuit.ry(i, returns[i] * math.pi)
    
    # Risk-return tradeoff
    circuit.cnot(0, 1)
    circuit.cnot(2, 3)
    circuit.cnot(1, 3)
    
    # Correlation terms
    for i in range(num_qubits):
        circuit.rz(i, risks[i] * 0.5)
    
    circuit.measure_all()
    return circuit

print("Quantum Financial Applications:")
print("-" * 50)

assets = [
    ("Stock", [0.1, 0.3]),
    ("Bond", [0.05, 0.1]),
    ("Commodity", [0.15, 0.4]),
    ("Real Estate", [0.08, 0.2]),
]

returns = [a[1][0] for a in assets]
risks = [a[1][1] for a in assets]

circuit = quantum_portfolio(returns, risks, None)
result = QuantumRun(circuit, shots=1000)
print(f"  Portfolio optimization: {result.counts}")

# ============================================
# Part 2: Risk analysis
# ============================================

print("\n\nRisk Analysis:")
print("-" * 50)

def quantum_risk_analysis(portfolio, market_data):
    """Quantum risk analysis."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QRisk")
    
    # Encode portfolio
    circuit.ry(0, portfolio[0] * math.pi)
    circuit.ry(1, portfolio[1] * math.pi)
    
    # Encode market data
    circuit.ry(2, market_data[0] * math.pi)
    circuit.ry(3, market_data[1] * math.pi)
    
    # Risk calculation
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

portfolios = [
    ("Conservative", [0.3, 0.7]),
    ("Balanced", [0.5, 0.5]),
    ("Aggressive", [0.7, 0.3]),
]

market_data = [0.6, 0.4]

for name, portfolio in portfolios:
    circuit = quantum_risk_analysis(portfolio, market_data)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 3: Algorithmic trading
# ============================================

print("\n\nAlgorithmic Trading:")
print("-" * 50)

def quantum_trading(signals, parameters):
    """Quantum trading strategy."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QTrading")
    
    # Encode signals
    circuit.ry(0, signals[0] * math.pi)
    circuit.ry(1, signals[1] * math.pi)
    
    # Encode parameters
    circuit.ry(2, parameters[0] * math.pi)
    circuit.ry(3, parameters[1] * math.pi)
    
    # Signal processing
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    
    # Decision
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    circuit.cnot(2, 3)
    
    circuit.measure_all()
    return circuit

signals = [
    ("Bullish", [0.8, 0.2]),
    ("Bearish", [0.2, 0.8]),
    ("Neutral", [0.5, 0.5]),
]

parameters = [0.6, 0.4]

for name, signal in signals:
    circuit = quantum_trading(signal, parameters)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

print("\nApplications:")
print("  - Investment optimization")
print("  - Risk management")
print("  - High-frequency trading")
print("  - Derivative pricing")
```
