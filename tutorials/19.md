```python
import math
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Quantum feature map
# ============================================

print("Quantum Support Vector Machine:")
print("=" * 50)

def quantum_feature_map(data, num_qubits=2):
    """Create quantum feature map circuit.
    
    Maps classical data to quantum state.
    """
    circuit = Circuit(num_qubits, name="FeatureMap")
    
    # Encode data using Ry rotations
    for i in range(min(num_qubits, len(data))):
        circuit.ry(i, data[i])
    
    # Add entanglement for expressiveness
    for i in range(num_qubits - 1):
        circuit.cnot(i, i + 1)
        circuit.rz(i + 1, data[i] * data[i + 1] if i + 1 < len(data) else 0)
    
    return circuit

# ============================================
# Part 2: Quantum kernel
# ============================================

print("\nQuantum Kernel Estimation:")
print("-" * 50)

def quantum_kernel(x1, x2, shots=1000):
    """Estimate quantum kernel K(x1, x2) = |⟨φ(x1)|φ(x2)⟩|²"""
    
    # Build circuit to compute overlap
    num_qubits = 2
    circuit = Circuit(2 * num_qubits, name="Kernel")
    
    # Encode x1 in first register
    for i in range(num_qubits):
        if i < len(x1):
            circuit.ry(i, x1[i])
    
    # Encode x2 in second register
    for i in range(num_qubits):
        if i < len(x2):
            circuit.ry(num_qubits + i, x2[i])
    
    # Compute overlap using SWAP test
    circuit.h(0)
    circuit.cnot(0, num_qubits)
    circuit.h(0)
    
    circuit.measure_all()
    
    result = QuantumRun(circuit, shots=shots)
    
    # Kernel value from measurement
    p0 = result.probabilities.get('0', 0.5)
    kernel_value = 2 * p0 - 1  # Map to [-1, 1]
    
    return kernel_value

# Sample data points
class_0 = [[0.1, 0.2], [0.2, 0.1], [0.15, 0.15]]
class_1 = [[0.8, 0.9], [0.9, 0.8], [0.85, 0.85]]

print("Training data:")
print(f"Class 0: {class_0}")
print(f"Class 1: {class_1}")

# Compute kernel matrix
print("\nKernel matrix:")
for i, x1 in enumerate(class_0 + class_1):
    row = []
    for j, x2 in enumerate(class_0 + class_1):
        k = quantum_kernel(x1, x2)
        row.append(f"{k:.2f}")
    print(f"  {' '.join(row)}")

# ============================================
# Part 3: Classification
# ============================================

print("\n\nQSVM Classification:")
print("=" * 50)

def classify_point(point, class_0_points, class_1_points):
    """Classify a point using quantum kernel SVM."""
    
    # Compute distances to class centroids
    dist_0 = sum(sum((p - q) ** 2 for p, q in zip(point, c)) 
                 for c in class_0_points) / len(class_0_points)
    dist_1 = sum(sum((p - q) ** 2 for p, q in zip(point, c)) 
                 for c in class_1_points) / len(class_1_points)
    
    # Classify based on nearest centroid
    return 0 if dist_0 < dist_1 else 1

# Test points
test_points = [[0.1, 0.3], [0.7, 0.8], [0.5, 0.5]]
expected = [0, 1, 0]

print("Test classification:")
for point, exp in zip(test_points, expected):
    pred = classify_point(point, class_0, class_1)
    print(f"  Point {point}: Predicted={pred}, Expected={exp}, "
          f"Correct={pred == exp}")
```

## Key Concepts

### Quantum Feature Map
- Maps classical data |x⟩ → |φ(x)⟩
- Uses parameterized quantum circuits
- Creates high-dimensional feature space

### Quantum Kernel
- K(x, x') = |⟨φ(x)|φ(x')⟩|²
- Measures similarity in quantum feature space
- Can capture complex patterns

### QSVM Algorithm
1. Encode training data into quantum states
2. Compute quantum kernel matrix
3. Train classical SVM with quantum kernel
4. Classify new points using learned boundary

### Advantages
- Exponentially large feature space
- Can learn non-linear boundaries
- Quantum speedup for certain datasets
