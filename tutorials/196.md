## Objective
Implement A/B testing frameworks for quantum models to compare performance and make data-driven decisions.

## What You'll Learn
- Quantum A/B test design
- Statistical significance testing
- Multi-armed bandit optimization

## Step-by-Step Code

```python
import math
import numpy as np
from abirqu import Circuit, H, X, CNOT, Ry, Rz, Measure
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Quantum A/B Test Design
# ============================================

print("Quantum A/B Testing:")
print("=" * 50)

print("A/B Test Configuration:")
ab_config = {
    "test_name": "Quantum Classifier v1.1 vs v1.0",
    "control": {"model": "quantum-classifier-v1.0", "qubits": 4, "traffic": 50},
    "treatment": {"model": "quantum-classifier-v1.1", "qubits": 4, "traffic": 50},
    "duration": "14 days",
    "minimum_sample_size": 10000,
    "significance_level": 0.05,
}

print(f"  Test Name: {ab_config['test_name']}")
print(f"  Control: {ab_config['control']['model']} ({ab_config['control']['traffic']}% traffic)")
print(f"  Treatment: {ab_config['treatment']['model']} ({ab_config['treatment']['traffic']}% traffic)")
print(f"  Duration: {ab_config['duration']}")
print(f"  Min Sample Size: {ab_config['minimum_sample_size']}")
print(f"  Significance Level: {ab_config['significance_level']}")

# Simulate control and treatment circuits
control_circuit = Circuit(4, name="Control")
control_circuit.h(0)
control_circuit.cnot(0, 1)
control_circuit.cnot(2, 3)
control_circuit.measure([0, 1, 2, 3])

treatment_circuit = Circuit(4, name="Treatment")
treatment_circuit.h(0)
treatment_circuit.ry(1, math.pi / 4)
treatment_circuit.cnot(0, 1)
treatment_circuit.cnot(2, 3)
treatment_circuit.measure([0, 1, 2, 3])

control_result = QuantumRun(control_circuit, shots=500)
treatment_result = QuantumRun(treatment_circuit, shots=500)

print(f"\nControl distribution: {dict(list(control_result.counts.items())[:3])}")
print(f"Treatment distribution: {dict(list(treatment_result.counts.items())[:3])}")

# ============================================
# Part 2: Statistical Significance Testing
# ============================================

print("\n\nStatistical Significance Testing:")
print("-" * 50)

test_results = [
    ("Conversion Rate", "12.5%", "14.2%", "p=0.023", "Significant"),
    ("Latency", "45ms", "42ms", "p=0.041", "Significant"),
    ("Error Rate", "2.1%", "1.8%", "p=0.156", "Not Significant"),
    ("User Satisfaction", "4.2/5", "4.5/5", "p=0.008", "Significant"),
]

print(f"{'Metric':20s} {'Control':10s} {'Treatment':10s} {'p-value':10s} {'Result'}")
print("-" * 65)
for metric, control, treatment, p_value, result in test_results:
    print(f"{metric:20s} {control:10s} {treatment:10s} {p_value:10s} {result}")

# Calculate statistical power
control_rate = 0.125
treatment_rate = 0.142
sample_size = 5000

effect_size = (treatment_rate - control_rate) / math.sqrt(control_rate * (1 - control_rate))
print(f"\nEffect Size: {effect_size:.3f}")
print(f"Statistical Power: ~85%")

confidence_intervals = [
    ("Conversion Rate", "1.2%", "2.2%", "95%"),
    ("Latency", "-5.1ms", "-0.9ms", "95%"),
    ("User Satisfaction", "0.15", "0.45", "95%"),
]

print("\nConfidence Intervals (95%):")
for metric, lower, upper, ci in confidence_intervals:
    print(f"  {metric:20s}: [{lower}, {upper}]")

circuit = Circuit(3, name="SignificanceTest")
circuit.h(0)
circuit.cnot(0, 1)
circuit.cnot(1, 2)
circuit.measure([0, 1, 2])

result = QuantumRun(circuit, shots=300)
print(f"\nSignificance test: {result.counts}")

# ============================================
# Part 3: Multi-Armed Bandit Optimization
# ============================================

print("\n\nMulti-Armed Bandit Optimization:")
print("-" * 50)

variants = [
    ("Variant-A", 0.12, 500, "Exploring"),
    ("Variant-B", 0.15, 300, "Exploiting"),
    ("Variant-C", 0.11, 100, "Exploring"),
    ("Variant-D", 0.14, 200, "Exploring"),
]

print(f"{'Variant':12s} {'CTR':6s} {'Impressions':12s} {'Strategy'}")
print("-" * 45)
for variant, ctr, impressions, strategy in variants:
    print(f"{variant:12s} {ctr:6.2f} {impressions:12d} {strategy}")

mab_algorithms = [
    ("Epsilon-Greedy", "Balance exploration and exploitation"),
    ("Thompson Sampling", "Bayesian approach to bandit problems"),
    ("UCB", "Upper Confidence Bound algorithm"),
    ("EXP3", "Exponential-weight algorithm"),
]

print("\nMAB Algorithms:")
for algorithm, description in mab_algorithms:
    print(f"  {algorithm:20s}: {description}")

circuit = Circuit(4, name="MABTest")
circuit.h(0)
circuit.ry(1, math.pi / 4)
circuit.cnot(0, 1)
circuit.rz(2, math.pi / 3)
circuit.cnot(2, 3)
circuit.measure([0, 1, 2, 3])

result = QuantumRun(circuit, shots=400)
print(f"\nMAB test: {result.counts}")

print("\n" + "=" * 50)
print("Quantum A/B Testing Complete!")
print("=" * 50)
```

- Multiple MAB algorithms for different scenarios
