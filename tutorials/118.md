import math
import numpy as np
from abirqu import Circuit, H, X, CNOT, Ry, Rz, Measure
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Quantum phase estimation
# ============================================

print("Cutting-Edge Quantum Chemistry:")
print("=" * 50)

def qpe_circuit(unitary_params, precision=4):
    """Quantum phase estimation."""
    circuit = Circuit(precision + 2, name="QPE")
    
    # Initialize precision qubits
    for i in range(precision):
        circuit.h(i)
    
    # Controlled unitaries
    for i in range(precision):
        power = 2 ** i
        circuit.ry(precision, unitary_params[0] * power)
        circuit.cnot(i, precision)
    
    # Inverse QFT
    for i in range(precision // 2):
        circuit.swap(i, precision - 1 - i)
    
    for i in range(precision):
        for j in range(i):
            circuit.rz(i, -math.pi / (2 ** (i - j)))
            circuit.cnot(j, i)
    
    circuit.measure(list(range(precision)))
    return circuit

print("Quantum Phase Estimation:")
print("-" * 50)

circuit = qpe_circuit([0.5], precision=4)
result = QuantumRun(circuit, shots=1000)
print(f"  Phase estimation: {len(result.counts)} states")

# ============================================
# Part 2: VQE for molecules
# ============================================

print("\n\nVQE for Molecules:")
print("-" * 50)

def molecular_vqe_ansatz(params, num_qubits=4):
    """VQE ansatz for molecular simulation."""
    circuit = Circuit(num_qubits, name="MolecularVQE")
    
    # Hartree-Fock state
    circuit.x(0)
    circuit.x(1)
    
    # Ansatz layers
    for layer in range(2):
        for i in range(num_qubits):
            circuit.ry(i, params[layer * num_qubits + i])
        
        for i in range(num_qubits - 1):
            circuit.cnot(i, i + 1)
    
    circuit.measure(list(range(num_qubits)))
    return circuit

molecules = [
    ("H2", [1.0, 0.5, 0.3, 0.2]),
    ("LiH", [0.8, 0.4, 0.6, 0.3]),
    ("H2O", [0.6, 0.7, 0.4, 0.5]),
]

for name, params in molecules:
    circuit = molecular_vqe_ansatz(params)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {len(result.counts)} states")

# ============================================
# Part 3: Full configuration interaction
# ============================================

print("\n\nFull Configuration Interaction:")
print("-" * 50)

def fci_circuit(num_electrons, num_orbitals):
    """Full configuration interaction."""
    num_qubits = num_orbitals
    circuit = Circuit(num_qubits, name="FCI")
    
    # Initialize electron configuration
    for i in range(num_electrons):
        circuit.x(i)
    
    # All possible excitations
    for i in range(num_electrons):
        for j in range(num_electrons, num_orbitals):
            circuit.cnot(i, j)
            circuit.ry(j, 0.5)
            circuit.cnot(i, j)
    
    circuit.measure(list(range(num_qubits)))
    return circuit

configs = [
    (2, 4),
    (3, 6),
    (4, 8),
]

for ne, no in configs:
    circuit = fci_circuit(ne, no)
    result = QuantumRun(circuit, shots=100)
    print(f"  FCI({ne},{no}): {len(result.counts)} states")

print("\nApplications:")
print("  - Molecular simulation")
print("  - Drug discovery")
print("  - Materials science")
print("  - Catalyst design")
