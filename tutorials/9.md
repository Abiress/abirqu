import math
from abirqu import Circuit
from abirqu.primitives import QuantumRun

def classical_shor_demo(N):
    """Demonstrate Shor's algorithm steps classically.
    
    This shows the quantum part (period finding) via simulation.
    """
    print(f"\nFactoring N = {N}")
    print("=" * 50)
    
    # Step 1: Choose random a < N
    a = 2  # In real algorithm, this is random
    
    # Step 2: Find gcd(a, N)
    g = math.gcd(a, N)
    if g > 1:
        print(f"gcd({a}, {N}) = {g} — Found factor directly!")
        return g, N // g
    
    print(f"Step 1: Choose a = {a}")
    print(f"Step 2: gcd({a}, {N}) = {g} (no trivial factor)")
    
    # Step 3: Find period r using quantum period finding
    # This is where the quantum computer would be used
    print(f"Step 3: Finding period of f(x) = {a}^x mod {N}")
    
    # Classical period finding (simulating quantum result)
    r = 1
    while pow(a, r, N) != 1:
        r += 1
    
    print(f"  Period r = {r}")
    print(f"  Verification: {a}^{r} mod {N} = {pow(a, r, N)}")
    
    # Step 4: Factor from period
    if r % 2 == 0:
        x = pow(a, r // 2, N)
        if x != N - 1:
            factor1 = math.gcd(x + 1, N)
            factor2 = math.gcd(x - 1, N)
            
            print(f"Step 4: a^(r/2) mod N = {x}")
            print(f"  gcd({x}+1, {N}) = {factor1}")
            print(f"  gcd({x}-1, {N}) = {factor2}")
            
            if factor1 > 1 and factor1 < N:
                print(f"\n✓ Factor found: {factor1} × {factor2} = {N}")
                return factor1, factor2
    
    print("  Period doesn't lead to factors, try different a")
    return None, None

# ============================================

print("Shor's Algorithm Demo")

for N in [15, 21, 35]:
    p, q = classical_shor_demo(N)

# For a=2, N=15, period r=4
# This shows what the quantum circuit would look like
circuit = Circuit(8, name="Period Finding")

# Counting qubits (4)
for i in range(4):
    circuit.h(i)

# Target qubit (prepared in |1⟩)
circuit.x(4)

# Controlled modular exponentiation (simplified)
# In real implementation, this uses many gates
circuit.cnot(0, 4)
circuit.cnot(1, 4)

# Measure counting qubits
circuit.measure_all()

result = QuantumRun(circuit, shots=1000)
print(f"Period finding measurements: {result.counts}")
print("Classical post-processing extracts period from these measurements")

## Key Concepts

### Shor's Algorithm Steps
1. **Random choice**: Pick random a < N
2. **GCD check**: If gcd(a,N) > 1, we found a factor
3. **Period finding**: Use quantum computer to find period r of f(x) = a^x mod N
4. **Factor extraction**: If r is even, factors are gcd(a^(r/2) ± 1, N)

### Why Quantum?
- Classical period finding: O(√N) (exponential in number of bits)
- Quantum period finding: O(n³) where n = log₂(N)
- Exponential speedup!

### RSA Implications
- RSA-2048 uses N = p × q where p, q are ~1024-bit primes
- Classical factoring: infeasible
- Shor's algorithm (if large enough quantum computer): feasible

### Current Limitations
- Requires thousands of logical qubits
- Error correction overhead is massive
- Current largest factored by quantum: 21 (IBM, 2012)
