```python
import math
import numpy as np
from abirqu import Circuit, H, X, CNOT, Ry, Rz, Measure
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Knowledge capture
# ============================================

print("Quantum Knowledge Management:")
print("=" * 50)

def quantum_knowledge_capture(expertise, documentation):
    """Quantum knowledge capture."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QKnowledge")
    
    # Encode expertise
    circuit.ry(0, expertise[0] * math.pi)
    circuit.ry(1, expertise[1] * math.pi)
    
    # Encode documentation
    circuit.ry(2, documentation[0] * math.pi)
    circuit.ry(3, documentation[1] * math.pi)
    
    # Capture
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure([2, 3])
    return circuit

print("Quantum Knowledge Management:")
print("-" * 50)

knowledge_sources = [
    ("Expert", [0.9, 0.1], [0.3, 0.7]),
    ("Document", [0.5, 0.5], [0.8, 0.2]),
    ("Process", [0.6, 0.4], [0.6, 0.4]),
    ("Tribal", [0.4, 0.6], [0.2, 0.8]),
]

for name, expertise, documentation in knowledge_sources:
    circuit = quantum_knowledge_capture(expertise, documentation)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 2: Information retrieval
# ============================================

print("\n\nInformation Retrieval:")
print("-" * 50)

def quantum_information_retrieval(query, database):
    """Quantum information retrieval."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QRetrieval")
    
    # Encode query
    circuit.ry(0, query[0] * math.pi)
    circuit.ry(1, query[1] * math.pi)
    
    # Encode database
    circuit.ry(2, database[0] * math.pi)
    circuit.ry(3, database[1] * math.pi)
    
    # Retrieval
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    
    circuit.measure([2, 3])
    return circuit

retrieval_scenarios = [
    ("Keyword", [0.5, 0.5], [0.5, 0.5]),
    ("Semantic", [0.6, 0.4], [0.6, 0.4]),
    ("Contextual", [0.7, 0.3], [0.7, 0.3]),
    ("Personalized", [0.4, 0.6], [0.4, 0.6]),
]

for name, query, database in retrieval_scenarios:
    circuit = quantum_information_retrieval(query, database)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 3: Knowledge sharing
# ============================================

print("\n\nKnowledge Sharing:")
print("-" * 50)

def quantum_knowledge_sharing(experts, learners):
    """Quantum knowledge sharing."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QSharing")
    
    # Encode experts
    circuit.ry(0, experts[0] * math.pi)
    circuit.ry(1, experts[1] * math.pi)
    
    # Encode learners
    circuit.ry(2, learners[0] * math.pi)
    circuit.ry(3, learners[1] * math.pi)
    
    # Sharing
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure([2, 3])
    return circuit

sharing_scenarios = [
    ("Mentoring", [0.8, 0.2], [0.3, 0.7]),
    ("Training", [0.6, 0.4], [0.5, 0.5]),
    ("Documentation", [0.5, 0.5], [0.6, 0.4]),
    ("Collaboration", [0.7, 0.3], [0.7, 0.3]),
]

for name, experts, learners in sharing_scenarios:
    circuit = quantum_knowledge_sharing(experts, learners)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

print("\nApplications:")
print("  - Knowledge capture")
print("  - Information systems")
print("  - Training programs")
print("  - Best practices")
```
