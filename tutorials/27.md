## Step-by-Step Code

```python
import math
import numpy as np
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Text encoding
# ============================================

print("Quantum NLP:")
print("=" * 50)

def text_to_vector(text, vocab_size=4):
    """Convert text to simple vector representation."""
    vector = [0] * vocab_size
    for char in text.lower():
        idx = ord(char) % vocab_size
        vector[idx] += 1
    # Normalize
    total = sum(vector)
    if total > 0:
        vector = [v / total for v in vector]
    return vector

# Sample texts
texts = [
    ("I love quantum computing", 1),
    ("Quantum algorithms are amazing", 1),
    ("I hate quantum computing", 0),
    ("Quantum is boring", 0),
]

print("Text samples:")
for text, label in texts:
    vec = text_to_vector(text)
    print(f"  '{text}' → {vec} (label={label})")

# ============================================
# Part 2: Quantum text similarity
# ============================================

print("\n\nQuantum Text Similarity:")
print("-" * 50)

def quantum_text_similarity(vec1, vec2):
    """Compute quantum similarity between text vectors."""
    circuit = Circuit(4, name="QTextSim")
    
    # Encode first text
    circuit.ry(0, vec1[0] * math.pi)
    circuit.ry(1, vec1[1] * math.pi)
    
    # Encode second text
    circuit.ry(2, vec2[0] * math.pi)
    circuit.ry(3, vec2[1] * math.pi)
    
    # Compute similarity
    circuit.h(0)
    circuit.cnot(0, 2)
    circuit.h(0)
    
    circuit.measure_all()
    return circuit

# Compute similarities
vec1 = text_to_vector("I love quantum")
vec2 = text_to_vector("Quantum is great")
vec3 = text_to_vector("I hate quantum")

print(f"Similarity('love quantum', 'is great'): ", end="")
circuit = quantum_text_similarity(vec1, vec2)
result = QuantumRun(circuit, shots=100)
print(f"{result.probabilities.get('0', 0.5):.3f}")

print(f"Similarity('love quantum', 'hate quantum'): ", end="")
circuit = quantum_text_similarity(vec1, vec3)
result = QuantumRun(circuit, shots=100)
print(f"{result.probabilities.get('0', 0.5):.3f}")

# ============================================
# Part 3: Quantum text classifier
# ============================================

print("\n\nQuantum Text Classifier:")
print("=" * 50)

def quantum_text_classifier(text_vec, params):
    """Classify text using quantum circuit."""
    circuit = Circuit(2, name="QTextClassifier")
    
    # Encode text
    circuit.ry(0, text_vec[0] * math.pi)
    circuit.ry(1, text_vec[1] * math.pi)
    
    # Classification layer
    circuit.ry(0, params[0])
    circuit.cnot(0, 1)
    circuit.ry(1, params[1])
    
    circuit.measure_all()
    return circuit

# Train classifier
params = [0.0, 0.0]
for epoch in range(20):
    for text, label in texts:
        vec = text_to_vector(text)
        circuit = quantum_text_classifier(vec, params)
        result = QuantumRun(circuit, shots=100)
        
        pred = result.probabilities.get('1', 0.5)
        error = pred - label
        
        params[0] -= 0.1 * error
        params[1] -= 0.1 * error

# Test
print("Test predictions:")
test_texts = ["I love quantum", "Quantum is bad", "Great quantum algorithms"]
for text in test_texts:
    vec = text_to_vector(text)
    circuit = quantum_text_classifier(vec, params)
    result = QuantumRun(circuit, shots=100)
    pred = 1 if result.probabilities.get('1', 0.5) > 0.5 else 0
    print(f"  '{text}' → {'Positive' if pred else 'Negative'}")
```

## Key Concepts

### Quantum NLP Pipeline
1. **Encode** text as quantum states
2. **Compute similarity** using quantum overlap
3. **Classify** using parameterized circuits
4. **Train** with classical optimization

### Quantum Advantages
- Exponential feature space
- Can capture complex semantics
- Natural for language data

### Applications
- **Sentiment analysis**: Classify positive/negative
- **Text classification**: Topic categorization
- **Machine translation**: Language alignment
- **Chatbots**: Intent recognition
