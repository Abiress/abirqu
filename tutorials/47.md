## Step-by-Step Code

```python
import math
from abirqu import Circuit, H, X, Z, CNOT, Measure
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Superdense coding
# ============================================

print("Quantum Communication Protocols:")
print("=" * 50)

def superdense_coding(message):
    """Encode 2 classical bits into 1 qubit."""
    circuit = Circuit(2, name="Superdense")
    
    # Create Bell pair
    circuit.h(0)
    circuit.cnot(0, 1)
    
    # Encode message
    if message == '00':
        pass  # Identity
    elif message == '01':
        circuit.x(0)
    elif message == '10':
        circuit.z(0)
    elif message == '11':
        circuit.x(0)
        circuit.z(0)
    
    return circuit

print("Superdense Coding:")
print("-" * 50)

for msg in ['00', '01', '10', '11']:
    circuit = superdense_coding(msg)
    
    # Decode
    circuit.cnot(0, 1)
    circuit.h(0)
    circuit.measure([0, 1])
    
    result = QuantumRun(circuit, shots=100)
    decoded = max(result.counts, key=result.counts.get)
    print(f"  Message: {msg} → Encoded → Decoded: {decoded}")

# ============================================
# Part 2: Quantum teleportation
# ============================================

print("\n\nQuantum Teleportation:")
print("-" * 50)

def teleportation_circuit(state_to_teleport):
    """Teleport quantum state from Alice to Bob."""
    circuit = Circuit(3, name="Teleport")
    
    # Prepare state to teleport
    if state_to_teleport == '1':
        circuit.x(0)
    
    # Create Bell pair between Alice (1) and Bob (2)
    circuit.h(1)
    circuit.cnot(1, 2)
    
    # Alice's Bell measurement
    circuit.cnot(0, 1)
    circuit.h(0)
    
    # Measure Alice's qubits
    circuit.measure([0, 1])
    
    return circuit

print("Teleportation Results:")
for state in ['0', '1']:
    circuit = teleportation_circuit(state)
    result = QuantumRun(circuit, shots=100)
    
    # Get measurement outcomes
    outcomes = result.counts
    print(f"  State |{state}⟩: Measurement outcomes = {outcomes}")
    print(f"    → Apply corrections based on outcome")

# ============================================
# Part 3: Entanglement swapping
# ============================================

print("\n\nEntanglement Swapping:")
print("-" * 50)

def entanglement_swapping():
    """Demonstrate entanglement swapping."""
    circuit = Circuit(4, name="Swap")
    
    # Create two Bell pairs
    circuit.h(0)
    circuit.cnot(0, 1)
    circuit.h(2)
    circuit.cnot(2, 3)
    
    # Bell measurement on qubits 1 and 2
    circuit.cnot(1, 2)
    circuit.h(1)
    
    # Measure
    circuit.measure([1, 2])
    
    return circuit

circuit = entanglement_swapping()
result = QuantumRun(circuit, shots=100)

print(f"Measurement outcomes: {result.counts}")
print("After corrections, qubits 0 and 3 become entangled!")
print("This enables quantum repeaters for long-distance communication.")
```

## Key Concepts

### Superdense Coding
- Send 2 classical bits using 1 qubit + shared entanglement
- Protocol: Encode → Send → Decode
- Double classical capacity

### Quantum Teleportation
- Transfer quantum state using entanglement + classical communication
- Protocol: Bell measure → Send classical bits → Apply corrections
- No cloning, original state destroyed

### Entanglement Swapping
- Entangle distant qubits via intermediate measurement
- Enables quantum repeaters
- Foundation for quantum networks

### Applications
- **Quantum internet**: Secure global communication
- **Quantum repeaters**: Long-distance quantum networks
- **Distributed computing**: Quantum cloud computing
