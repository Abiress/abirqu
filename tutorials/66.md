def ft_t_gate(circuit, data_qubit, magic_state_qubit):
    """Fault-tolerant T gate using magic state."""
    # Magic state injection
    circuit.cnot(magic_state_qubit, data_qubit)
    circuit.t(data_qubit)
    circuit.h(magic_state_qubit)
    circuit.measure(magic_state_qubit)
    return circuit

print("Fault-Tolerant Gates:")
print("-" * 50)

circuit = Circuit(2, name="FT_T")
circuit.h(0)  # Data qubit
ft_t_gate(circuit, 0, 1)
print(f"  T gate: 1 magic state, 1 measurement")

def distillation_overhead(target_error, initial_error):
    """Calculate distillation overhead."""
    rounds = 0
    error = initial_error
    
    while error > target_error:
        error *= 0.1  # 10x improvement per round
        rounds += 1
    
    return rounds, 15 ** rounds

target = 1e-12
initial = 0.01

rounds, overhead = distillation_overhead(target, initial)
print(f"  Target error: {target:.0e}")
print(f"  Initial error: {initial}")
print(f"  Distillation rounds: {rounds}")
print(f"  Resource overhead: {overhead:.2e}x")

## Key Concepts

### Fault-Tolerant Gates
- **Transversal**: Independent gate on each qubit
- **Magic state**: Enable non-Clifford gates
- **Measurement-based**: Use ancilla qubits

### Magic State Distillation
- Improve noisy magic states
- 15-to-1 protocol
- Exponential overhead but polynomial depth

### Threshold Theorem
- Below threshold: arbitrarily long computation
- Above threshold: errors accumulate
- Key result for quantum computing feasibility

### Applications
- **Cryptanalysis**: Breaking RSA
- **Simulation**: Quantum chemistry
- **Optimization**: Combinatorial problems
- **Machine learning**: Quantum advantage
