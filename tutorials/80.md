import math

# ============================================
# Part 1: Quantum channel capacity
# ============================================

print("Expert Quantum Information Theory:")
print("=" * 50)

def channel_capacity(channel_type, noise=0.1):
    """Compute channel capacity."""
    if channel_type == 'depolarizing':
        # C = 1 - H(p)
        p = noise
        if p == 0 or p == 1:
            return 1
        H = -p * math.log2(p) - (1-p) * math.log2(1-p)
        return 1 - H
    elif channel_type == 'erasure':
        # C = 1 - p
        return 1 - noise
    elif channel_type == 'amplitude_damping':
        # Simplified
        return 1 - noise * 2
    return 0

channels = [
    ('depolarizing', 0.01),
    ('depolarizing', 0.1),
    ('erasure', 0.1),
    ('amplitude_damping', 0.05),
]

print("Channel Capacities:")
for channel, noise in channels:
    C = channel_capacity(channel, noise)
    print(f"  {channel:20s} (p={noise:.2f}): {C:.4f} bits")
