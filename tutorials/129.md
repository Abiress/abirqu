```python
import math
import numpy as np
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Grid optimization
# ============================================

print("Quantum Energy Systems:")
print("=" * 50)

def quantum_grid_optimization(generation, demand, constraints):
    """Quantum power grid optimization."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QGrid")
    
    # Encode generation
    circuit.ry(0, generation[0] * math.pi)
    circuit.ry(1, generation[1] * math.pi)
    
    # Encode demand
    circuit.ry(2, demand[0] * math.pi)
    circuit.ry(3, demand[1] * math.pi)
    
    # Balance
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

print("Quantum Energy Systems:")
print("-" * 50)

generation = [0.6, 0.4]
demand = [0.7, 0.3]
constraints = [0.1, 0.2]

circuit = quantum_grid_optimization(generation, demand, constraints)
result = QuantumRun(circuit, shots=1000)
print(f"  Grid optimization: {result.counts}")

# ============================================
# Part 2: Renewable integration
# ============================================

print("\n\nRenewable Integration:")
print("-" * 50)

def quantum_renewable(solar, wind, storage):
    """Quantum renewable energy integration."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QRenewable")
    
    # Encode solar
    circuit.ry(0, solar * math.pi)
    
    # Encode wind
    circuit.ry(1, wind * math.pi)
    
    # Encode storage
    circuit.ry(2, storage[0] * math.pi)
    circuit.ry(3, storage[1] * math.pi)
    
    # Integration
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    
    # Optimization
    circuit.rz(2, 0.3)
    circuit.rz(3, 0.7)
    
    circuit.measure_all()
    return circuit

scenarios = [
    ("Sunny", 0.8, 0.2, [0.5, 0.5]),
    ("Windy", 0.2, 0.8, [0.5, 0.5]),
    ("Cloudy", 0.3, 0.3, [0.5, 0.5]),
    ("Stormy", 0.1, 0.9, [0.5, 0.5]),
]

for name, solar, wind, storage in scenarios:
    circuit = quantum_renewable(solar, wind, storage)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 3: Energy storage
# ============================================

print("\n\nEnergy Storage:")
print("-" * 50)

def quantum_energy_storage(charge, discharge, price):
    """Quantum energy storage optimization."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QStorage")
    
    # Encode charge state
    circuit.ry(0, charge * math.pi)
    
    # Encode discharge rate
    circuit.ry(1, discharge * math.pi)
    
    # Encode price
    circuit.ry(2, price[0] * math.pi)
    circuit.ry(3, price[1] * math.pi)
    
    # Optimization
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

storage_states = [
    ("Full", 0.9, 0.1, [0.8, 0.2]),
    ("Half", 0.5, 0.5, [0.5, 0.5]),
    ("Empty", 0.1, 0.9, [0.2, 0.8]),
]

for name, charge, discharge, price in storage_states:
    circuit = quantum_energy_storage(charge, discharge, price)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

print("\nApplications:")
print("  - Power grid management")
print("  - Renewable integration")
print("  - Battery optimization")
print("  - Demand response")
```
