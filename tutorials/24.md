## Step-by-Step Code

```python
import math
import numpy as np
from abirqu import Circuit, H, X, Y, Z
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Classical K-means for comparison
# ============================================

print("Quantum K-Means Clustering:")
print("=" * 50)

# Simple dataset
X = np.array([[1, 2], [2, 3], [8, 9], [9, 10], [5, 5]])

def classical_kmeans(X, k=2, max_iters=10):
    """Classical K-means."""
    centroids = X[:k].copy()
    
    for _ in range(max_iters):
        # Assign points to nearest centroid
        labels = []
        for x in X:
            distances = [np.linalg.norm(x - c) for c in centroids]
            labels.append(np.argmin(distances))
        
        # Update centroids
        for i in range(k):
            cluster_points = X[np.array(labels) == i]
            if len(cluster_points) > 0:
                centroids[i] = cluster_points.mean(axis=0)
    
    return labels, centroids

labels_classical, centroids_classical = classical_kmeans(X, k=2)
print("Classical K-means:")
print(f"Labels: {labels_classical}")
print(f"Centroids: {centroids_classical}")

# ============================================
# Part 2: Quantum distance calculation
# ============================================

print("\n\nQuantum Distance Calculation:")
print("-" * 50)

def quantum_distance_circuit(x1, x2):
    """Build circuit to compute distance between x1 and x2."""
    circuit = Circuit(4, name="QDist")
    
    # Encode first point
    circuit.ry(0, x1[0] * math.pi / 10)
    circuit.ry(1, x1[1] * math.pi / 10)
    
    # Encode second point
    circuit.ry(2, x2[0] * math.pi / 10)
    circuit.ry(3, x2[1] * math.pi / 10)
    
    # Compute overlap (distance)
    circuit.h(0)
    circuit.cnot(0, 2)
    circuit.h(0)
    
    circuit.measure_all()
    return circuit

# Compute distances quantumly
print("Quantum distances between points:")
for i in range(len(X)):
    for j in range(i + 1, len(X)):
        circuit = quantum_distance_circuit(X[i], X[j])
        result = QuantumRun(circuit, shots=100)
        overlap = result.probabilities.get('0', 0.5)
        print(f"  d({i},{j}) = {1 - overlap:.3f}")

# ============================================
# Part 3: Quantum K-means concept
# ============================================

print("\n\nQuantum K-Means Concept:")
print("=" * 50)

print("1. Encode data points in quantum states")
print("2. Use quantum circuits to compute distances")
print("3. Quantum parallelism: compute all distances simultaneously")
print("4. Update centroids using quantum averaging")
print("5. Repeat until convergence")
```

## Key Concepts

### Quantum K-Means Algorithm
1. **Encode** data points as quantum states
2. **Compute distances** using quantum overlap
3. **Assign clusters** based on quantum measurements
4. **Update centroids** using quantum averaging
5. **Repeat** until convergence

### Quantum Advantages
- Exponential speedup in distance computation
- Can handle high-dimensional data
- Natural for quantum data
