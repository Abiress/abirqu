## Step-by-Step Code

```python
import math
import random
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Quantum digital signatures
# ============================================

print("Expert Quantum Cryptography (Part 2):")
print("=" * 50)

def quantum_signature_circuit(message, private_key):
    """Create quantum digital signature."""
    circuit = Circuit(2, name="QSignature")
    
    # Encode message
    if message == 1:
        circuit.x(0)
    
    # Sign with private key
    circuit.ry(0, private_key)
    circuit.cnot(0, 1)
    
    return circuit

print("Quantum Digital Signatures:")
print("-" * 50)

# Sign a message
message = 1
private_key = random.uniform(0, 2 * math.pi)
circuit = quantum_signature_circuit(message, private_key)
circuit.measure_all()

result = QuantumRun(circuit, shots=100)
print(f"  Message: {message}")
print(f"  Signature: {result.counts}")

# ============================================
# Part 2: Quantum secret sharing
# ============================================

print("\n\nQuantum Secret Sharing:")
print("-" * 50)

def quantum_secret_sharing_circuit(secret, num_shares=3):
    """Split secret into quantum shares."""
    circuit = Circuit(num_shares, name="QSS")
    
    # Create GHZ state
    circuit.h(0)
    for i in range(num_shares - 1):
        circuit.cnot(i, i + 1)
    
    # Encode secret
    if secret == 1:
        circuit.x(0)
    
    circuit.measure_all()
    return circuit

secret = 1
circuit = quantum_secret_sharing_circuit(secret, num_shares=3)
result = QuantumRun(circuit, shots=100)

print(f"  Secret: {secret}")
print(f"  Shares: {result.counts}")
print(f"  Reconstruction: All shares needed")

# ============================================
# Part 3: Quantum oblivious transfer
# ============================================

print("\n\nQuantum Oblivious Transfer:")
print("-" * 50)

def quantum_ot_circuit(sender_messages, receiver_choice):
    """Quantum oblivious transfer."""
    circuit = Circuit(2, name="QOT")
    
    # Sender encodes both messages
    circuit.ry(0, sender_messages[0] * math.pi)
    circuit.ry(1, sender_messages[1] * math.pi)
    
    # Receiver's choice
    if receiver_choice == 1:
        circuit.x(1)
    
    circuit.measure_all()
    return circuit

messages = [0.3, 0.7]
choice = 0

circuit = quantum_ot_circuit(messages, choice)
result = QuantumRun(circuit, shots=100)

print(f"  Sender's messages: {messages}")
print(f"  Receiver's choice: {choice}")
print(f"  Result: {result.counts}")

print("\nApplications:")
print("  - Secure multi-party computation")
print("  - Privacy-preserving protocols")
print("  - Quantum money")
print("  - Quantum voting")
print("  - Secure auctions")
```

## Key Concepts

### Quantum Digital Signatures
- Sign messages using quantum states
- Unforgeable signatures
- Non-repudiation

### Quantum Secret Sharing
- Split secret into multiple shares
- All shares needed to reconstruct
- Threshold schemes possible

### Quantum Oblivious Transfer
- Send one of two messages
- Sender doesn't know which received
- Receiver only gets one

### Applications
- **Secure MPC**: Multi-party computation
- **Privacy**: Data protection
- **Money**: Unforgeable currency
- **Voting**: Secure elections
