## Step-by-Step Code

```python
import math
import numpy as np
from abirqu import Circuit, H, X, Y, Z
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: XY spin chain
# ============================================

print("Quantum Spin Chain Simulation:")
print("=" * 50)

def xy_spin_chain(num_spins=4, J=1.0, time=0.5):
    """Simulate XY model spin chain.
    
    H = J Σ (X_i X_{i+1} + Y_i Y_{i+1})
    """
    circuit = Circuit(num_spins, name="XYChain")
    
    # Initialize with single excitation
    circuit.x(0)
    
    # Trotter evolution
    for _ in range(3):
        # XX interactions
        for i in range(num_spins - 1):
            circuit.h(i)
            circuit.h(i + 1)
            circuit.cnot(i, i + 1)
            circuit.rz(i + 1, 2 * J * time)
            circuit.cnot(i, i + 1)
            circuit.h(i)
            circuit.h(i + 1)
        
        # YY interactions
        for i in range(num_spins - 1):
            circuit.s(i)
            circuit.s(i + 1)
            circuit.h(i)
            circuit.h(i + 1)
            circuit.cnot(i, i + 1)
            circuit.rz(i + 1, 2 * J * time)
            circuit.cnot(i, i + 1)
            circuit.h(i)
            circuit.h(i + 1)
            circuit.s_dag(i)
            circuit.s_dag(i + 1)
    
    return circuit

print("XY Spin Chain:")
print("-" * 50)

circuit = xy_spin_chain(num_spins=4, time=0.5)
circuit.measure_all()

result = QuantumRun(circuit, shots=1000)
print(f"State distribution: {result.counts}")

# Compute magnetization per site
mag_per_site = []
for i in range(4):
    mag = 0
    for bitstring, count in result.counts.items():
        if bitstring[i] == '0':
            mag += count
        else:
            mag -= count
    mag_per_site.append(mag / 1000)

print(f"Magnetization per site: {[f'{m:.3f}' for m in mag_per_site]}")

# ============================================
# Part 2: Heisenberg spin chain
# ============================================

print("\n\nHeisenberg Spin Chain:")
print("-" * 50)

def heisenberg_chain(num_spins=4, J=1.0, time=0.3):
    """Simulate Heisenberg model."""
    circuit = Circuit(num_spins, name="Heisenberg")
    
    # Initial state: alternating up/down
    for i in range(num_spins):
        if i % 2 == 0:
            circuit.x(i)
    
    # Trotter step
    for i in range(num_spins - 1):
        # XX
        circuit.h(i)
        circuit.h(i + 1)
        circuit.cnot(i, i + 1)
        circuit.rz(i + 1, 2 * J * time)
        circuit.cnot(i, i + 1)
        circuit.h(i)
        circuit.h(i + 1)
        
        # YY
        circuit.s(i)
        circuit.s(i + 1)
        circuit.h(i)
        circuit.h(i + 1)
        circuit.cnot(i, i + 1)
        circuit.rz(i + 1, 2 * J * time)
        circuit.cnot(i, i + 1)
        circuit.h(i)
        circuit.h(i + 1)
        circuit.s_dag(i)
        circuit.s_dag(i + 1)
        
        # ZZ
        circuit.cnot(i, i + 1)
        circuit.rz(i + 1, 2 * J * time)
        circuit.cnot(i, i + 1)
    
    return circuit

circuit = heisenberg_chain(num_spins=4, time=0.3)
circuit.measure_all()

result = QuantumRun(circuit, shots=1000)
print(f"Heisenberg chain state: {result.counts}")

# ============================================
# Part 3: Magnon propagation
# ============================================

print("\n\nMagnon Propagation:")
print("-" * 50)

print("Magnon = single spin excitation propagating through chain")
print("Dispersion relation: ω(k) = 2J(1 - cos(k))")

for k in [0, math.pi/4, math.pi/2, 3*math.pi/4, math.pi]:
    omega = 2 * (1 - math.cos(k))
    print(f"  k={k:.2f}: ω={omega:.3f}")
```

## Key Concepts

### Spin Chain Models
1. **XY Model**: Only X-X and Y-Y interactions
2. **Heisenberg Model**: X-X + Y-Y + Z-Z interactions
3. **Ising Model**: Only Z-Z interactions

### Magnons
- Quantized spin waves
- Single excitation propagating through chain
- Useful for quantum information transfer

### Applications
- **Quantum communication**: Spin chain as quantum wire
- **Magnetic materials**: Understanding magnetism
- **Quantum simulation**: Condensed matter physics
