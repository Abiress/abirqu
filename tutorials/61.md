import math
import numpy as np
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Quantum word embeddings
# ============================================

print("Advanced Quantum NLP:")
print("=" * 50)

def quantum_word_embedding(word_id, vocab_size=8, embed_dim=4):
    """Create quantum word embedding."""
    circuit = Circuit(embed_dim, name="WordEmbed")
    
    # Encode word ID
    binary = format(word_id, f'0{embed_dim}b')
    for i, bit in enumerate(reversed(binary)):
        if bit == '1':
            circuit.x(i)
    
    # Add superposition for expressiveness
    for i in range(embed_dim):
        circuit.h(i)
    
    # Entanglement
    for i in range(embed_dim - 1):
        circuit.cnot(i, i + 1)
    
    return circuit

print("Quantum Word Embeddings:")
print("-" * 50)

words = ["quantum", "computing", "algorithm", "circuit"]
for i, word in enumerate(words):
    circuit = quantum_word_embedding(i)
    result = QuantumRun(circuit, shots=100)
    print(f"  '{word}': {len(result.counts)} unique states")

# ============================================
# Part 2: Quantum attention
# ============================================

print("\n\nQuantum Attention Mechanism:")
print("-" * 50)

def quantum_attention(query, key, value):
    """Compute quantum attention."""
    circuit = Circuit(6, name="QAttention")
    
    # Encode query, key, value
    for i in range(2):
        circuit.ry(i, query[i])
        circuit.ry(i + 2, key[i])
        circuit.ry(i + 4, value[i])
    
    # Compute attention scores
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    
    # Apply attention
    circuit.h(4)
    circuit.cnot(4, 5)
    
    circuit.measure_all()
    return circuit

query = [0.5, 0.3]
key = [0.4, 0.6]
value = [0.7, 0.2]

circuit = quantum_attention(query, key, value)
result = QuantumRun(circuit, shots=100)
print(f"Attention output: {result.counts}")

# ============================================
# Part 3: Quantum transformer
# ============================================

print("\n\nQuantum Transformer Layer:")
print("-" * 50)

def quantum_transformer_layer(num_qubits=4, depth=2):
    """Build quantum transformer layer."""
    circuit = Circuit(num_qubits, name="QTransformer")
    
    for _ in range(depth):
        # Self-attention
        for i in range(num_qubits - 1):
            circuit.cnot(i, i + 1)
            circuit.ry(i + 1, 0.3)
        
        # Feed-forward
        for i in range(num_qubits):
            circuit.ry(i, 0.2)
            circuit.rz(i, 0.1)
    
    return circuit

circuit = quantum_transformer_layer(num_qubits=4, depth=2)
print(f"Transformer layer depth: {circuit.depth()}")
print(f"Number of qubits: {4}")

print("\nApplications:")

## Key Concepts

### Quantum Word Embeddings
- Map words to quantum states
- Entanglement captures semantic relationships
- Superposition enables context-dependent meaning

### Quantum Attention
- Compute attention scores quantumly
- Potential speedup for long sequences
- Natural for parallel processing

### Quantum Transformers
- Replace classical layers with quantum circuits
- Parameter-efficient: fewer parameters
- Potential quantum advantage

### Applications
- **Text classification**: Sentiment, topic
- **Translation**: Language alignment
- **Summarization**: Document compression
- **Question answering**: Information retrieval
