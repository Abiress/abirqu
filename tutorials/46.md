```python
import math
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Ramsey interferometry
# ============================================

print("Quantum Sensing:")
print("=" * 50)

def ramsey_circuit(phase, tau):
    """Build Ramsey interferometry circuit."""
    circuit = Circuit(1, name="Ramsey")
    
    # Prepare superposition
    circuit.h(0)
    
    # Accumulate phase
    circuit.rz(0, phase * tau)
    
    # Interfere
    circuit.h(0)
    
    return circuit

print("Ramsey Interferometry:")
print("-" * 50)

# Measure phase
phases = [0, math.pi/4, math.pi/2, 3*math.pi/4, math.pi]
for phase in phases:
    circuit = ramsey_circuit(phase, tau=1.0)
    circuit.measure_all()
    
    result = QuantumRun(circuit, shots=1000)
    p0 = result.probabilities.get('0', 0.5)
    
    # Phase estimation
    estimated_phase = 2 * math.acos(math.sqrt(p0))
    print(f"  True phase: {phase:.3f} rad | "
          f"Estimated: {estimated_phase:.3f} rad | "
          f"Error: {abs(phase - estimated_phase):.3f}")

# ============================================
# Part 2: Quantum-enhanced sensitivity
# ============================================

print("\n\nQuantum-Enhanced Sensitivity:")
print("-" * 50)

def enhanced_sensitivity_circuit(n_qubits, phase):
    """Build circuit for quantum-enhanced sensing."""
    circuit = Circuit(n_qubits, name="Enhanced")
    
    # Create GHZ state for Heisenberg limit
    circuit.h(0)
    for i in range(n_qubits - 1):
        circuit.cnot(i, i + 1)
    
    # Accumulate phase on all qubits
    for i in range(n_qubits):
        circuit.rz(i, phase)
    
    # Interfere
    for i in range(n_qubits - 1, -1, -1):
        if i < n_qubits - 1:
            circuit.cnot(i, i + 1)
    circuit.h(0)
    
    return circuit

print("Classical vs Quantum Sensitivity:")
print("  Classical: δφ ~ 1/√N (Standard Quantum Limit)")
print("  Quantum:   δφ ~ 1/N (Heisenberg Limit)")
print("  Improvement: √N times better!")

for n in [1, 4, 9, 16]:
    sql = 1 / math.sqrt(n)
    hl = 1 / n
    print(f"  N={n:2d}: SQL={sql:.3f}, HL={hl:.3f}, "
          f"Gain={sql/hl:.1f}x")

# ============================================
# Part 3: Applications
# ============================================

print("\n\nQuantum Sensing Applications:")
print("-" * 50)

applications = [
    ("Magnetic field sensing", "SQUIDs, NV centers"),
    ("Gravity measurement", "Gravitational wave detection"),
    ("Time keeping", "Atomic clocks"),
    ("Temperature sensing", "Quantum thermometry"),
    ("Electric field sensing", "Quantum electrodes"),
]

for app, implementation in applications:
    print(f"  {app}: {implementation}")
```

## Key Concepts

### Quantum Metrology
- **Standard Quantum Limit**: δφ ~ 1/√N (classical)
- **Heisenberg Limit**: δφ ~ 1/N (quantum)
- **Entanglement**: Enables Heisenberg limit

### Sensing Protocols
1. **Ramsey**: Phase estimation via interference
2. **Spin echo**: Refocuses low-frequency noise
3. **DD**: Extends coherence time

### Applications
- **Magnetic resonance**: MRI, NMR
- **Gravitational waves**: LIGO
- **Atomic clocks**: GPS, time standards
- **Navigation**: Quantum gyroscopes
