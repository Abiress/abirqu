```python
import math
import numpy as np
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Content recommendation
# ============================================

print("Quantum Entertainment:")
print("=" * 50)

def quantum_content_recommendation(user_profile, content):
    """Quantum content recommendation."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QRecommend")
    
    # Encode user profile
    circuit.ry(0, user_profile[0] * math.pi)
    circuit.ry(1, user_profile[1] * math.pi)
    
    # Encode content
    circuit.ry(2, content[0] * math.pi)
    circuit.ry(3, content[1] * math.pi)
    
    # Matching
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

print("Quantum Entertainment:")
print("-" * 50)

users = [
    ("Action Fan", [0.8, 0.2]),
    ("Drama Lover", [0.5, 0.5]),
    ("Comedy Fan", [0.3, 0.7]),
    ("Sci-Fi Geek", [0.7, 0.3]),
]

content_types = [
    ("Movie", [0.6, 0.4]),
    ("Series", [0.4, 0.6]),
    ("Documentary", [0.5, 0.5]),
    ("Animation", [0.3, 0.7]),
]

for user_name, user_profile in users:
    for content_name, content in content_types:
        circuit = quantum_content_recommendation(user_profile, content)
        result = QuantumRun(circuit, shots=100)
        print(f"  {user_name} + {content_name}: {result.counts}")

# ============================================
# Part 2: Creative generation
# ============================================

print("\n\nCreative Generation:")
print("-" * 50)

def quantum_creative_generation(style, constraints):
    """Quantum creative generation."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QCreative")
    
    # Encode style
    circuit.ry(0, style[0] * math.pi)
    circuit.ry(1, style[1] * math.pi)
    
    # Encode constraints
    circuit.ry(2, constraints[0] * math.pi)
    circuit.ry(3, constraints[1] * math.pi)
    
    # Generation
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

creative_styles = [
    ("Realistic", [0.8, 0.2], [0.3, 0.7]),
    ("Abstract", [0.2, 0.8], [0.7, 0.3]),
    ("Surreal", [0.4, 0.6], [0.5, 0.5]),
    ("Minimalist", [0.6, 0.4], [0.4, 0.6]),
]

for name, style, constraints in creative_styles:
    circuit = quantum_creative_generation(style, constraints)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 3: Audience analysis
# ============================================

print("\n\nAudience Analysis:")
print("-" * 50)

def quantum_audience_analysis(demographics, engagement):
    """Quantum audience analysis."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QAudience")
    
    # Encode demographics
    circuit.ry(0, demographics[0] * math.pi)
    circuit.ry(1, demographics[1] * math.pi)
    
    # Encode engagement
    circuit.ry(2, engagement[0] * math.pi)
    circuit.ry(3, engagement[1] * math.pi)
    
    # Analysis
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    
    circuit.measure_all()
    return circuit

audiences = [
    ("Youth", [0.2, 0.8], [0.8, 0.2]),
    ("Adult", [0.5, 0.5], [0.5, 0.5]),
    ("Senior", [0.8, 0.2], [0.3, 0.7]),
    ("Family", [0.4, 0.6], [0.6, 0.4]),
]

for name, demographics, engagement in audiences:
    circuit = quantum_audience_analysis(demographics, engagement)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

print("\nApplications:")
print("  - Streaming platforms")
print("  - Gaming")
print("  - Music production")
print("  - Live events")
```
