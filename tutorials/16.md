## Step-by-Step Code

```python
import math
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Simple minimum finding
# ============================================

print("Quantum Minimum Finding:")
print("=" * 50)

def minimum_finding_circuit(num_qubits=3):
    """Build quantum minimum finding circuit.
    
    Uses quantum walk on the function domain.
    """
    total_qubits = 2 * num_qubits + 1  # Two registers + coin
    circuit = Circuit(total_qubits, name="MinFind")
    
    # Initialize first register (candidate)
    for i in range(num_qubits):
        circuit.h(i)
    
    # Initialize coin qubit
    circuit.h(2 * num_qubits)
    
    # Grover-like iterations to find minimum
    for _ in range(int(math.pi / 4 * math.sqrt(2 ** num_qubits))):
        # Oracle marks current minimum
        circuit.z(0)
        
        # Diffusion operator
        for i in range(num_qubits):
            circuit.h(i)
            circuit.x(i)
        circuit.z(0)
        for i in range(num_qubits):
            circuit.x(i)
            circuit.h(i)
    
    # Measure
    circuit.measure_all()
    
    return circuit

# Classical comparison
def classical_minimum(f, domain):
    """Find minimum classically."""
    min_val = float('inf')
    min_idx = 0
    for i, x in enumerate(domain):
        val = f(x)
        if val < min_val:
            min_val = val
            min_idx = i
    return min_idx, min_val

# Test function
def f(x):
    """Simple test function with minimum at x=3."""
    return (x - 3) ** 2

domain = list(range(8))
min_idx_classical, min_val_classical = classical_minimum(f, domain)
print(f"Classical minimum: f({min_idx_classical}) = {min_val_classical}")

# Quantum approach
circuit = minimum_finding_circuit(num_qubits=3)
result = QuantumRun(circuit, shots=1000)

# Most likely answer
most_common = max(result.counts, key=result.counts.get)
min_idx_quantum = int(most_common, 2)
min_val_quantum = f(min_idx_quantum)

print(f"Quantum minimum: f({min_idx_quantum}) = {min_val_quantum}")
print(f"Correct: {min_idx_quantum == min_idx_classical}")

# ============================================
# Part 2: Scaling comparison
# ============================================

print("\n\nScaling Comparison:")
print("=" * 50)

print("Domain size N | Classical queries | Quantum queries")
print("-" * 50)

for n_log in [3, 4, 5, 6, 7, 8, 9, 10]:
    N = 2 ** n_log
    classical = N
    quantum = int(math.pi / 4 * math.sqrt(N) * math.log(N))
    print(f"{N:>13} | {classical:>17} | {quantum:>14}")
```

## Key Concepts

### Algorithm Overview
1. Initialize candidate register in superposition
2. Apply Grover-like iterations to amplify minimum
3. Measure to get index of minimum element

### Speedup
- Classical: O(N) queries needed
- Quantum: O(√N log N) queries needed
- Quadratic speedup for unstructured search

### Applications
- **Optimization**: Finding global minimum
- **Machine learning**: Hyperparameter tuning
- **Cryptography**: Breaking certain schemes
- **Database**: Finding minimum record
