```python
import math
import numpy as np
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Startup evaluation
# ============================================

print("Quantum Venture Capital:")
print("=" * 50)

def quantum_startup_evaluation(team, market):
    """Quantum startup evaluation."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QStartup")
    
    # Encode team
    circuit.ry(0, team[0] * math.pi)
    circuit.ry(1, team[1] * math.pi)
    
    # Encode market
    circuit.ry(2, market[0] * math.pi)
    circuit.ry(3, market[1] * math.pi)
    
    # Evaluation
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

print("Quantum Venture Capital:")
print("-" * 50)

startup_stages = [
    ("Seed", [0.6, 0.4], [0.5, 0.5]),
    ("Series A", [0.7, 0.3], [0.6, 0.4]),
    ("Series B", [0.8, 0.2], [0.7, 0.3]),
    ("Growth", [0.9, 0.1], [0.8, 0.2]),
]

for name, team, market in startup_stages:
    circuit = quantum_startup_evaluation(team, market)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 2: Portfolio optimization
# ============================================

print("\n\nPortfolio Optimization:")
print("-" * 50)

    """Quantum VC portfolio optimization."""

    circuit = Circuit(num_qubits, name="QVCPortfolio")
    
    # Encode startups
    circuit.ry(0, startups[0] * math.pi)
    circuit.ry(1, startups[1] * math.pi)
    
    # Encode strategy
    circuit.ry(2, strategy[0] * math.pi)
    circuit.ry(3, strategy[1] * math.pi)
    
    # Optimization
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

portfolio_scenarios = [
    ("Diversified", [0.5, 0.5], [0.5, 0.5]),
    ("Focused", [0.7, 0.3], [0.7, 0.3]),
    ("Stage-Specific", [0.6, 0.4], [0.6, 0.4]),
    ("Sector-Specific", [0.4, 0.6], [0.4, 0.6]),
]

for name, startups, strategy in portfolio_scenarios:
    circuit = quantum_vc_portfolio(startups, strategy)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 3: Exit planning
# ============================================

print("\n\nExit Planning:")
print("-" * 50)

def quantum_exit_planning(startup, market):
    """Quantum exit planning."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QExit")
    
    # Encode startup
    circuit.ry(0, startup[0] * math.pi)
    circuit.ry(1, startup[1] * math.pi)
    
    # Encode market
    circuit.ry(2, market[0] * math.pi)
    circuit.ry(3, market[1] * math.pi)
    
    # Planning
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

exit_scenarios = [
    ("IPO", [0.8, 0.2], [0.7, 0.3]),
    ("Acquisition", [0.6, 0.4], [0.5, 0.5]),
    ("Secondary", [0.4, 0.6], [0.4, 0.6]),
    ("Write-Off", [0.2, 0.8], [0.2, 0.8]),

for name, startup, market in exit_scenarios:
    circuit = quantum_exit_planning(startup, market)
    result = QuantumRun(circuit, shots=100)

print("\nApplications:")
print("  - Deal sourcing")
print("  - Due diligence")
print("  - Portfolio management")
print("  - Exit strategy")
```
