```python
import math
import numpy as np
from abirqu import Circuit, H, X, Y, Z
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Fault-tolerant computing
# ============================================

print("Advanced Quantum Error Correction:")
print("=" * 50)

def distance_3_code_circuit(logical_state):
    """Encode using distance-3 code."""
    circuit = Circuit(5, name="Dist3Code")
    
    # Encode logical qubit
    if logical_state == 1:
        circuit.x(0)
    
    # Encoding circuit
    circuit.cnot(0, 1)
    circuit.cnot(0, 2)
    circuit.h(3)
    circuit.cnot(3, 4)
    circuit.cnot(0, 3)
    circuit.cnot(1, 4)
    
    circuit.measure_all()
    return circuit

print("Fault-Tolerant Computing:")
print("-" * 50)

for state in [0, 1]:
    circuit = distance_3_code_circuit(state)
    result = QuantumRun(circuit, shots=100)
    print(f"  Logical |{state}⟩: {result.counts}")

# ============================================
# Part 2: Logical operations
# ============================================

print("\n\nLogical Operations:")
print("-" * 50)

def logical_X_circuit(encoded_state):
    """Apply logical X gate."""
    circuit = Circuit(5, name="LogicalX")
    
    # Load encoded state
    for i, bit in enumerate(encoded_state):
        if bit == 1:
            circuit.x(i)
    
    # Logical X = X on all qubits
    for i in range(5):
        circuit.x(i)
    
    circuit.measure_all()
    return circuit

def logical_Z_circuit(encoded_state):
    """Apply logical Z gate."""
    circuit = Circuit(5, name="LogicalZ")
    
    # Load encoded state
    for i, bit in enumerate(encoded_state):
        if bit == 1:
            circuit.x(i)
    
    # Logical Z = Z on data qubits
    for i in range(3):
        circuit.rz(i, math.pi)
    
    circuit.measure_all()
    return circuit

states = [
    ("|0_L⟩", [0, 0, 0, 0, 0]),
    ("|1_L⟩", [1, 1, 1, 0, 0]),
]

for name, state in states:
    circuit = logical_X_circuit(state)
    result = QuantumRun(circuit, shots=100)
    print(f"  X({name}): {result.counts}")
    
    circuit = logical_Z_circuit(state)
    result = QuantumRun(circuit, shots=100)
    print(f"  Z({name}): {result.counts}")

# ============================================
# Part 3: Threshold analysis
# ============================================

print("\n\nThreshold Analysis:")
print("-" * 50)

error_rates = [0.001, 0.01, 0.1]
code_distances = [3, 5, 7]

print(f"  {'Error Rate':<12} {'Distance-3':<12} {'Distance-5':<12} {'Distance-7':<12}")
print(f"  {'-'*12} {'-'*12} {'-'*12} {'-'*12}")

for p in error_rates:
    row = [f"  {p:<12.3f}"]
    for d in code_distances:
        # Approximate logical error rate
        p_L = (p / 0.1) ** ((d + 1) / 2)
        row.append(f"{p_L:<12.6f}")
    print("".join(row))

print("\nApplications:")
print("  - Reliable quantum computing")
print("  - Long computations")
print("  - Quantum memory")
print("  - Distributed quantum computing")
```

## Key Concepts

### Fault-Tolerant Computing
- Encode logical qubits
- Protect against errors
- Perform logical operations

### Logical Operations
- **Logical X**: Bit flip
- **Logical Z**: Phase flip
- **Logical CNOT**: Entangle logical qubits

### Threshold Analysis
- Error threshold: ~0.1%
- Higher distance = lower logical error
- Trade-off: overhead vs. reliability

### Applications
- **Reliable computing**: Fault tolerance
- **Long computations**: Error prevention
- **Quantum memory**: Data protection
- **Distributed QC**: Network reliability
