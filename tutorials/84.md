```python
import math
import numpy as np
from abirqu import Circuit, H, X, CNOT, Ry, Rz, Measure
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Full CI
# ============================================

print("Expert Quantum Chemistry (Part 2):")
print("=" * 50)

def full_ci_circuit(num_electrons, num_orbitals):
    """Full configuration interaction circuit."""
    num_qubits = num_orbitals
    circuit = Circuit(num_qubits, name="FullCI")
    
    # Fill first orbitals
    for i in range(num_electrons):
        circuit.x(i)
    
    # Generate all excitations
    for i in range(num_electrons):
        for j in range(num_electrons, num_orbitals):
            # Single excitation
            circuit.ry(i, 0.1)
            circuit.cnot(i, j)
    
    circuit.measure(list(range(num_qubits)))
    return circuit

print("Full Configuration Interaction:")
print("-" * 50)

circuit = full_ci_circuit(num_electrons=2, num_orbitals=4)
result = QuantumRun(circuit, shots=1000)

print(f"  Electrons: 2")
print(f"  Orbitals: 4")
print(f"  Qubits: {circuit.num_qubits}")
print(f"  Configurations: {len(result.counts)}")

# ============================================
# Part 2: Coupled cluster
# ============================================

print("\n\nCoupled Cluster Theory:")
print("-" * 50)

def coupled_cluster_circuit(num_qubits=4, t_params=None):
    """Coupled cluster singles and doubles."""
    if t_params is None:
        t_params = [0.1] * (num_qubits * (num_qubits - 1) // 2)
    
    circuit = Circuit(num_qubits, name="CCSD")
    
    # Reference state
    circuit.x(0)
    circuit.x(1)
    
    # Singles
    for i in range(2):
        for j in range(2, num_qubits):
            circuit.ry(i, t_params[0])
            circuit.cnot(i, j)
    
    # Doubles
    circuit.ry(0, t_params[1])
    circuit.cnot(0, 1)
    circuit.cnot(2, 3)
    
    return circuit

circuit = coupled_cluster_circuit(num_qubits=4)
print(f"  CCSD circuit depth: {circuit.depth}")

# ============================================
# Part 3: Benchmarks
# ============================================

print("\n\nQuantum Chemistry Benchmarks:")
print("-" * 50)

benchmarks = [

    ("H2O", "Sto-3g", 7, 0.01),
    ("BeH2", "Sto-3g", 7, 0.008),
    ("N2", "Sto-3g", 12, 0.02),
]

print(f"  {'Molecule':10s} {'Basis':8s} {'Qubits':7s} {'Error (Ha)':10s}")
print(f"  {'-'*10} {'-'*8} {'-'*7} {'-'*10}")

for mol, basis, qubits, error in benchmarks:
    print(f"  {mol:10s} {basis:8s} {qubits:7d} {error:10.4f}")

print("\nApplications:")
```

## Key Concepts

### Full CI
- Exact solution within basis set
- Exponentially expensive
- Benchmark for other methods

### Coupled Cluster
- **CCSD**: Singles and doubles
- **CCSD(T)**: With perturbative triples
- Gold standard in quantum chemistry

### Benchmarks
- Compare quantum vs classical methods
- Validate quantum advantage
- Guide algorithm development

- **Drug design**: Molecular properties
- **Materials**: Electronic structure
