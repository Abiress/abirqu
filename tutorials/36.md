import math
import numpy as np
from abirqu import Circuit, H, X, CNOT, Ry, Rz, Measure
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Ising model
# ============================================

print("Many-Body Quantum Simulation:")
print("=" * 50)

def ising_model_circuit(num_qubits=4, J=1.0, h=0.5, time=0.5):
    """Build circuit for Ising model simulation.
    
    H = -J Σ Z_i Z_{i+1} - h Σ X_i
    """
    circuit = Circuit(num_qubits, name="Ising")
    
    # Initialize in superposition (transverse field)
    for i in range(num_qubits):
        circuit.h(i)
    
    # Time evolution (Trotter decomposition)
    # ZZ interactions
    for i in range(num_qubits - 1):
        circuit.cnot(i, i + 1)
        circuit.rz(i + 1, 2 * J * time)
        circuit.cnot(i, i + 1)
    
    # Transverse field
    for i in range(num_qubits):
        circuit.rx(i, 2 * h * time)
    
    return circuit

print("Ising Model Simulation:")
print("-" * 50)

# Simulate at different times
times = [0.0, 0.25, 0.5, 0.75, 1.0]
for t in times:
    circuit = ising_model_circuit(num_qubits=4, time=t)
    circuit.measure([0, 1, 2, 3])
    
    result = QuantumRun(circuit, shots=1000)
    
    # Compute magnetization
    mag = 0
    for bitstring, count in result.counts.items():
        for bit in bitstring:
            if bit == '0':
                mag += count
            else:
                mag -= count
    mag /= (4 * 1000)  # Normalize
    
    print(f"  t={t:.2f}: Magnetization = {mag:.3f}")

# ============================================
# Part 2: Heisenberg model
# ============================================

print("\n\nHeisenberg Model:")
print("-" * 50)

def heisenberg_circuit(num_qubits=4, J=1.0, time=0.5):
    """Build circuit for Heisenberg model.
    
    H = J Σ (X_i X_{i+1} + Y_i Y_{i+1} + Z_i Z_{i+1})
    """
    circuit = Circuit(num_qubits, name="Heisenberg")
    
    # Initialize
    for i in range(num_qubits):
        circuit.h(i)
    
    # Trotter step
    # XX interactions
    for i in range(num_qubits - 1):
        circuit.h(i)
        circuit.h(i + 1)
        circuit.cnot(i, i + 1)
        circuit.rz(i + 1, 2 * J * time)
        circuit.cnot(i, i + 1)
        circuit.h(i)
        circuit.h(i + 1)
    
    # YY interactions
    for i in range(num_qubits - 1):
        circuit.s(i)
        circuit.s(i + 1)
        circuit.h(i)
        circuit.h(i + 1)
        circuit.cnot(i, i + 1)
        circuit.rz(i + 1, 2 * J * time)
        circuit.cnot(i, i + 1)
        circuit.h(i)
        circuit.h(i + 1)
        circuit.sdg(i)
        circuit.sdg(i + 1)
    
    # ZZ interactions
    for i in range(num_qubits - 1):
        circuit.cnot(i, i + 1)
        circuit.rz(i + 1, 2 * J * time)
        circuit.cnot(i, i + 1)
    
    return circuit

circuit = heisenberg_circuit(num_qubits=4, time=0.5)
result = QuantumRun(circuit, shots=1000)
print(f"Heisenberg model measurement: {len(result.counts)} unique states")

# ============================================
# Part 3: Correlation functions
# ============================================

print("\n\nCorrelation Functions:")
print("-" * 50)

def compute_correlation(circuit, qubit_i, qubit_j, shots=1000):
    """Compute ZZ correlation between two qubits."""
    measure_circuit = circuit.copy()
    measure_circuit.measure([qubit_i, qubit_j])
    
    result = QuantumRun(measure_circuit, shots=shots)
    
    # Compute <Z_i Z_j>
    corr = 0
    for bitstring, count in result.counts.items():
        zi = 1 if bitstring[0] == '0' else -1
        zj = 1 if bitstring[1] == '0' else -1
        corr += zi * zj * count / shots
    
    return corr

# Compute correlations for Ising model
circuit = ising_model_circuit(num_qubits=4, time=0.5)
print("Ising model correlations:")
for i in range(3):
    corr = compute_correlation(circuit, i, i + 1)
    print(f"  <Z_{i} Z_{i+1}> = {corr:.3f}")

## Key Concepts

### Many-Body Hamiltonians
1. **Ising model**: Z-Z interactions + transverse field
2. **Heisenberg model**: X-X + Y-Y + Z-Z interactions
3. **Hubbard model**: Electron hopping + interactions

### Trotter Decomposition
- e^(-i(A+B)t) ≈ e^(-iAt)e^(-iBt) + O(t²)
- Breaks complex evolution into simple gates
- Error decreases with smaller time steps

### Applications
- **Condensed matter**: Phase transitions
- **High-energy physics**: Lattice gauge theories
- **Quantum chemistry**: Electron dynamics
- **Quantum materials**: Topological phases
