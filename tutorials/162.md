import math
import numpy as np
from abirqu import Circuit, H, X, CNOT, Ry, Rz, Measure
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Credit scoring
# ============================================

print("Quantum Banking:")
print("=" * 50)

def quantum_credit_scoring(history, income):
    """Quantum credit scoring."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QCredit")
    
    # Encode history
    circuit.ry(0, history[0] * math.pi)
    circuit.ry(1, history[1] * math.pi)
    
    # Encode income
    circuit.ry(2, income[0] * math.pi)
    circuit.ry(3, income[1] * math.pi)
    
    # Scoring
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure([2])
    return circuit

print("Quantum Banking:")
print("-" * 50)

credit_profiles = [
    ("Excellent", [0.9, 0.1], [0.8, 0.2]),
    ("Good", [0.7, 0.3], [0.6, 0.4]),
    ("Fair", [0.5, 0.5], [0.4, 0.6]),
    ("Poor", [0.3, 0.7], [0.2, 0.8]),
]

for name, history, income in credit_profiles:
    circuit = quantum_credit_scoring(history, income)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================

print("\n\nFraud Detection:")

    """Quantum fraud detection."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QFraud")
    
    # Encode transaction
    circuit.ry(0, transaction[0] * math.pi)
    circuit.ry(1, transaction[1] * math.pi)
    
    # Encode behavior
    circuit.ry(2, behavior[0] * math.pi)
    circuit.ry(3, behavior[1] * math.pi)
    
    # Detection
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure([2])
    return circuit

transactions = [
    ("Normal", [0.3, 0.7], [0.4, 0.6]),
    ("Suspicious", [0.7, 0.3], [0.8, 0.2]),
    ("Fraudulent", [0.9, 0.1], [0.9, 0.1]),

for name, transaction, behavior in transactions:

# ============================================
# Part 3: Risk management
# ============================================

print("\n\nRisk Management:")

def quantum_risk_management(portfolio, market):
    """Quantum risk management."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QRiskMgmt")
    
    # Encode portfolio
    circuit.ry(0, portfolio[0] * math.pi)
    circuit.ry(1, portfolio[1] * math.pi)
    
    # Encode market
    circuit.ry(2, market[0] * math.pi)

    # Risk assessment
    circuit.cnot(0, 2)

    circuit.rz(2, 0.5)

    return circuit

risk_scenarios = [

for name, portfolio, market in risk_scenarios:
