class QuantumNetwork:
    """Advanced quantum network simulator."""
    
    def __init__(self, num_nodes):
        self.num_nodes = num_nodes
        self.links = {}
        self.entanglement = {}
    
    def add_link(self, node1, node2, fidelity=0.9):
        """Add link with fidelity."""
        self.links[(node1, node2)] = fidelity
        self.entanglement[(node1, node2)] = 0
    
    def route(self, source, destination, metric='fidelity'):
        """Route entanglement."""
        # Simple Dijkstra-like routing
        visited = {source}
        queue = [(source, [source], 1.0)]
        
        while queue:
            node, path, fidelity = queue.pop(0)
            
            if node == destination:
                return path, fidelity
            
            for (n1, n2), f in self.links.items():
                next_node = None
                if n1 == node and n2 not in visited:
                    next_node = n2
                elif n2 == node and n1 not in visited:
                    next_node = n1
                
                if next_node:
                    visited.add(next_node)
                    new_fidelity = fidelity * f
                    queue.append((next_node, path + [next_node], new_fidelity))
        
        return None, 0

# Create network

class EntanglementManager:
    """Manage entanglement across network."""
    
    def __init__(self):
        self.pools = {}
        self.stats = {'created': 0, 'consumed': 0, 'decayed': 0}
    
    def create(self, node1, node2):
        """Create entanglement."""
        key = (node1, node2)
        if key not in self.pools:
            self.pools[key] = []
        self.pools[key].append({'fidelity': 0.9, 'age': 0})
        self.stats['created'] += 1
    
    def consume(self, node1, node2):
        """Consume entanglement."""
        key = (node1, node2)
        if key in self.pools and self.pools[key]:
            self.pools[key].pop()
            self.stats['consumed'] += 1
            return True
        return False
    
    def get_stats(self):
        """Get statistics."""
        return self.stats
