import math
import numpy as np
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Quantum semidefinite programming
# ============================================

print("Expert Quantum Optimization:")
print("=" * 50)

def quantum_sdp_circuit(num_qubits=3):
    """Build circuit for quantum SDP."""
    circuit = Circuit(num_qubits, name="QSDP")
    
    # Initialize in maximally entangled state
    circuit.h(0)
    for i in range(num_qubits - 1):
        circuit.cnot(i, i + 1)
    
    # SDP constraints (simplified)
    for i in range(num_qubits):
        circuit.ry(i, math.pi / 4)
    
    circuit.measure_all()
    return circuit

print("Quantum Semidefinite Programming:")
print("-" * 50)

circuit = quantum_sdp_circuit()
result = QuantumRun(circuit, shots=1000)
print(f"SDP solution sample: {result.counts}")

# ============================================
# Part 2: Combinatorial optimization
# ============================================

print("\n\nCombinatorial Optimization:")
print("-" * 50)

def max_cut_qaoa(edges, p=3):
    """QAOA for MaxCut with higher p."""
    num_qubits = max(max(e) for e in edges) + 1
    circuit = Circuit(num_qubits, name="MaxCut")
    
    # Initial superposition
    for i in range(num_qubits):
        circuit.h(i)
    
    # QAOA layers
    for layer in range(p):
        gamma = layer * math.pi / (2 * p)
        beta = (p - layer) * math.pi / (4 * p)
        
        # Problem unitary
        for i, j in edges:
            circuit.cnot(i, j)
            circuit.rz(j, gamma)
            circuit.cnot(i, j)
        
        # Mixer unitary
        for i in range(num_qubits):
            circuit.rx(i, beta)
    
    circuit.measure_all()
    return circuit

# Triangle graph
edges = [(0, 1), (1, 2), (0, 2)]
circuit = max_cut_qaoa(edges, p=3)
result = QuantumRun(circuit, shots=1000)

print(f"MaxCut solutions: {len(result.counts)} unique")

# ============================================
# Part 3: Portfolio optimization
# ============================================

print("\n\nPortfolio Optimization:")
print("-" * 50)

def portfolio_qaoa(returns, risks, correlation, budget=1.0):
    """QAOA for portfolio optimization."""
    num_assets = len(returns)
    circuit = Circuit(num_assets, name="Portfolio")
    
    # Initial state
    for i in range(num_assets):
        circuit.h(i)
    
    # Risk term
    for i in range(num_assets):
        for j in range(i + 1, num_assets):
            if correlation[i][j] > 0.5:
                circuit.cnot(i, j)
                circuit.rz(j, risks[i] * risks[j])
                circuit.cnot(i, j)
    
    # Return term
    for i in range(num_assets):
        circuit.ry(i, returns[i])
    
    circuit.measure_all()
    return circuit

# Example portfolio
returns = [0.1, 0.15, 0.12]
risks = [0.2, 0.3, 0.25]
correlation = [[1.0, 0.5, 0.3], [0.5, 1.0, 0.4], [0.3, 0.4, 1.0]]

circuit = portfolio_qaoa(returns, risks, correlation)
result = QuantumRun(circuit, shots=1000)

print(f"Portfolio solutions: {result.counts}")
