```python
import math
import numpy as np
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Nanoparticle design
# ============================================

print("Quantum Nanotechnology:")
print("=" * 50)

def quantum_nanoparticle(size, composition, shape):
    """Quantum nanoparticle design."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QNanoparticle")
    
    # Encode size
    circuit.ry(0, size * math.pi)
    
    # Encode composition
    circuit.ry(1, composition[0] * math.pi)
    circuit.ry(2, composition[1] * math.pi)
    
    # Encode shape
    circuit.ry(3, shape * math.pi)
    
    # Property calculation
    circuit.cnot(0, 1)
    circuit.cnot(1, 2)
    circuit.cnot(2, 3)
    
    circuit.measure_all()
    return circuit

print("Quantum Nanotechnology:")
print("-" * 50)

nanoparticles = [
    ("Gold", 0.5, [0.8, 0.2], 0.3),
    ("Silver", 0.4, [0.7, 0.3], 0.4),
    ("Quantum Dot", 0.2, [0.5, 0.5], 0.5),
    ("Graphene", 0.1, [0.9, 0.1], 0.2),
]

for name, size, composition, shape in nanoparticles:
    circuit = quantum_nanoparticle(size, composition, shape)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 2: Self-assembly
# ============================================

print("\n\nSelf-Assembly Optimization:")
print("-" * 50)

def quantum_self_assembly(components, conditions):
    """Quantum self-assembly optimization."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QAssembly")
    
    # Encode components
    circuit.ry(0, components[0] * math.pi)
    circuit.ry(1, components[1] * math.pi)
    
    # Encode conditions
    circuit.ry(2, conditions[0] * math.pi)
    circuit.ry(3, conditions[1] * math.pi)
    
    # Assembly optimization
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

assemblies = [
    ("Liposome", [0.6, 0.4], [0.7, 0.3]),
    ("Micelle", [0.5, 0.5], [0.5, 0.5]),
    ("Vesicle", [0.4, 0.6], [0.3, 0.7]),
    ("Nanotube", [0.3, 0.7], [0.4, 0.6]),
]

for name, components, conditions in assemblies:
    circuit = quantum_self_assembly(components, conditions)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 3: Nanoscale characterization
# ============================================

print("\n\nNanoscale Characterization:")
print("-" * 50)

def quantum_characterization(probe, sample):
    """Quantum nanoscale characterization."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QNChar")
    
    # Encode probe
    circuit.ry(0, probe[0] * math.pi)
    circuit.ry(1, probe[1] * math.pi)
    
    # Encode sample
    circuit.ry(2, sample[0] * math.pi)
    circuit.ry(3, sample[1] * math.pi)
    
    # Interaction
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    
    # Measurement
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

probes = [
    ("AFM", [0.8, 0.2]),
    ("STM", [0.7, 0.3]),
    ("TEM", [0.6, 0.4]),
    ("SEM", [0.5, 0.5]),
]

samples = [
    ("Surface", [0.4, 0.6]),
    ("Bulk", [0.6, 0.4]),
    ("Thin Film", [0.5, 0.5]),
]

for probe_name, probe in probes:
    for sample_name, sample in samples:
        circuit = quantum_characterization(probe, sample)
        result = QuantumRun(circuit, shots=100)
        print(f"  {probe_name} + {sample_name}: {result.counts}")

print("\nApplications:")
print("  - Nanoparticle synthesis")
print("  - Self-assembly control")
print("  - Nanoscale imaging")
print("  - Material characterization")
```
