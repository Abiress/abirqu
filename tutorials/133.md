```python
import math
import numpy as np
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Material property prediction
# ============================================

print("Quantum Materials Science:")
print("=" * 50)

def quantum_material_properties(composition, structure):
    """Quantum material property prediction."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QMaterial")
    
    # Encode composition
    circuit.ry(0, composition[0] * math.pi)
    circuit.ry(1, composition[1] * math.pi)
    
    # Encode structure
    circuit.ry(2, structure[0] * math.pi)
    circuit.ry(3, structure[1] * math.pi)
    
    # Property calculation
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

print("Quantum Materials Science:")
print("-" * 50)

materials = [
    ("Steel", [0.7, 0.3], [0.8, 0.2]),
    ("Aluminum", [0.3, 0.7], [0.6, 0.4]),
    ("Titanium", [0.5, 0.5], [0.7, 0.3]),
    ("Copper", [0.4, 0.6], [0.5, 0.5]),
]

for name, composition, structure in materials:
    circuit = quantum_material_properties(composition, structure)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 2: Crystal structure
# ============================================

print("\n\nCrystal Structure Optimization:")
print("-" * 50)

def quantum_crystal_structure(lattice_params, basis):
    """Quantum crystal structure optimization."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QCrystal")
    
    # Encode lattice
    for i in range(2):
        circuit.ry(i, lattice_params[i] * math.pi)
    
    # Encode basis
    for i in range(2):
        circuit.ry(i + 2, basis[i] * math.pi)
    
    # Energy minimization
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

crystals = [
    ("FCC", [1.0, 1.0], [0.0, 0.5]),
    ("BCC", [0.9, 0.9], [0.5, 0.0]),
    ("HCP", [1.0, 0.8], [0.0, 0.3]),
    ("Diamond", [1.2, 1.2], [0.0, 0.25]),
]

for name, lattice, basis in crystals:
    circuit = quantum_crystal_structure(lattice, basis)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 3: Alloy design
# ============================================

print("\n\nAlloy Design:")
print("-" * 50)

def quantum_alloy_design(elements, ratios, properties):
    """Quantum alloy design."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QAlloy")
    
    # Encode elements
    circuit.ry(0, elements[0] * math.pi)
    circuit.ry(1, elements[1] * math.pi)
    
    # Encode ratios
    circuit.ry(2, ratios * math.pi)
    
    # Encode target properties
    circuit.ry(3, properties * math.pi)
    
    # Optimization
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

alloys = [
    ("Steel", [0.7, 0.3], 0.7, 0.8),
    ("Brass", [0.6, 0.4], 0.6, 0.7),
    ("Bronze", [0.5, 0.5], 0.5, 0.6),
    ("Inconel", [0.8, 0.2], 0.8, 0.9),
]

for name, elements, ratios, properties in alloys:
    circuit = quantum_alloy_design(elements, ratios, properties)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

print("\nApplications:")
print("  - Material discovery")
print("  - Property optimization")
print("  - Alloy development")
print("  - Structural materials")
```
