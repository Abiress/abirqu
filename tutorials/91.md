```python
import math
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Quantum metrology
# ============================================

print("Expert Quantum Sensing:")
print("=" * 50)

def quantum_metrology_circuit(num_qubits=4, phase=0.5):
    """Heisenberg-limited metrology."""
    circuit = Circuit(num_qubits, name="Metrology")
    
    # Create GHZ state
    circuit.h(0)
    for i in range(num_qubits - 1):
        circuit.cnot(i, i + 1)
    
    # Phase accumulation
    for i in range(num_qubits):
        circuit.rz(i, phase)
    
    # Interference
    for i in range(num_qubits - 1, -1, -1):
        if i < num_qubits - 1:
            circuit.cnot(i, i + 1)
    circuit.h(0)
    
    circuit.measure_all()
    return circuit

print("Quantum Metrology:")
print("-" * 50)

phase = 0.5
for n in [1, 4, 9, 16]:
    circuit = quantum_metrology_circuit(num_qubits=n, phase=phase)
    result = QuantumRun(circuit, shots=1000)
    
    p0 = result.probabilities.get('0', 0.5)
    estimated = 2 * math.acos(math.sqrt(p0))
    error = abs(phase - estimated)
    
    print(f"  N={n:2d}: Error = {error:.4f} (SQL: {1/math.sqrt(n):.4f})")

# ============================================
# Part 2: Entanglement-enhanced sensing
# ============================================

print("\n\nEntanglement Enhancement:")
print("-" * 50)

def entanglement_sensing_circuit(num_sensors=4):
    """Entanglement-enhanced sensing."""
    circuit = Circuit(num_sensors + 1, name="EntSense")
    
    # Create GHZ state
    circuit.h(0)
    for i in range(num_sensors):
        circuit.cnot(i, i + 1)
    
    # Each sensor measures
    for i in range(num_sensors + 1):
        circuit.rz(i, 0.1 * (i + 1))
    
    # Reverse entanglement
    for i in range(num_sensors - 1, -1, -1):
        circuit.cnot(i, i + 1)
    circuit.h(0)
    
    circuit.measure_all()
    return circuit

circuit = entanglement_sensing_circuit(num_sensors=4)
result = QuantumRun(circuit, shots=1000)

print(f"  Sensors: 4")
print(f"  Measurement outcomes: {result.counts}")

# ============================================
# Part 3: Sensor networks
# ============================================

print("\n\nQuantum Sensor Networks:")
print("-" * 50)

def sensor_network_circuit(num_sensors=6):
    """Distributed quantum sensing."""
    circuit = Circuit(num_sensors, name="SensorNet")
    
    # Create cluster state
    circuit.h(0)
    for i in range(num_sensors - 1):
        circuit.cnot(i, i + 1)
    
    # Each sensor measures local field
    for i in range(num_sensors):
        circuit.rz(i, 0.1 * (i + 1))
    
    circuit.measure_all()
    return circuit

circuit = sensor_network_circuit(num_sensors=6)
result = QuantumRun(circuit, shots=1000)

print(f"  Network size: 6 sensors")
print(f"  Unique outcomes: {len(result.counts)}")

print("\nApplications:")
print("  - Gravitational wave detection (LIGO)")
print("  - Magnetic resonance imaging (MRI)")
print("  - Navigation (quantum gyroscopes)")
print("  - Geophysical exploration")
print("  - Medical diagnostics")
```

## Key Concepts

### Quantum Metrology
- **SQL**: δφ ~ 1/√N (classical)
- **Heisenberg**: δφ ~ 1/N (quantum)
- **Entanglement**: Enables Heisenberg limit

### Entanglement Enhancement
- GHZ states for phase sensing
- Spin-squeezed states
- Cluster states

### Sensor Networks
- Distributed quantum sensing
- Correlated measurements
- Network advantage

### Applications

- **Medical imaging**: MRI, MEG
- **Navigation**: Quantum gyroscopes
