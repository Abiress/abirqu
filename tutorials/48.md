```python
import math
from abirqu import Circuit, H, CNOT, Measure
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Network topology
# ============================================

print("Quantum Networks:")
print("=" * 50)

class QuantumNetwork:
    """Simple quantum network simulator."""
    
    def __init__(self, num_nodes):
        self.num_nodes = num_nodes
        self.links = []
        self.entanglement = {}
    
    def add_link(self, node1, node2):
        """Add quantum link between nodes."""
        self.links.append((node1, node2))
        self.entanglement[(node1, node2)] = 0
    
    def create_entanglement(self, node1, node2):
        """Create entanglement between nodes."""
        if (node1, node2) in self.links:
            self.entanglement[(node1, node2)] += 1
            return True
        return False
    
    def get_entanglement(self, node1, node2):
        """Get entanglement between nodes."""
        return self.entanglement.get((node1, node2), 0)

# Create network
network = QuantumNetwork(4)
network.add_link(0, 1)
network.add_link(1, 2)
network.add_link(2, 3)
network.add_link(0, 3)

print("Network topology:")
print(f"  Nodes: {network.num_nodes}")
print(f"  Links: {network.links}")

# Create entanglement
for i in range(3):
    network.create_entanglement(0, 1)
    network.create_entanglement(1, 2)
    network.create_entanglement(2, 3)

print(f"\nEntanglement distribution:")
for link, count in network.entanglement.items():
    print(f"  Link {link}: {count} pairs")

# ============================================
# Part 2: Entanglement routing
# ============================================

print("\n\nEntanglement Routing:")
print("-" * 50)

def route_entanglement(network, source, destination):
    """Route entanglement through network."""
    # Simple BFS routing
    visited = {source}
    queue = [(source, [source])]
    
    while queue:
        node, path = queue.pop(0)
        
        if node == destination:
            return path
        
        for link in network.links:
            next_node = None
            if link[0] == node and link[1] not in visited:
                next_node = link[1]
            elif link[1] == node and link[0] not in visited:
                next_node = link[0]
            
            if next_node is not None:
                visited.add(next_node)
                queue.append((next_node, path + [next_node]))
    
    return None

# Find path
path = route_entanglement(network, 0, 3)
print(f"  Route from 0 to 3: {path}")

# ============================================
# Part 3: Quantum network protocols
# ============================================

print("\n\nNetwork Protocols:")
print("-" * 50)

def quantum_key_distribution(network, node1, node2):
    """Simulate QKD over network."""
    # Check if path exists
    path = route_entanglement(network, node1, node2)
    
    if path:
        key_length = len(path) * 8  # bits per hop
        return key_length
    return 0

key_len = quantum_key_distribution(network, 0, 3)
print(f"  QKD key length (0→3): {key_len} bits")

print("\nQuantum network applications:")
print("  - Secure communication")
print("  - Distributed quantum computing")
print("  - Quantum cloud services")
```

## Key Concepts

### Network Components
1. **Nodes**: Quantum processors
2. **Links**: Quantum channels (fiber, free-space)
3. **Entanglement**: Shared quantum correlations
4. **Repeaters**: Extend entanglement distance

### Protocols
- **QKD**: Quantum key distribution
- **Teleportation**: State transfer
- **Entanglement swapping**: Extend connectivity
- **Superdense coding**: Enhanced classical capacity

### Applications
- **Quantum internet**: Global quantum network
- **Distributed computing**: Multi-node quantum algorithms
- **Quantum cloud**: Access remote quantum computers
- **Secure communication**: Government, finance
