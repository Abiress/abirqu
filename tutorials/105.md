```python
import math
from abirqu import Circuit, H, X, CNOT, Ry, Rz, Measure
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Multi-qubit decomposition
# ============================================

print("Expert Quantum Compilation (Part 2):")
print("=" * 50)

def decompose_fredkin(circuit, qubits):
    """Decompose Fredkin (CSWAP) gate."""
    a, b, c = qubits
    
    circuit.cnot(c, b)
    circuit.h(a)
    circuit.cnot(a, c)
    circuit.tdg(c)
    circuit.cnot(a, b)
    circuit.t(b)
    circuit.cnot(a, c)
    circuit.t(c)
    circuit.cnot(a, b)
    circuit.tdg(b)
    circuit.cnot(a, c)
    circuit.h(a)
    circuit.cnot(c, b)
    
    return circuit

print("Fredkin Gate Decomposition:")
print("-" * 50)

circuit = Circuit(3, name="Fredkin")
decompose_fredkin(circuit, [0, 1, 2])
print(f"  Native gates: {circuit.depth}")
print("  Fredkin = 7 basic gates")

# ============================================
# Part 2: Circuit synthesis
# ============================================

print("\n\nCircuit Synthesis:")
print("-" * 50)

def synthesize_arbitrary_rotation(angle, axis):
    """Synthesize arbitrary rotation."""
    circuit = Circuit(1, name="Synth")
    
    if axis == 'X':
        circuit.ry(angle / 2, 0)
        circuit.rz(math.pi / 2, 0)
        circuit.ry(angle / 2, 0)
        circuit.rz(-math.pi / 2, 0)
    elif axis == 'Y':
        circuit.rz(angle / 2, 0)
        circuit.ry(angle / 2, 0)
        circuit.rz(-angle / 2, 0)
    elif axis == 'Z':
        circuit.rz(angle, 0)
    
    return circuit

print("Arbitrary Rotation Synthesis:")
for axis in ['X', 'Y', 'Z']:
    circuit = synthesize_arbitrary_rotation(math.pi / 4, axis)
    print(f"  {axis}-rotation: {circuit.depth} gates")

# ============================================
# Part 3: Compilation strategies
# ============================================

print("\n\nCompilation Strategies:")
print("-" * 50)

strategies = [
    ("Greedy", "Locally optimal choices", "Fast"),
    ("Dynamic Programming", "Optimal substructure", "Optimal"),
    ("Simulated Annealing", "Random exploration", "Good quality"),
    ("Genetic Algorithm", "Evolutionary optimization", "Global optimum"),
    ("Machine Learning", "Learned compilation", "Adaptive"),
]

for strategy, description, advantage in strategies:
    print(f"  {strategy:20s}: {description:30s} ({advantage})")

# Final demonstration
print("\n\nFinal Demonstration:")
print("-" * 50)

circuit = Circuit(3, name="Final")
circuit.h(0)
circuit.cnot(0, 1)
circuit.cnot(1, 2)
circuit.measure([0, 1, 2])

result = QuantumRun(circuit, shots=1000)
print(f"GHZ state: {result.counts}")
```
