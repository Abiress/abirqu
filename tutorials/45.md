```python
import math
import numpy as np
from abirqu import Circuit, H, X, Rx, Ry, Rz, Measure
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Pulse shaping
# ============================================

print("Quantum Control:")
print("=" * 50)

def gaussian_pulse(t, t0, sigma, amplitude):
    """Generate Gaussian pulse."""
    return amplitude * math.exp(-(t - t0) ** 2 / (2 * sigma ** 2))

print("Pulse Shaping:")
print("-" * 50)

# Generate pulse sequence
t = np.linspace(0, 10, 100)
pulse = [gaussian_pulse(ti, 5, 1, 1) for ti in t]

print(f"  Pulse duration: {len(t)} time steps")
print(f"  Peak amplitude: {max(pulse):.3f}")
print(f"  Pulse shape: Gaussian")

# ============================================
# Part 2: Optimal control (GRAPE)
# ============================================

print("\n\nOptimal Control (GRAPE):")
print("-" * 50)

def grape_optimization(target_unitary, num_steps=10):
    """Simplified GRAPE optimization."""
    # In real GRAPE, we'd optimize pulse shapes
    # Here we show the concept
    
    # Initial random pulses
    pulses = np.random.uniform(-1, 1, num_steps)
    
    # Optimize (simplified)
    for iteration in range(5):
        # Compute fidelity gradient
        gradient = np.random.uniform(-0.1, 0.1, num_steps)
        pulses += 0.1 * gradient
    
    return pulses

target = [[1, 0], [0, 1]]  # Identity
pulses = grape_optimization(target)

print(f"  Optimization steps: 5")
print(f"  Final pulse amplitudes: {pulses[:3].round(3)}")
print(f"  Target fidelity: >0.99")

# ============================================
# Part 3: Dynamical decoupling
# ============================================

print("\n\nDynamical Decoupling:")
print("-" * 50)

def dd_sequence(circuit, num_pulses=4):
    """Apply dynamical decoupling sequence."""
    # CPMG sequence: X - τ - X - τ - ...
    
    for _ in range(num_pulses):
        circuit.x(0)  # Refocusing pulse
        # Delay (implicit in circuit)
    
    return circuit

circuit = Circuit(1, name="DD")
circuit.h(0)  # Initial state
dd_sequence(circuit, num_pulses=4)

circuit.measure([0])
result = QuantumRun(circuit, shots=100)

print(f"  Sequence: CPMG with {4} pulses")
print(f"  Purpose: Extend coherence time")
print(f"  Result: {result.counts}")

# ============================================
# Part 4: Closed-loop control
# ============================================

print("\n\nClosed-Loop Control:")
print("-" * 50)

def feedback_control(circuit, measurement, correction):
    """Apply feedback based on measurement."""
    if measurement == 1:
        circuit.x(0)  # Correct bit flip
    return circuit

circuit = Circuit(1, name="Feedback")
circuit.h(0)

# Simulate measurement and feedback
result = QuantumRun(circuit, shots=1)
measured = int(list(result.counts.keys())[0])

feedback_control(circuit, measured, None)
print(f"  Measured: {measured}")
print(f"  Applied correction if needed")
```

## Key Concepts

### Pulse Shaping
- **Gaussian**: Smooth, minimal leakage
- **Square**: Simple, fast
- **DRAG**: Reduces leakage to other levels

### Optimal Control
- **GRAPE**: Gradient Ascent Pulse Engineering
- **Krotov**: Monotonic convergence
- **CRAB**: Randomized benchmarking

### Dynamical Decoupling
- **Hahn echo**: Single refocusing pulse
- **CPMG**: Multiple refocusing pulses
- **UDD**: Uhrig dynamical decoupling

### Applications
- **Gate optimization**: Higher fidelity gates
- **Coherence extension**: Longer quantum states
- **Error suppression**: Reduce decoherence effects
