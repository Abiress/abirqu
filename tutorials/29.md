## Step-by-Step Code

```python
import math
import numpy as np
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Time series encoding
# ============================================

print("Quantum Time Series Analysis:")
print("=" * 50)

def encode_time_series(series, num_qubits=4):
    """Encode time series as quantum state."""
    circuit = Circuit(num_qubits, name="TSEnc")
    
    # Encode each time step
    for i in range(min(num_qubits, len(series))):
        angle = series[i] * math.pi
        circuit.ry(i, angle)
    
    # Add temporal correlations
    for i in range(num_qubits - 1):
        circuit.cnot(i, i + 1)
        circuit.rz(i + 1, series[i] * 0.1 if i < len(series) else 0)
    
    return circuit

# Sample time series (sine wave with noise)
t = np.linspace(0, 2 * np.pi, 16)
series = np.sin(t) + 0.1 * np.random.randn(16)

print("Time series (first 8 points):")
print(series[:8].round(3))

# Encode
circuit = encode_time_series(series[:4])
result = QuantumRun(circuit, shots=100)

print(f"\nEncoded state: {result.counts}")

# ============================================
# Part 2: Quantum pattern matching
# ============================================

print("\n\nQuantum Pattern Matching:")
print("-" * 50)

def quantum_pattern_match(series, pattern):
    """Match pattern in time series using quantum circuit."""
    circuit = Circuit(4, name="PatternMatch")
    
    # Encode series
    for i in range(4):
        circuit.ry(i, series[i] * math.pi)
    
    # Encode pattern
    circuit.ry(0, pattern[0] * math.pi)
    
    # Compare
    circuit.cnot(0, 1)
    circuit.x(1)
    
    circuit.measure_all()
    return circuit

# Define pattern (sine wave peak)
pattern = [0.8, 0.9, 1.0, 0.9]

# Search for pattern
print("Pattern search results:")
for i in range(0, 12, 4):
    segment = series[i:i + 4]
    circuit = quantum_pattern_match(segment, pattern)
    result = QuantumRun(circuit, shots=100)
    
    # Pattern match score
    match_score = result.probabilities.get('00', 0)
    print(f"  Segment {i}-{i+3}: Match score = {match_score:.3f}")

# ============================================
# Part 3: Quantum forecasting
# ============================================

print("\n\nQuantum Forecasting:")
print("=" * 50)

def quantum_forecaster(history, params):
    """Forecast next value using quantum circuit."""
    circuit = Circuit(4, name="Forecast")
    
    # Encode history
    for i in range(4):
        circuit.ry(i, history[i] * math.pi)
    
    # Forecast layer
    for i in range(3):
        circuit.cnot(i, i + 1)
    for i in range(4):
        circuit.ry(i, params[i])
    
    # Measure to get forecast
    circuit.measure_all()
    return circuit

# Simple forecasting
params = [0.0] * 4
history = series[:4]

print("Forecasting:")
for step in range(4):
    circuit = quantum_forecaster(history, params)
    result = QuantumRun(circuit, shots=100)
    
    # Get forecast value
    forecast = result.probabilities.get('1', 0.5)
    actual = series[4 + step] if 4 + step < len(series) else 0
    
    print(f"  Step {step+1}: Forecast={forecast:.3f}, Actual={actual:.3f}")
    
    # Update history
    history = np.roll(history, -1)
    history[-1] = actual
```

## Key Concepts

### Quantum Time Series Pipeline
1. **Encode** time steps as quantum states
2. **Find patterns** using quantum similarity
3. **Forecast** using parameterized circuits
4. **Update** with new observations

### Quantum Advantages
- Exponential memory for temporal patterns
- Parallel processing of all time steps
- Natural for quantum sensor data

### Applications
- **Financial forecasting**: Stock prices
- **Weather prediction**: Climate modeling
- **IoT analytics**: Sensor data
- **Healthcare**: Patient monitoring
