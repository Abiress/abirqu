import math
import numpy as np
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Revenue management
# ============================================

print("Quantum Hospitality:")
print("=" * 50)

def quantum_revenue_management(demand, pricing):
    """Quantum hotel revenue management."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QRevenue")
    
    # Encode demand
    circuit.ry(0, demand[0] * math.pi)
    circuit.ry(1, demand[1] * math.pi)
    
    # Encode pricing
    circuit.ry(2, pricing[0] * math.pi)
    circuit.ry(3, pricing[1] * math.pi)
    
    # Revenue optimization
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

print("Quantum Hospitality:")
print("-" * 50)

seasons = [
    ("Peak", [0.9, 0.1], [0.8, 0.2]),
    ("Off-Peak", [0.3, 0.7], [0.4, 0.6]),
    ("Holiday", [0.8, 0.2], [0.9, 0.1]),
    ("Weekend", [0.6, 0.4], [0.6, 0.4]),
]

for name, demand, pricing in seasons:
    circuit = quantum_revenue_management(demand, pricing)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 2: Guest experience
# ============================================

print("\n\nGuest Experience:")

    """Quantum guest experience optimization."""

    circuit = Circuit(num_qubits, name="QGuest")
    
    # Encode satisfaction
    circuit.ry(0, satisfaction[0] * math.pi)
    circuit.ry(1, satisfaction[1] * math.pi)
    
    # Encode preferences

    circuit.ry(3, preferences[1] * math.pi)
    
    # Experience optimization

    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

guest_types = [
    ("Business", [0.7, 0.3], [0.6, 0.4]),
    ("Leisure", [0.5, 0.5], [0.4, 0.6]),
    ("Family", [0.6, 0.4], [0.5, 0.5]),
    ("Luxury", [0.8, 0.2], [0.7, 0.3]),
]

for name, satisfaction, preferences in guest_types:
    circuit = quantum_guest_experience(satisfaction, preferences)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 3: Operations optimization
# ============================================

print("\n\nOperations Optimization:")
print("-" * 50)

def quantum_operations(staffing, maintenance):
    """Quantum hospitality operations."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QOperations")
    
    # Encode staffing
    circuit.ry(0, staffing[0] * math.pi)
    circuit.ry(1, staffing[1] * math.pi)
    
    # Encode maintenance
    circuit.ry(2, maintenance[0] * math.pi)
    circuit.ry(3, maintenance[1] * math.pi)
    
    # Operations optimization
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

operations = [
    ("Front Desk", [0.6, 0.4], [0.5, 0.5]),
    ("Housekeeping", [0.7, 0.3], [0.6, 0.4]),
    ("F&B", [0.5, 0.5], [0.4, 0.6]),
    ("Maintenance", [0.4, 0.6], [0.7, 0.3]),

for name, staffing, maintenance in operations:

    result = QuantumRun(circuit, shots=100)

print("\nApplications:")
print("  - Pricing strategy")
