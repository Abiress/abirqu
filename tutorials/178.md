```python
import math
import numpy as np
from abirqu import Circuit, H, X, CNOT, Ry, Rz, Measure
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Capital structure
# ============================================

print("Quantum Corporate Finance:")
print("=" * 50)

def quantum_capital_structure(debt, equity):
    """Quantum capital structure optimization."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QCapital")
    
    # Encode debt
    circuit.ry(0, debt[0] * math.pi)
    circuit.ry(1, debt[1] * math.pi)
    
    # Encode equity
    circuit.ry(2, equity[0] * math.pi)
    circuit.ry(3, equity[1] * math.pi)
    
    # Optimization
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure([2, 3])
    return circuit

print("Quantum Corporate Finance:")
print("-" * 50)

capital_scenarios = [
    ("Conservative", [0.2, 0.8], [0.8, 0.2]),
    ("Moderate", [0.4, 0.6], [0.6, 0.4]),
    ("Aggressive", [0.6, 0.4], [0.4, 0.6]),
    ("Highly Leveraged", [0.8, 0.2], [0.2, 0.8]),
]

for name, debt, equity in capital_scenarios:
    circuit = quantum_capital_structure(debt, equity)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 2: Dividend policy
# ============================================

print("\n\nDividend Policy:")
print("-" * 50)

def quantum_dividend_policy(profits, growth):
    """Quantum dividend policy."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QDividend")
    
    # Encode profits
    circuit.ry(0, profits[0] * math.pi)
    circuit.ry(1, profits[1] * math.pi)
    
    # Encode growth
    circuit.ry(2, growth[0] * math.pi)
    circuit.ry(3, growth[1] * math.pi)
    
    # Policy
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure([2])
    return circuit

dividend_policies = [
    ("High Yield", [0.8, 0.2], [0.3, 0.7]),
    ("Moderate", [0.5, 0.5], [0.5, 0.5]),
    ("Growth", [0.4, 0.6], [0.7, 0.3]),
    ("No Dividend", [0.2, 0.8], [0.9, 0.1]),
]

for name, profits, growth in dividend_policies:
    circuit = quantum_dividend_policy(profits, growth)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 3: Working capital
# ============================================

print("\n\nWorking Capital:")
print("-" * 50)

def quantum_working_capital(current, optimization):
    """Quantum working capital management."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QWorking")
    
    # Encode current assets/liabilities
    circuit.ry(0, current[0] * math.pi)
    circuit.ry(1, current[1] * math.pi)
    
    # Encode optimization
    circuit.ry(2, optimization[0] * math.pi)
    circuit.ry(3, optimization[1] * math.pi)
    
    # Management
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure([2, 3])
    return circuit

working_scenarios = [
    ("Efficient", [0.6, 0.4], [0.8, 0.2]),
    ("Balanced", [0.5, 0.5], [0.5, 0.5]),
    ("Conservative", [0.4, 0.6], [0.3, 0.7]),
    ("Aggressive", [0.7, 0.3], [0.7, 0.3]),
]

for name, current, optimization in working_scenarios:
    circuit = quantum_working_capital(current, optimization)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

print("\nApplications:")
print("  - Capital optimization")
print("  - Dividend decisions")
print("  - Cash management")
print("  - Financial planning")
```
