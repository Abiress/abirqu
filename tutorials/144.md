```python
import math
import numpy as np
from abirqu import Circuit, H, X, CNOT, Ry, Rz, Measure
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Property valuation
# ============================================

print("Quantum Real Estate:")
print("=" * 50)

def quantum_property_valuation(location, features):
    """Quantum property valuation."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QProperty")
    
    # Encode location
    circuit.ry(0, location[0] * math.pi)
    circuit.ry(1, location[1] * math.pi)
    
    # Encode features
    circuit.ry(2, features[0] * math.pi)
    circuit.ry(3, features[1] * math.pi)
    
    # Valuation model
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure([2, 3])
    return circuit

print("Quantum Real Estate:")
print("-" * 50)

properties = [
    ("Urban", [0.8, 0.2], [0.7, 0.3]),
    ("Suburban", [0.5, 0.5], [0.5, 0.5]),
    ("Rural", [0.2, 0.8], [0.3, 0.7]),
    ("Commercial", [0.7, 0.3], [0.8, 0.2]),
]

for name, location, features in properties:
    circuit = quantum_property_valuation(location, features)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 2: Market analysis
# ============================================

print("\n\nMarket Analysis:")
print("-" * 50)

def quantum_market_analysis(supply, demand, trends):
    """Quantum market analysis."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QMarket")
    
    # Encode supply
    circuit.ry(0, supply * math.pi)
    
    # Encode demand
    circuit.ry(1, demand * math.pi)
    
    # Encode trends
    circuit.ry(2, trends[0] * math.pi)
    circuit.ry(3, trends[1] * math.pi)
    
    # Analysis
    circuit.cnot(0, 1)
    circuit.cnot(1, 2)
    circuit.cnot(2, 3)
    
    circuit.measure([0, 1])
    return circuit

markets = [
    ("Hot", 0.3, 0.9, [0.8, 0.2]),
    ("Balanced", 0.5, 0.5, [0.5, 0.5]),
    ("Cold", 0.7, 0.3, [0.2, 0.8]),
    ("Bull", 0.2, 0.8, [0.9, 0.1]),
]

for name, supply, demand, trends in markets:
    circuit = quantum_market_analysis(supply, demand, trends)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 3: Investment optimization
# ============================================

print("\n\nInvestment Optimization:")
print("-" * 50)

def quantum_real_estate_investment(properties, budget):
    """Quantum real estate investment."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QInvestment")
    
    # Encode properties
    circuit.ry(0, properties[0] * math.pi)
    circuit.ry(1, properties[1] * math.pi)
    
    # Encode budget
    circuit.ry(2, budget[0] * math.pi)
    circuit.ry(3, budget[1] * math.pi)
    
    # Optimization
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure([2, 3])
    return circuit

investments = [
    ("Residential", [0.6, 0.4], [0.7, 0.3]),
    ("Commercial", [0.4, 0.6], [0.5, 0.5]),
    ("Mixed-Use", [0.5, 0.5], [0.6, 0.4]),
    ("Industrial", [0.3, 0.7], [0.4, 0.6]),
]

for name, properties, budget in investments:
    circuit = quantum_real_estate_investment(properties, budget)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

print("\nApplications:")
print("  - Property appraisal")
print("  - Market prediction")
print("  - Portfolio management")
print("  - Development planning")
```
