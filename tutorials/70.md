import math
from abirqu import Circuit, H, X, CNOT, Ry, Rz, Measure
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Quantum computing paradigms
# ============================================

print("Expert Quantum Computing:")
print("=" * 50)

paradigms = {
    'Gate Model': {
        'description': 'Sequential gate operations',
        'universality': 'Universal',
        'current_status': 'Most mature',
    },
    'Measurement-Based': {
        'description': 'Single-qubit measurements on cluster state',
        'universality': 'Universal',
        'current_status': 'Theoretical',
    },
    'Adiabatic': {
        'description': 'Slow Hamiltonian evolution',
        'universality': 'Universal',
        'current_status': 'D-Wave systems',
    },
    'Topological': {
        'description': 'Anyonic braiding',
        'universality': 'Universal',
        'current_status': 'Experimental',
    },
}

for paradigm, properties in paradigms.items():
    print(f"\n{paradigm}:")
    for prop, value in properties.items():
        print(f"  {prop}: {value}")

# ============================================
# Part 2: Quantum advantage
# ============================================

print("\n\nQuantum Advantage:")
print("-" * 50)

advantages = [
    ("Shor's Algorithm", "Integer factorization", "Exponential"),
    ("Grover's Search", "Unstructured search", "Quadratic"),
    ("HHL Algorithm", "Linear systems", "Exponential"),
    ("Quantum Simulation", "Many-body physics", "Exponential"),
    ("Quantum Machine Learning", "Certain ML tasks", "Polynomial-Exponential"),
]

for algorithm, problem, speedup in advantages:
    print(f"  {algorithm}: {problem} [{speedup} speedup]")

# ============================================
# Part 3: Quantum computing challenges
# ============================================

print("\n\nCurrent Challenges:")
print("-" * 50)

challenges = [
    ("Decoherence", "Quantum states decay quickly"),
    ("Error rates", "Gates are imperfect"),
    ("Scalability", "Adding qubits is hard"),
    ("Connectivity", "Limited qubit interactions"),
    ("Control", "Precise control needed"),
]

for challenge, description in challenges:
    print(f"  {challenge}: {description}")

# ============================================
# Part 4: Future outlook
# ============================================

print("\n\nFuture of Quantum Computing:")
print("-" * 50)

timeline = [
    ("2024-2025", "1000+ qubit processors"),
    ("2025-2027", "Early fault tolerance"),
    ("2027-2030", "10,000+ qubit systems"),
    ("2030-2035", "Full-scale quantum computers"),
    ("2035+", "Quantum internet and networks"),
]

for year, milestone in timeline:
    print(f"  {year}: {milestone}")

# Final demonstration
print("\n\nFinal Demonstration:")
print("-" * 50)

circuit = Circuit(3, name="FinalDemo")
circuit.h(0)
circuit.cnot(0, 1)
circuit.cnot(1, 2)
circuit.measure([0, 1, 2])

result = QuantumRun(circuit, shots=1000)
print(f"GHZ state: {result.counts}")
print(f"Total states: {sum(result.counts.values())}")

print("\nCongratulations! You've completed expert quantum computing tutorials!")

## Key Concepts

### Quantum Computing Paradigms
- **Gate Model**: Most common, universal
- **Measurement-Based**: Theoretical, alternative
- **Adiabatic**: Specialized, D-Wave
- **Topological**: Protected, experimental

### Quantum Advantage
- **Exponential**: Shor, simulation
- **Quadratic**: Grover, optimization
- **Polynomial**: Some ML tasks

### Current State
- NISQ era (Noisy Intermediate-Scale Quantum)
- 100-1000 qubits
- Limited error correction

### Future Directions
- Fault-tolerant quantum computing
- Quantum internet
- Quantum cloud services
