```python
import math
import numpy as np
from abirqu import Circuit, H, X, CNOT, Ry, Rz, Measure
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Brain connectivity
# ============================================

print("Quantum Neuroscience:")
print("=" * 50)

def quantum_brain_connectivity(regions, connections):
    """Quantum brain connectivity analysis."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QBrain")
    
    # Encode regions
    circuit.ry(0, regions[0] * math.pi)
    circuit.ry(1, regions[1] * math.pi)
    
    # Encode connections
    circuit.ry(2, connections[0] * math.pi)
    circuit.ry(3, connections[1] * math.pi)
    
    # Connectivity analysis
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure([2, 3])
    return circuit

print("Quantum Neuroscience:")
print("-" * 50)

brain_regions = [
    ("Frontal", [0.8, 0.2]),
    ("Parietal", [0.6, 0.4]),
    ("Temporal", [0.4, 0.6]),
    ("Occipital", [0.2, 0.8]),
]

connections = [0.7, 0.3]

for name, regions in brain_regions:
    circuit = quantum_brain_connectivity(regions, connections)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 2: Neural coding
# ============================================

print("\n\nNeural Coding:")
print("-" * 50)

def quantum_neural_coding(stimulus, neurons):
    """Quantum neural coding."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QNeuralCode")
    
    # Encode stimulus
    circuit.ry(0, stimulus[0] * math.pi)
    circuit.ry(1, stimulus[1] * math.pi)
    
    # Encode neurons
    circuit.ry(2, neurons[0] * math.pi)
    circuit.ry(3, neurons[1] * math.pi)
    
    # Coding
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    
    circuit.measure([2, 3])
    return circuit

stimuli = [
    ("Visual", [0.8, 0.2]),
    ("Auditory", [0.6, 0.4]),
    ("Tactile", [0.4, 0.6]),
    ("Olfactory", [0.2, 0.8]),
]

neurons = [0.5, 0.5]

for name, stimulus in stimuli:
    circuit = quantum_neural_coding(stimulus, neurons)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 3: Cognitive modeling
# ============================================

print("\n\nCognitive Modeling:")
print("-" * 50)

def quantum_cognitive_model(perception, memory, decision):
    """Quantum cognitive model."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QCognitive")
    
    # Encode perception
    circuit.ry(0, perception * math.pi)
    
    # Encode memory
    circuit.ry(1, memory[0] * math.pi)
    circuit.ry(2, memory[1] * math.pi)
    
    # Encode decision
    circuit.ry(3, decision * math.pi)
    
    # Cognitive process
    circuit.cnot(0, 1)
    circuit.cnot(1, 2)
    circuit.cnot(2, 3)
    
    circuit.measure([3])
    return circuit

cognitive_states = [
    ("Attention", 0.8, [0.6, 0.4], 0.7),
    ("Recall", 0.5, [0.8, 0.2], 0.5),
    ("Reasoning", 0.6, [0.5, 0.5], 0.6),
    ("Learning", 0.7, [0.3, 0.7], 0.8),
]

for name, perception, memory, decision in cognitive_states:
    circuit = quantum_cognitive_model(perception, memory, decision)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

print("\nApplications:")
print("  - Brain-computer interfaces")
print("  - Cognitive enhancement")
print("  - Neurological disorders")
print("  - Artificial intelligence")
```
