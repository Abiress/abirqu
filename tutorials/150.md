import math
import numpy as np
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Trend prediction
# ============================================

print("Quantum Fashion Technology:")
print("=" * 50)

def quantum_trend_prediction(history, social_media):
    """Quantum fashion trend prediction."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QTrend")
    
    # Encode history
    circuit.ry(0, history[0] * math.pi)
    circuit.ry(1, history[1] * math.pi)
    
    # Encode social media
    circuit.ry(2, social_media[0] * math.pi)
    circuit.ry(3, social_media[1] * math.pi)
    
    # Trend analysis
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

print("Quantum Fashion Technology:")
print("-" * 50)

fashion_categories = [
    ("Casual", [0.5, 0.5], [0.6, 0.4]),
    ("Formal", [0.7, 0.3], [0.4, 0.6]),
    ("Streetwear", [0.3, 0.7], [0.8, 0.2]),
    ("Luxury", [0.8, 0.2], [0.3, 0.7]),
]

for name, history, social_media in fashion_categories:
    circuit = quantum_trend_prediction(history, social_media)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 2: Size optimization
# ============================================

print("\n\nSize Optimization:")

    # Encode demographics
    circuit.ry(0, demographics[0] * math.pi)
    circuit.ry(1, demographics[1] * math.pi)
    
    # Encode preferences
    circuit.ry(2, preferences[0] * math.pi)
    circuit.ry(3, preferences[1] * math.pi)
    
    # Size optimization
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

sizes = [
    ("XS", [0.1, 0.9], [0.2, 0.8]),
    ("S", [0.3, 0.7], [0.3, 0.7]),
    ("M", [0.5, 0.5], [0.5, 0.5]),
    ("L", [0.7, 0.3], [0.7, 0.3]),
    ("XL", [0.9, 0.1], [0.8, 0.2]),
]

for name, demographics, preferences in sizes:
    circuit = quantum_size_optimization(demographics, preferences)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 3: Fashion supply chain
# ============================================

print("\n\nFashion Supply Chain:")
print("-" * 50)

def quantum_fashion_supply(production, distribution):
    """Quantum fashion supply chain."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QFashionSupply")
    
    # Encode production
    circuit.ry(0, production[0] * math.pi)
    circuit.ry(1, production[1] * math.pi)
    
    # Encode distribution
    circuit.ry(2, distribution[0] * math.pi)
    circuit.ry(3, distribution[1] * math.pi)
    
    # Supply optimization
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

supply_scenarios = [
    ("Fast Fashion", [0.8, 0.2], [0.7, 0.3]),
    ("Seasonal", [0.5, 0.5], [0.5, 0.5]),
    ("Sustainable", [0.3, 0.7], [0.4, 0.6]),

for name, production, distribution in supply_scenarios:

print("\nApplications:")
