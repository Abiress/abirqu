```python
import math
from abirqu import Circuit, H, X, Z, CNOT, Measure
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Surface code concept
# ============================================

print("Surface Codes:")
print("=" * 50)

# 4.8.8 Surface code (simplified)
print("Surface Code Properties:")
print("  Lattice: Planar graph with qubits on edges")
print("  Stabilizers: X and Z operators on faces/vertices")
print("  Distance: d (can correct ⌊(d-1)/2⌋ errors)")

# ============================================
# Part 2: Stabilizer measurement
# ============================================

print("\n\nStabilizer Measurement:")
print("-" * 50)

def surface_code_circuit(data_errors=None):
    """Build surface code syndrome measurement circuit."""
    # 9-qubit surface code (3x3 lattice)
    circuit = Circuit(9, name="SurfaceCode")
    
    # Data qubits in superposition
    for i in range(9):
        circuit.h(i)
    
    # Apply errors if specified
    if data_errors:
        for qubit in data_errors:
            circuit.x(qubit)
    
    # X stabilizers (face operators)
    circuit.cnot(0, 1)
    circuit.cnot(1, 2)
    circuit.cnot(3, 4)
    circuit.cnot(4, 5)
    
    # Z stabilizers (vertex operators)
    circuit.cnot(0, 3)
    circuit.cnot(1, 4)
    circuit.cnot(2, 5)
    
    # Measure syndrome qubits
    circuit.measure([6, 7, 8])
    
    return circuit

# Test with error on qubit 4
circuit = surface_code_circuit(data_errors=[4])
result = QuantumRun(circuit, shots=100)

print("Syndrome measurement results:")
print(f"  Counts: {result.counts}")
print("  Syndrome identifies error location!")

# ============================================
# Part 3: Decoding
# ============================================

print("\n\nError Decoding:")
print("-" * 50)

def decode_syndrome(syndrome):
    """Simple syndrome decoder."""
    # In real implementation, use minimum weight perfect matching
    # Here we show the concept
    
    syndrome_map = {
        '00': 'No error',
        '01': 'Error on qubit 1',
        '10': 'Error on qubit 2',
        '11': 'Error on qubit 3',
    }
    
    return syndrome_map.get(syndrome, 'Unknown error')

print("Decoding results:")
for syndrome in ['00', '01', '10', '11']:
    result = decode_syndrome(syndrome)
    print(f"  Syndrome {syndrome}: {result}")

print("\nSurface code threshold: ~1% physical error rate")
```

## Key Concepts

### Surface Code Structure
1. **Data qubits**: Store logical information
2. **Stabilizer qubits**: Detect errors
3. **X stabilizers**: Detect phase errors
4. **Z stabilizers**: Detect bit-flip errors

### Error Correction
- Syndrome measurement reveals error pattern
- Decoder finds most likely error
- Correction applied based on decoder output

### Applications
- **Fault-tolerant computing**: Reliable quantum computers
- **Quantum memories**: Long-term storage
- **Quantum networks**: Distributed computing
