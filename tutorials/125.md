import math
import numpy as np
from abirqu import Circuit, H, X, CNOT, Ry, Rz, Measure
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: SIR model
# ============================================

print("Quantum Epidemiology:")
print("=" * 50)

def quantum_sir_model(susceptible, infected, recovered, beta, gamma):
    """Quantum SIR model."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QSIR")
    
    # Encode compartments
    circuit.ry(0, susceptible * math.pi)
    circuit.ry(1, infected * math.pi)
    circuit.ry(2, recovered * math.pi)
    
    # Transmission
    circuit.cnot(0, 1)
    circuit.rz(1, beta)
    
    # Recovery
    circuit.cnot(1, 2)
    circuit.rz(2, gamma)
    
    circuit.measure([0, 1, 2])
    return circuit

print("Quantum Epidemiology:")
print("-" * 50)

compartments = [
    ("S", 0.9), ("I", 0.1), ("R", 0.0),
]

beta = 0.3
gamma = 0.1

circuit = quantum_sir_model(0.9, 0.1, 0.0, beta, gamma)
result = QuantumRun(circuit, shots=1000)
print(f"  SIR model: {result.counts}")

# ============================================
# Part 2: Contact tracing
# ============================================

print("\n\nContact Tracing:")
print("-" * 50)

def quantum_contact_tracing(infection_prob, contact_network):
    """Quantum contact tracing."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QTracing")
    
    # Encode infection probability
    circuit.ry(0, infection_prob * math.pi)
    
    # Encode contact network
    for i in range(3):
        circuit.ry(i + 1, contact_network[i] * math.pi)
    
    # Trace contacts
    circuit.cnot(0, 1)
    circuit.cnot(0, 2)
    circuit.cnot(0, 3)
    
    # Risk assessment
    circuit.rz(1, 0.5)
    circuit.rz(2, 0.3)
    circuit.rz(3, 0.2)
    
    circuit.measure([1, 2, 3])
    return circuit

infection_prob = 0.7
contact_network = [0.8, 0.5, 0.3]

circuit = quantum_contact_tracing(infection_prob, contact_network)
result = QuantumRun(circuit, shots=1000)
print(f"  Contact tracing: {result.counts}")

# ============================================
# Part 3: Intervention optimization
# ============================================

print("\n\nIntervention Optimization:")
print("-" * 50)

def quantum_intervention(coverage, efficacy, cost):
    """Quantum intervention optimization."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QIntervention")
    
    # Encode intervention parameters
    circuit.ry(0, coverage * math.pi)
    circuit.ry(1, efficacy * math.pi)
    circuit.ry(2, cost * math.pi)
    
    # Cost-benefit analysis
    circuit.cnot(0, 1)
    circuit.rz(1, 0.5)
    circuit.cnot(1, 2)
    
    circuit.measure([0, 1])
    return circuit

interventions = [
    ("Vaccination", 0.8, 0.9, 0.3),
    ("Masking", 0.6, 0.5, 0.1),
    ("Lockdown", 0.9, 0.8, 0.8),
    ("Testing", 0.7, 0.7, 0.4),
]

for name, coverage, efficacy, cost in interventions:
    circuit = quantum_intervention(coverage, efficacy, cost)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

print("\nApplications:")
print("  - Pandemic forecasting")
print("  - Resource allocation")
print("  - Policy evaluation")
print("  - Surveillance systems")
