```python
import math
import numpy as np
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Traffic optimization
# ============================================

print("Quantum Transportation:")
print("=" * 50)

def quantum_traffic_optimization(traffic_flow, signal_timing):
    """Quantum traffic signal optimization."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QTraffic")
    
    # Encode traffic flow
    for i in range(num_qubits):
        circuit.ry(i, traffic_flow[i] * math.pi)
    
    # Signal timing
    circuit.cnot(0, 1)
    circuit.cnot(2, 3)
    circuit.rz(1, signal_timing[0])
    circuit.rz(3, signal_timing[1])
    
    # Coordination
    circuit.cnot(1, 3)
    
    circuit.measure_all()
    return circuit

print("Quantum Transportation:")
print("-" * 50)

traffic_flows = [
    ("Morning Rush", [0.9, 0.3, 0.7, 0.2]),
    ("Evening Rush", [0.3, 0.9, 0.2, 0.7]),
    ("Night", [0.1, 0.1, 0.1, 0.1]),
]

signal_timing = [0.5, 0.5]

for name, flow in traffic_flows:
    circuit = quantum_traffic_optimization(flow, signal_timing)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 2: Fleet management
# ============================================

print("\n\nFleet Management:")
print("-" * 50)

def quantum_fleet_management(vehicles, routes, demands):
    """Quantum fleet management."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QFleet")
    
    # Encode vehicles
    circuit.ry(0, vehicles[0] * math.pi)
    circuit.ry(1, vehicles[1] * math.pi)
    
    # Encode routes
    circuit.ry(2, routes[0] * math.pi)
    circuit.ry(3, routes[1] * math.pi)
    
    # Assignment
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    
    # Optimization
    circuit.rz(2, demands[0])
    circuit.rz(3, demands[1])
    
    circuit.measure_all()
    return circuit

fleets = [
    ("Urban", [0.7, 0.3], [0.6, 0.4]),
    ("Highway", [0.3, 0.7], [0.4, 0.6]),
    ("Mixed", [0.5, 0.5], [0.5, 0.5]),
]

demands = [0.5, 0.5]

for name, vehicles, routes in fleets:
    circuit = quantum_fleet_management(vehicles, routes, demands)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 3: Autonomous vehicles
# ============================================

print("\n\nAutonomous Vehicles:")
print("-" * 50)

def quantum_autonomous(perception, planning, control):
    """Quantum autonomous vehicle control."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QAutonomous")
    
    # Encode perception
    circuit.ry(0, perception[0] * math.pi)
    circuit.ry(1, perception[1] * math.pi)
    
    # Encode planning
    circuit.ry(2, planning[0] * math.pi)
    circuit.ry(3, planning[1] * math.pi)
    
    # Decision making
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    
    # Control output
    circuit.rz(2, control[0])
    circuit.rz(3, control[1])
    
    circuit.measure_all()
    return circuit

scenarios = [
    ("Highway", [0.8, 0.2], [0.6, 0.4], [0.5, 0.5]),
    ("City", [0.3, 0.7], [0.4, 0.6], [0.3, 0.7]),
    ("Parking", [0.1, 0.9], [0.2, 0.8], [0.1, 0.9]),
]

for name, perception, planning, control in scenarios:
    circuit = quantum_autonomous(perception, planning, control)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

print("\nApplications:")
print("  - Traffic management")
print("  - Fleet optimization")
print("  - Autonomous driving")
print("  - Public transportation")
```
