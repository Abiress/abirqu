```python
import math
import numpy as np
from abirqu import Circuit, H, X, CNOT, Ry, Rz, Measure
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Ground state energy
# ============================================

print("Quantum Chemistry Algorithms:")
print("=" * 50)

def compute_ground_state(molecule='H2', bond_length=0.74):
    """Compute ground state energy of molecule."""
    # Simplified energies for demonstration
    energies = {
        'H2': -1.1745,
        'He': -2.9037,
        'LiH': -7.8630,
        'H2O': -75.7260,
    }
    return energies.get(molecule, -1.0)

print("Ground State Energies:")
for mol in ['H2', 'He', 'LiH', 'H2O']:
    energy = compute_ground_state(mol)
    print(f"  {mol}: {energy:.4f} Ha")

# ============================================
# Part 2: Excited states (VQD)
# ============================================

print("\n\nExcited States (VQD):")
print("-" * 50)

def vqd_circuit(num_qubits=2, state_index=0, params=None):
    """Build VQD circuit for excited states."""
    circuit = Circuit(num_qubits, name="VQD")
    
    if params is None:
        params = [0.1] * (num_qubits * 2)
    
    # Initial state
    for i in range(num_qubits):
        circuit.ry(i, params[i])
    
    # Variational layers
    for i in range(num_qubits - 1):
        circuit.cnot(i, i + 1)
    
    for i in range(num_qubits):
        circuit.ry(i, params[num_qubits + i])
    
    return circuit

# Compute ground and first excited state
print("Computing states:")
for state in ['Ground', '1st Excited', '2nd Excited']:
    energy = -1.1745 + state.count('Excited') * 0.5
    print(f"  {state}: {energy:.4f} Ha")

# ============================================
# Part 3: Reaction energy
# ============================================

print("\n\nReaction Energy:")
print("-" * 50)

def reaction_energy(reactants, products):
    """Compute reaction energy."""
    e_reactants = sum(compute_ground_state(r) for r in reactants)
    e_products = sum(compute_ground_state(p) for p in products)
    return e_products - e_reactants

# H2 + 1/2 O2 → H2O
delta_e = reaction_energy(['H2', 'H2'], ['H2O'])
print(f"H2 + H2 → H2O: ΔE = {delta_e:.4f} Ha")

# ============================================
# Part 4: Molecular properties
# ============================================

print("\n\nMolecular Properties:")
print("-" * 50)

def compute_dipole(molecule='H2'):
    """Compute dipole moment."""
    dipoles = {'H2': 0.0, 'H2O': 1.85, 'LiH': 5.88}
    return dipoles.get(molecule, 0.0)

def compute_polarizability(molecule='H2'):
    """Compute polarizability."""
    pols = {'H2': 0.80, 'H2O': 1.45, 'LiH': 20.0}
    return pols.get(molecule, 1.0)

print("Properties:")
for mol in ['H2', 'H2O', 'LiH']:
    dipole = compute_dipole(mol)
    pol = compute_polarizability(mol)
    print(f"  {mol}: Dipole={dipole:.2f} D, Pol={pol:.2f} Å³")
```

## Key Concepts

### VQD (Variational Quantum Deflation)
- Finds excited states by adding penalty terms
- Ensures orthogonality to lower states
- Can find multiple states sequentially

### Reaction Energy
- Difference between product and reactant energies
- Predicts reaction feasibility
- Key for catalyst design

### Molecular Properties
- **Dipole moment**: Charge distribution
- **Polarizability**: Response to electric field
- **Force constants**: Bond stiffness

### Applications
- **Drug design**: Binding energies
- **Materials**: Band gaps
- **Catalysis**: Reaction barriers
- **Spectroscopy**: Excitation energies
