```python
import math
import numpy as np
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: DNA sequence alignment
# ============================================

print("Quantum Genomics:")
print("=" * 50)

def quantum_dna_alignment(sequence_a, sequence_b):
    """Quantum DNA sequence alignment."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QDNA")
    
    # Encode sequence A
    for i in range(2):
        circuit.ry(i, sequence_a[i] * math.pi)
    
    # Encode sequence B
    for i in range(2):
        circuit.ry(i + 2, sequence_b[i] * math.pi)
    
    # Alignment scoring
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    # Gap penalty
    circuit.cnot(2, 3)
    
    circuit.measure_all()
    return circuit

print("Quantum Genomics:")
print("-" * 50)

sequences_a = [
    ("Gene1", [0.2, 0.8]),
    ("Gene2", [0.5, 0.5]),
    ("Gene3", [0.9, 0.1]),
]

sequences_b = [
    ("Gene1", [0.3, 0.7]),
    ("Gene2", [0.4, 0.6]),
    ("Gene3", [0.8, 0.2]),
]

for name_a, seq_a in sequences_a:
    for name_b, seq_b in sequences_b:
        circuit = quantum_dna_alignment(seq_a, seq_b)
        result = QuantumRun(circuit, shots=100)
        print(f"  {name_a} vs {name_b}: {result.counts}")

# ============================================
# Part 2: Gene expression analysis
# ============================================

print("\n\nGene Expression Analysis:")
print("-" * 50)

def quantum_gene_expression(expression_levels, condition):
    """Quantum differential expression."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QGeneExpr")
    
    # Encode expression levels
    for i in range(num_qubits):
        circuit.ry(i, expression_levels[i] * math.pi)
    
    # Condition encoding
    circuit.ry(0, condition * math.pi)
    
    # Differential analysis
    circuit.cnot(0, 1)
    circuit.cnot(2, 3)
    circuit.rz(1, condition * 0.5)
    circuit.rz(3, (1 - condition) * 0.5)
    
    circuit.measure_all()
    return circuit

genes = [
    ("TP53", [0.8, 0.2, 0.6, 0.4]),
    ("BRCA1", [0.3, 0.7, 0.5, 0.5]),
    ("MYC", [0.6, 0.4, 0.8, 0.2]),
]

conditions = [("Normal", 0.0), ("Disease", 1.0)]

for gene_name, expression in genes:
    for cond_name, condition in conditions:
        circuit = quantum_gene_expression(expression, condition)
        result = QuantumRun(circuit, shots=100)
        print(f"  {gene_name} ({cond_name}): {result.counts}")

# ============================================
# Part 3: Variant calling
# ============================================

print("\n\nVariant Calling:")
print("-" * 50)

def quantum_variant_calling(reads, reference):
    """Quantum variant calling."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QVariant")
    
    # Encode reference
    for i in range(2):
        circuit.ry(i, reference[i] * math.pi)
    
    # Encode reads
    for i in range(2):
        circuit.ry(i + 2, reads[i] * math.pi)
    
    # Variant detection
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    
    # Quality scoring
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure_all()
    return circuit

reference = [0.5, 0.5]
reads = [
    ([0.5, 0.5], "Ref"),
    ([0.4, 0.6], "SNP"),
    ([0.5, 0.5], "Ref"),
    ([0.3, 0.7], "Indel"),
]

for read, variant_type in reads:
    circuit = quantum_variant_calling(read, reference)
    result = QuantumRun(circuit, shots=100)
    print(f"  {variant_type}: {result.counts}")

print("\nApplications:")
print("  - Sequence alignment")
print("  - Gene discovery")
print("  - Disease diagnosis")
print("  - Personalized medicine")
```
