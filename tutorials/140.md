```python
import math
import numpy as np
from abirqu import Circuit, H, X, CNOT, Ry, Rz, Measure
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Urban planning
# ============================================

print("Quantum Smart Cities:")
print("=" * 50)

def quantum_urban_planning(population, infrastructure):
    """Quantum urban planning."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QUrban")
    
    # Encode population
    circuit.ry(0, population[0] * math.pi)
    circuit.ry(1, population[1] * math.pi)
    
    # Encode infrastructure
    circuit.ry(2, infrastructure[0] * math.pi)
    circuit.ry(3, infrastructure[1] * math.pi)
    
    # Planning optimization
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure([2, 3])
    return circuit

print("Quantum Smart Cities:")
print("-" * 50)

cities = [
    ("Downtown", [0.9, 0.1], [0.8, 0.2]),
    ("Suburban", [0.5, 0.5], [0.5, 0.5]),
    ("Industrial", [0.3, 0.7], [0.4, 0.6]),
    ("Rural", [0.1, 0.9], [0.2, 0.8]),
]

for name, population, infrastructure in cities:
    circuit = quantum_urban_planning(population, infrastructure)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 2: Resource management
# ============================================

print("\n\nResource Management:")
print("-" * 50)

def quantum_resource_management(water, energy, waste):
    """Quantum resource management."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QResource")
    
    # Encode water
    circuit.ry(0, water * math.pi)
    
    # Encode energy
    circuit.ry(1, energy[0] * math.pi)
    circuit.ry(2, energy[1] * math.pi)
    
    # Encode waste
    circuit.ry(3, waste * math.pi)
    
    # Management optimization
    circuit.cnot(0, 1)
    circuit.cnot(1, 2)
    circuit.cnot(2, 3)
    
    circuit.measure([0, 1])
    return circuit

resources = [
    ("Residential", 0.6, [0.7, 0.3], 0.4),
    ("Commercial", 0.4, [0.5, 0.5], 0.6),
    ("Industrial", 0.3, [0.4, 0.6], 0.8),
    ("Mixed Use", 0.5, [0.6, 0.4], 0.5),
]

for name, water, energy, waste in resources:
    circuit = quantum_resource_management(water, energy, waste)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 3: Infrastructure optimization
# ============================================

print("\n\nInfrastructure Optimization:")
print("-" * 50)

def quantum_infrastructure(transportation, communication, utilities):
    """Quantum infrastructure optimization."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QInfra")
    
    # Encode transportation
    circuit.ry(0, transportation * math.pi)
    
    # Encode communication
    circuit.ry(1, communication[0] * math.pi)
    circuit.ry(2, communication[1] * math.pi)
    
    # Encode utilities
    circuit.ry(3, utilities * math.pi)
    
    # Optimization
    circuit.cnot(0, 1)
    circuit.cnot(1, 2)
    circuit.cnot(2, 3)
    
    circuit.measure([1, 2])
    return circuit

infra_types = [
    ("Smart Grid", 0.8, [0.7, 0.3], 0.6),
    ("IoT Network", 0.6, [0.5, 0.5], 0.4),
    ("5G Coverage", 0.7, [0.6, 0.4], 0.5),
    ("EV Charging", 0.5, [0.4, 0.6], 0.7),
]

for name, transportation, communication, utilities in infra_types:
    circuit = quantum_infrastructure(transportation, communication, utilities)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

print("\nApplications:")
print("  - Urban development")
print("  - Resource efficiency")
print("  - Infrastructure planning")
print("  - Sustainability")
```
