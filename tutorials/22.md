## Step-by-Step Code

```python
import math
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Quantum Generator
# ============================================

print("Quantum GAN:")
print("=" * 50)

def quantum_generator(noise, params):
    """Generate quantum state from noise."""
    circuit = Circuit(2, name="Generator")
    
    # Encode noise
    for i, n in enumerate(noise):
        circuit.ry(i, n * math.pi)
    
    # Parameterized generation layer
    circuit.ry(0, params[0])
    circuit.ry(1, params[1])
    circuit.cnot(0, 1)
    circuit.ry(0, params[2])
    
    circuit.measure_all()
    return circuit

def quantum_discriminator(data, params):
    """Discriminate between real and fake data."""
    circuit = Circuit(2, name="Discriminator")
    
    # Encode data
    circuit.ry(0, data[0])
    circuit.ry(1, data[1])
    
    # Parameterized discrimination layer
    circuit.ry(0, params[0])
    circuit.cnot(0, 1)
    circuit.ry(1, params[1])
    
    circuit.measure_all()
    return circuit

# ============================================
# Part 2: Training
# ============================================

print("\nTraining QGAN:")
print("-" * 50)

# Real data distribution (simple)
real_data = [[0.5, 0.5], [0.6, 0.4], [0.4, 0.6]]

# Initialize parameters
gen_params = [0.0, 0.0, 0.0]
disc_params = [0.0, 0.0]

for epoch in range(20):
    # Generate fake data
    import random
    noise = [random.random() for _ in range(2)]
    gen_circuit = quantum_generator(noise, gen_params)
    gen_result = QuantumRun(gen_circuit, shots=100)
    
    # Get generated sample
    most_common = max(gen_result.counts, key=gen_result.counts.get)
    fake_data = [int(most_common[0]) * 0.5, int(most_common[1]) * 0.5]
    
    # Train discriminator
    real_sample = random.choice(real_data)
    
    # Real sample
    disc_circuit_real = quantum_discriminator(real_sample, disc_params)
    disc_result_real = QuantumRun(disc_circuit_real, shots=100)
    real_score = disc_result_real.probabilities.get('1', 0.5)
    
    # Fake sample
    disc_circuit_fake = quantum_discriminator(fake_data, disc_params)
    disc_result_fake = QuantumRun(disc_circuit_fake, shots=100)
    fake_score = disc_result_fake.probabilities.get('1', 0.5)
    
    # Update discriminator
    disc_params[0] += 0.1 * (real_score - fake_score)
    disc_params[1] += 0.1 * (real_score - fake_score)
    
    # Update generator (fool discriminator)
    gen_params[0] += 0.05 * (fake_score - 0.5)
    gen_params[1] += 0.05 * (fake_score - 0.5)
    gen_params[2] += 0.05 * (fake_score - 0.5)
    
    if (epoch + 1) % 5 == 0:
        print(f"  Epoch {epoch+1}: Real={real_score:.3f}, Fake={fake_score:.3f}")

# ============================================
# Part 3: Generation
# ============================================

print("\n\nGenerated samples:")
print("=" * 50)

for i in range(5):
    noise = [random.random() for _ in range(2)]
    gen_circuit = quantum_generator(noise, gen_params)
    result = QuantumRun(gen_circuit, shots=100)
    sample = max(result.counts, key=result.counts.get)
    print(f"  Sample {i+1}: |{sample}⟩")
```

## Key Concepts

### Quantum GAN Architecture
1. **Generator**: Maps noise to quantum states
2. **Discriminator**: Distinguishes real from generated
3. **Adversarial training**: Both improve simultaneously

### Quantum Advantages
- Exponentially large state space
- Natural for quantum data
- Potential speedup in training

### Applications
- **Data augmentation**: Generate training data
- **Quantum simulation**: Learn quantum states
- **Image generation**: Quantum image synthesis
- **Drug discovery**: Molecular generation
