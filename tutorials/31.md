import math
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Zero-noise extrapolation
# ============================================

print("Quantum Error Mitigation:")
print("=" * 50)

def noisy_circuit(noise_level=0.1):
    """Circuit with adjustable noise."""
    circuit = Circuit(2, name="Noisy")
    circuit.h(0)
    circuit.cnot(0, 1)
    
    # Add noise (simplified)
    if noise_level > 0:
        circuit.rz(0, noise_level)
        circuit.rz(1, noise_level)
    
    circuit.measure_all()
    return circuit

print("Zero-Noise Extrapolation:")
print("-" * 50)

# Run at different noise levels
noise_levels = [0.0, 0.05, 0.1, 0.15, 0.2]
results = {}

for noise in noise_levels:
    circuit = noisy_circuit(noise)
    result = QuantumRun(circuit, shots=1000)
    fidelity = result.probabilities.get('00', 0) + result.probabilities.get('11', 0)
    results[noise] = fidelity
    print(f"  Noise={noise:.2f}: Fidelity={fidelity:.4f}")

# Extrapolate to zero noise
# Linear fit: fidelity = a - b * noise
import numpy as np
x = np.array(noise_levels)
y = np.array([results[n] for n in noise_levels])
coeffs = np.polyfit(x, y, 1)
zero_noise_fidelity = coeffs[1]  # Intercept

print(f"\n  Extrapolated zero-noise fidelity: {zero_noise_fidelity:.4f}")

# ============================================
# Part 2: Measurement error mitigation
# ============================================

print("\n\nMeasurement Error Mitigation:")

def build_calibration_circuit(num_qubits=2):

    return circuit

    calibration[state] = result.counts
    print(f"  |{state}⟩ → {result.counts}")

## Key Concepts

### Error Mitigation vs Error Correction
- **Error correction**: Requires additional qubits (overhead)
- **Error mitigation**: No additional qubits (classical post-processing)

### Techniques
1. **ZNE**: Run at multiple noise levels, extrapolate
2. **PEC**: Represent noise as quasi-probability distribution
3. **Readout mitigation**: Calibrate and invert measurement errors

### Applications
- **NISQ devices**: Get useful results from noisy hardware
- **VQE**: Improve energy estimates
- **QAOA**: Better optimization results
