```python
import math
import numpy as np
from abirqu import Circuit, H, X, CNOT, Ry, Rz, Measure
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Crop optimization
# ============================================

print("Quantum Agriculture:")
print("=" * 50)

def quantum_crop_optimization(soil, water, nutrients):
    """Quantum crop optimization."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QCrop")
    
    # Encode soil
    circuit.ry(0, soil[0] * math.pi)
    circuit.ry(1, soil[1] * math.pi)
    
    # Encode water
    circuit.ry(2, water * math.pi)
    
    # Encode nutrients
    circuit.ry(3, nutrients * math.pi)
    
    # Optimization
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure([2, 3])
    return circuit

print("Quantum Agriculture:")
print("-" * 50)

crops = [
    ("Wheat", [0.6, 0.4], 0.5, 0.6),
    ("Rice", [0.8, 0.2], 0.9, 0.5),
    ("Corn", [0.5, 0.5], 0.6, 0.7),
    ("Soybean", [0.4, 0.6], 0.4, 0.8),
]

for name, soil, water, nutrients in crops:
    circuit = quantum_crop_optimization(soil, water, nutrients)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 2: Pest detection
# ============================================

print("\n\nPest Detection:")
print("-" * 50)

def quantum_pest_detection(image_features, params):
    """Quantum pest detection."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QPest")
    
    # Encode image features
    for i in range(num_qubits):
        circuit.ry(i, image_features[i] * math.pi)
    
    # Feature extraction
    circuit.cnot(0, 1)
    circuit.cnot(2, 3)
    circuit.cnot(1, 3)
    
    # Classification
    for i in range(num_qubits):
        circuit.ry(i, params[i])
    
    circuit.measure([0, 1])
    return circuit

pest_images = [
    ("Healthy", [0.2, 0.3, 0.4, 0.5]),
    ("Aphids", [0.8, 0.2, 0.6, 0.4]),
    ("Blight", [0.3, 0.9, 0.2, 0.7]),
    ("Worm", [0.5, 0.4, 0.8, 0.3]),
]

params = [0.5, 0.5, 0.5, 0.5]

for name, features in pest_images:
    circuit = quantum_pest_detection(features, params)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

# ============================================
# Part 3: Yield prediction
# ============================================

print("\n\nYield Prediction:")
print("-" * 50)

def quantum_yield_prediction(weather, soil, management):
    """Quantum yield prediction."""
    num_qubits = 4
    circuit = Circuit(num_qubits, name="QYield")
    
    # Encode weather
    circuit.ry(0, weather * math.pi)
    
    # Encode soil
    circuit.ry(1, soil * math.pi)
    
    # Encode management
    circuit.ry(2, management[0] * math.pi)
    circuit.ry(3, management[1] * math.pi)
    
    # Prediction
    circuit.cnot(0, 2)
    circuit.cnot(1, 3)
    circuit.rz(2, 0.5)
    circuit.rz(3, 0.5)
    
    circuit.measure([2])
    return circuit

conditions = [
    ("Optimal", 0.8, 0.7, [0.9, 0.8]),
    ("Average", 0.5, 0.5, [0.5, 0.5]),
    ("Poor", 0.2, 0.3, [0.3, 0.2]),
]

for name, weather, soil, management in conditions:
    circuit = quantum_yield_prediction(weather, soil, management)
    result = QuantumRun(circuit, shots=100)
    print(f"  {name}: {result.counts}")

print("\nApplications:")
print("  - Precision agriculture")
print("  - Crop monitoring")
print("  - Resource management")
print("  - Food security")
```
