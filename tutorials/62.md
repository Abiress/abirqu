import math
import numpy as np
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Quantum CNN
# ============================================

print("Advanced Quantum Computer Vision:")
print("=" * 50)

def quantum_cnn_layer(num_qubits=4, kernel_size=2):
    """Build quantum CNN layer."""
    circuit = Circuit(num_qubits, name="QCNN")
    
    # Convolution (entanglement)
    for i in range(num_qubits - kernel_size + 1):
        for j in range(kernel_size):
            circuit.cnot(i, i + j)
    
    # Pooling (measurement-based)
    for i in range(0, num_qubits, 2):
        circuit.ry(i, 0.3)
    
    return circuit

print("Quantum CNN Architecture:")
print("-" * 50)

# Build QCNN
circuit = Circuit(8, name="QCNN_Full")

# Layer 1
for i in range(8):
    circuit.ry(i, 0.5)
for i in range(7):
    circuit.cnot(i, i + 1)

# Layer 2
for i in range(0, 8, 2):
    circuit.ry(i, 0.3)
for i in range(0, 6, 2):
    circuit.cnot(i, i + 2)

# Layer 3
circuit.ry(0, 0.2)
circuit.ry(4, 0.2)
circuit.cnot(0, 4)

print(f"  Total qubits: {8}")
print(f"  Circuit depth: {circuit.depth()}")

# ============================================
# Part 2: Quantum object detection
# ============================================

print("\n\nQuantum Object Detection:")
print("-" * 50)

def quantum_object_detector(image_region, params):
    """Detect objects in image region."""
    circuit = Circuit(4, name="ObjDetect")
    
    # Encode image region
    for i in range(4):
        circuit.ry(i, image_region[i] * math.pi)
    
    # Detection layers
    circuit.cnot(0, 1)
    circuit.cnot(2, 3)
    circuit.cnot(1, 3)
    
    # Classification
    circuit.ry(0, params[0])
    circuit.ry(3, params[1])
    
    circuit.measure_all()
    return circuit

# Test detection
image_region = [0.2, 0.8, 0.3, 0.7]
params = [0.5, 0.5]

circuit = quantum_object_detector(image_region, params)
result = QuantumRun(circuit, shots=100)

print(f"Detection result: {result.counts}")

# ============================================
# Part 3: Quantum image generation
# ============================================

print("\n\nQuantum Image Generation:")
print("-" * 50)

def quantum_image_generator(params, num_qubits=4):
    """Generate quantum image."""
    circuit = Circuit(num_qubits, name="QImgGen")
    
    # Initialize
    for i in range(num_qubits):
        circuit.h(i)
    
    # Generation layers
    for layer in range(3):
        for i in range(num_qubits):
            circuit.ry(i, params[layer * num_qubits + i])
        
        for i in range(num_qubits - 1):
            circuit.cnot(i, i + 1)
    
    circuit.measure_all()
    return circuit

# Train generator
params = np.random.uniform(0, math.pi, 12)
circuit = quantum_image_generator(params)
result = QuantumRun(circuit, shots=1000)

print(f"Generated image distribution: {len(result.counts)} unique pixels")

print("\nApplications:")
print("  - Medical imaging")
print("  - Autonomous vehicles")
print("  - Surveillance")
print("  - Augmented reality")
