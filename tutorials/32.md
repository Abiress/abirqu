```python
import math
import random
from abirqu import Circuit, H, X, Y, Z, CNOT, Rx, Ry, Rz, Measure
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Randomized benchmarking
# ============================================

print("Quantum Benchmarking:")
print("=" * 50)

def random_clifford(num_qubits=1, depth=10):
    """Generate random Clifford circuit."""
    circuit = Circuit(num_qubits, name="RandomClifford")
    
    gates = [H, X, Y, Z]
    
    for _ in range(depth):
        gate = random.choice(gates)
        if gate == H:
            circuit.h(0)
        elif gate == X:
            circuit.x(0)
        elif gate == Y:
            circuit.y(0)
        elif gate == Z:
            circuit.z(0)
    
    return circuit

def inverse_clifford(circuit):
    """Compute inverse of Clifford circuit (simplified)."""
    # In real implementation, compute inverse via group theory
    # Here we just create a simple inverse
    inv = Circuit(circuit.num_qubits, name="Inverse")
    for _ in range(circuit.depth):
        inv.z(0)  # Simplified inverse
    return inv

print("Randomized Benchmarking:")
print("-" * 50)

depths = [1, 2, 4, 8, 16, 32]
fidelities = []

for depth in depths:
    # Run multiple random circuits
    successes = 0
    trials = 20
    
    for _ in range(trials):
        # Random Clifford
        circuit = random_clifford(depth=depth)
        
        # Add inverse
        inv = inverse_clifford(circuit)
        
        # Combine
        full_circuit = Circuit(1, name="RB")
        full_circuit.h(0)  # Initial state
        full_circuit.compose(circuit)
        full_circuit.compose(inv)
        full_circuit.measure([0])
        
        result = QuantumRun(full_circuit, shots=100)
        if result.probabilities.get('0', 0) > 0.5:
            successes += 1
    
    fidelity = successes / trials
    fidelities.append(fidelity)
    print(f"  Depth {depth:2d}: Fidelity = {fidelity:.3f}")

# ============================================
# Part 2: Gate fidelity estimation
# ============================================

print("\n\nGate Fidelity Estimation:")
print("-" * 50)

def estimate_single_qubit_fidelity(num_tests=100):
    """Estimate average single-qubit gate fidelity."""
    correct = 0
    
    for _ in range(num_tests):
        # Random gate
        gate_type = random.choice(['H', 'X', 'Y', 'Z', 'I'])
        
        circuit = Circuit(1, name="FidelityTest")
        
        if gate_type == 'H':
            circuit.h(0)
        elif gate_type == 'X':
            circuit.x(0)
        elif gate_type == 'Y':
            circuit.y(0)
        elif gate_type == 'Z':
            circuit.z(0)
        # I does nothing
        
        circuit.measure([0])
        result = QuantumRun(circuit, shots=100)
        
        # Check if result is correct (simplified)
        most_common = max(result.counts, key=result.counts.get)
        if most_common in ['0', '1']:
            correct += 1
    
    return correct / num_tests

fidelity = estimate_single_qubit_fidelity()
print(f"  Single-qubit gate fidelity: {fidelity:.4f}")

# ============================================
# Part 3: Cross-entropy benchmarking
# ============================================

print("\n\nCross-Entropy Benchmarking:")
print("-" * 50)

def cross_entropy_benchmark(num_qubits=3, depth=10):
    """Perform cross-entropy benchmarking."""
    # Generate random circuit
    circuit = Circuit(num_qubits, name="XEB")
    
    for _ in range(depth):
        # Random single-qubit gates
        for i in range(num_qubits):
            gate = random.choice(['H', 'X', 'Y', 'Z'])
            if gate == 'H':
                circuit.h(i)
            elif gate == 'X':
                circuit.x(i)
            elif gate == 'Y':
                circuit.y(i)
            elif gate == 'Z':
                circuit.z(i)
        
        # Random two-qubit gates
        if random.random() > 0.5:
            circuit.cnot(0, 1)
    
    circuit.measure(list(range(num_qubits)))
    
    # Run
    result = QuantumRun(circuit, shots=1000)
    
    # Compute cross-entropy (simplified)
    ideal_dist = {format(i, f'0{num_qubits}b'): 1/2**num_qubits 
                  for i in range(2**num_qubits)}
    
    experimental_dist = {k: v/1000 for k, v in result.counts.items()}
    
    # Fidelity from cross-entropy
    fidelity = sum(ideal_dist.get(k, 0) * v for k, v in experimental_dist.items())
    
    return fidelity

fidelity = cross_entropy_benchmark()
print(f"  XEB fidelity: {fidelity:.4f}")
```

## Key Concepts

### Benchmarking Methods
1. **Randomized Benchmarking**: Average gate fidelity via random sequences
2. **Gate Set Tomography**: Complete characterization of gate set
3. **Cross-Entropy Benchmarking**: For large circuits

### Metrics
- **Average gate fidelity**: How accurate are gates
- **Coherence time**: How long quantum states last
- **Circuit depth**: How deep circuits can be

### Applications
- **Hardware validation**: Test quantum computers
- **Quality control**: Compare devices
- **Optimization**: Improve gate parameters
