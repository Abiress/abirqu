```python
import math
from abirqu import Circuit
from abirqu.primitives import QuantumRun

# ============================================
# Part 1: Quantum architectures
# ============================================

print("Expert Quantum Computing (Part 3):")
print("=" * 50)

architectures = {
    'Superconducting': {
        'qubits': '~1000',
        'gate_time': '~10-100 ns',
        'coherence': '~100 μs',
        'connectivity': 'Nearest-neighbor',
        'companies': 'IBM, Google, Rigetti',
    },
    'Trapped Ion': {
        'qubits': '~32',
        'gate_time': '~1-10 μs',
        'coherence': '~1-10 s',
        'connectivity': 'All-to-all',
        'companies': 'IonQ, Quantinuum',
    },
    'Photonic': {
        'qubits': '~200',
        'gate_time': '~1 ns',
        'coherence': 'N/A',
        'connectivity': 'Programmable',
        'companies': 'Xanadu, PsiQuantum',
    },
    'Topological': {
        'qubits': '~8',
        'gate_time': 'TBD',
        'coherence': 'Topologically protected',
        'connectivity': 'TBD',
        'companies': 'Microsoft',
    },
}

for arch, specs in architectures.items():
    print(f"\n{arch}:")
    for spec, value in specs.items():
        print(f"  {spec}: {value}")

# ============================================
# Part 2: Quantum hardware
# ============================================

print("\n\nQuantum Hardware Challenges:")
print("-" * 50)

challenges = [
    ("Decoherence", "Quantum states decay quickly"),
    ("Error rates", "Gates are imperfect"),
    ("Scalability", "Adding qubits is hard"),
    ("Connectivity", "Limited qubit interactions"),
    ("Control", "Precise control needed"),
    ("Cryogenics", "Superconducting needs mK temperatures"),
]

for challenge, description in challenges:
    print(f"  {challenge:15s}: {description}")

# ============================================
# Part 3: Quantum software
# ============================================

print("\n\nQuantum Software Stack:")
print("-" * 50)

stack = [
    ("Applications", "Qiskit, Cirq, PennyLane"),
    ("Compilers", "Tket, Qiskit Transpiler"),
    ("Optimizers", "Gate fusion, circuit optimization"),
    ("Error mitigation", "Zero-noise extrapolation, PEC"),
    ("Hardware control", "Pulse scheduling, calibration"),
]

for layer, tools in stack:
    print(f"  {layer:20s}: {tools}")

# Final demonstration
print("\n\nFinal Demonstration:")
print("-" * 50)

circuit = Circuit(3, name="Final")
circuit.h(0)
circuit.cnot(0, 1)
circuit.cnot(1, 2)
circuit.measure_all()

result = QuantumRun(circuit, shots=1000)
print(f"GHZ state: {result.counts}")
```

## Key Concepts

### Quantum Architectures
- **Superconducting**: Fast, scalable, needs cooling
- **Trapped Ion**: High fidelity, slow gates

- **Topological**: Error-resistant, experimental

- **Decoherence**: Quantum states decay
- **Errors**: Gates are imperfect
- **Scalability**: Adding qubits is hard

- **Applications**: High-level programming
